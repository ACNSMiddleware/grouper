103   /**
104    * Find the subjects in the ldap server.
105    * 
106    * If account-creation is enabled with createMissingAccounts, this will create missing entries.
107    * @param subjectsToFetch
108    * @return
109    */
110   protected Map<Subject, LdapUser> fetchTargetSystemUsers( Collection<Subject> subjectsToFetch) 
111       throws PspException {
112     LOG.info("Fetching {} users from target system", subjectsToFetch.size());
113     
114     if ( subjectsToFetch.size() > config.getUserSearch_batchSize() )
115       throw new IllegalArgumentException("LdapProvisioner.fetchTargetSystemUsers: invoked with too many subjects to fetch");
116     
117     StringBuilder combinedLdapFilter = new StringBuilder();
118     
119     // Start the combined ldap filter as an OR-query
120     combinedLdapFilter.append("(|");
121     
122     for ( Subject subject : subjectsToFetch ) {
123       SearchFilter f = getUserLdapFilter(subject);
124       
125       String filterString = f.format();
126       
127       // Wrap the subject's filter in (...) if it doesn't start with (
128       if ( filterString.startsWith("(") )
129         combinedLdapFilter.append(filterString);
130       else
131         combinedLdapFilter.append('(').append(filterString).append(')');
132     }
133     combinedLdapFilter.append(')');
134 
135     // Actually do the search
136     List<LdapObject> searchResult;
137     
138     try {
139       searchResult = performLdapSearchRequest(
140         new SearchRequest(config.getUserSearchBaseDn(), 
141               combinedLdapFilter.toString(), 
142               config.getUserSearchAttributes()));
143       
144       LOG.info("Read {} user objects from directory", searchResult.size());
145     }
146     catch (PspException e) {
147       LOG.error("Problem searching for subjects with filter {}", combinedLdapFilter);
148       throw e;
149     }
150     
151     // Now we have a bag of LdapObjects, but we don't know which goes with which subject.
152     // Generally, we're going to go through the Subjects and their filters and compare
153     // them to the Ldap data we've fetched into memory.
154     // 
155     // This is complicated a bit because Ldaptive doesn't have a way to run a filter in memory
156     // against an LdapObject. Therefore, we're going to use unboundid classes to do
157     // some of this work.
158     Map<Subject, LdapUser> result = new HashMap<Subject, LdapUser>();
159 
160     Set<LdapObject> matchedFetchResults = new HashSet<LdapObject>();
161     
162     // For every subject we tried to bulk fetch, find the matching LdapObject that came back
163     for ( Subject subjectToFetch : subjectsToFetch ) {
164       SearchFilter f = getUserLdapFilter(subjectToFetch);
165           
166       for ( LdapObject aFetchedLdapObject : searchResult ) {
167         if ( aFetchedLdapObject.matchesLdapFilter(f)) {
168           result.put(subjectToFetch, new LdapUser(aFetchedLdapObject));
169           matchedFetchResults.add(aFetchedLdapObject);
170           break;
171         }
172       }
173     }
174 
175     Set<LdapObject> unmatchedFetchResults = new HashSet<LdapObject>(searchResult);
176     unmatchedFetchResults.removeAll(matchedFetchResults);
177     
178     for ( LdapObject unmatchedFetchResult : unmatchedFetchResults )
179       LOG.error("{}: User data from ldap server was not matched with a grouper subject "
180           + "(perhaps attributes are used in userSearchFilter ({}) that are not included "
181           + "in userSearchAttributes ({})?): {}",
182           new Object[] {getName(), config.getUserSearchFilter(), config.getUserSearchAttributes(), 
183           unmatchedFetchResult.getDn()});
184     
185     return result;
186   }

189   protected SearchFilter getUserLdapFilter(Subject subject) {
190     String result = evaluateJexlExpression(config.getUserSearchFilter(), subject, null);
191     if ( StringUtils.isEmpty(result) )
192       throw new RuntimeException("User searching requires userSearchFilter to be configured correctly");
193     
194     // If the filter contains '||', then this filter is requesting parameter substitution
195     String filterPieces[] = result.split("\\|\\|");
196     SearchFilter filter = new SearchFilter(filterPieces[0]);
197     for (int i=1; i<filterPieces.length; i++)
198       filter.setParameter(i-1, filterPieces[i].trim());
199     
200     LOG.debug("{}: User LDAP filter for subject {}: {}",
201         new Object[]{getName(), subject.getId(), filter});
202     return filter;
203   }

205   @Override
206   protected LdapUser createUser(Subject personSubject) throws PspException {
207     GrouperUtil.assertion(config.isCreatingMissingUsersEnabled(), "Can't create users unless createMissingUsers is enabled");
208     GrouperUtil.assertion(StringUtils.isNotEmpty(config.getUserCreationLdifTemplate()), "Can't create users unless userCreationLdifTemplate is defined");
209     GrouperUtil.assertion(StringUtils.isNotEmpty(config.getUserCreationBaseDn()), "Can't create users unless userCreationBaseDn is defined");
210     
211     LOG.info("Creating LDAP account for Subject: {} ", personSubject);
212     String ldif = config.getUserCreationLdifTemplate();
213     ldif = ldif.replaceAll("\\|\\|", "\n");
214     ldif = evaluateJexlExpression(ldif, personSubject, null);
215     
216     Connection conn = getLdapConnection();
217     try {
218       Reader reader = new StringReader(ldif);
219       LdifReader ldifReader = new LdifReader(reader);
220       SearchResult ldifResult = ldifReader.read();
221       LdapEntry ldifEntry = ldifResult.getEntry();
222       
223       // Update DN to be relative to userCreationBaseDn
224       String actualDn = String.format("%s,%s", ldifEntry.getDn(),config.getUserCreationBaseDn());
225       ldifEntry.setDn(actualDn);
226 
227       performLdapAdd(ldifEntry);
228       
229       // Read the acount that was just created
230       LOG.debug("Reading account that was just added to ldap server: {}", personSubject);
231       return fetchTargetSystemUser(personSubject);
232     } catch (PspException e) {
233       LOG.error("Problem while creating new user: {}: {}", ldif, e.getMessage());
234       throw e;
235     } catch ( IOException e ) {
236       LOG.error("Problem while processing ldif to create new user: {}", ldif, e);
237       throw new PspException("LDIF problem creating user: %s", e.getMessage());
238     }
239     finally {
240       conn.close();
241     }
242   }

244   protected void performLdapAdd(LdapEntry entryToAdd) throws PspException {
245     LOG.info("{}: Creating LDAP object: {}", getName(), entryToAdd.getDn());
246 
247     ensureLdapOusExist(entryToAdd.getDn(), false);
248     
249     Connection conn = getLdapConnection();
250     try {
251       // Actually ADD the account
252       conn.open();
253       conn.getProviderConnection().add(new AddRequest(entryToAdd.getDn(), entryToAdd.getAttributes()));
254     } catch (LdapException e) {
255       LOG.error("Problem while creating new object: {}", entryToAdd, e);
256       throw new PspException("LDAP problem creating object: %s", e.getMessage());
257     }
258     finally {
259       conn.close();
260     }

282   protected List<LdapObject> performLdapSearchRequest(String searchBaseDn, SearchScope scope, Collection<String> attributesToReturn, String filterTemplate, Object... filterParams) 
283   throws PspException {
284     SearchFilter filter = new SearchFilter(filterTemplate);
285     for (int i=0; i<filterParams.length; i++)
286       filter.setParameter(i, filterParams[i]);
287     
288     SearchRequest request = new SearchRequest(searchBaseDn, filter, attributesToReturn.toArray(new String[0]));
289     request.setSearchScope(scope);
290     return performLdapSearchRequest(request);
291   }

293   /**
294    * 
295    * @param request
296    * @param requestedAttributes A case-insensitive TreeSet. This is a copy of request.getReturnAttributes
297    * but is it broken out as a separate parameter so the exact same Set can be used across 1000s of 
298    * LdapObjects.
299    * 
300    * @return
301    * @throws LdapException
302    */
303   protected List<LdapObject> performLdapSearchRequest(SearchRequest request) throws PspException {
304     // Make sure a baseDn mentioned in our configuration exists.
305     ensureLdapOusExist(request.getBaseDn(), true);
306     
307     LOG.debug("Doing ldap search: {} / {} / {}", 
308         new Object[] {request.getSearchFilter(), request.getBaseDn(), Arrays.toString(request.getReturnAttributes())});
309     List<LdapObject> result = new ArrayList<LdapObject>();
310     
311     Connection conn = getLdapConnection();
312     try {
313       conn.open();
314       
315       // Turn on attribute-value paging if this is an active directory target
316       if ( config.isActiveDirectory() ) {
317         LOG.debug("Using attribute-value paging");
318         request.setSearchEntryHandlers(new RangeEntryHandler());
319       }
320       
321       Response<SearchResult> response;
322       
323       // Perform search. This is slightly different if paging is enabled or not. 
324       if ( config.isSearchResultPagingEnabled() ) {
325         PagedResultsClient client = new PagedResultsClient(conn, config.getSearchResultPagingSize());
326         LOG.debug("Using ldap search-result paging");
327         response = client.executeToCompletion(request);
328       }
329       else {
330         LOG.debug("Not using ldap search-result paging");
331         SearchOperation searchOp = new SearchOperation(conn);
332         response = searchOp.execute(request);
333       }
334       
335       SearchResult searchResult = response.getResult();
336       for (LdapEntry entry : searchResult.getEntries()) {
337         result.add(new LdapObject(entry, request.getReturnAttributes()));
338       }
339       
340       return result;
341     }
342     catch (LdapException e) {
343       LOG.error("Problem during ldap search {}", request, e);
344       throw new PspException("LDAP problem while searching: " + e.getMessage());
345     }
346     finally {
347       if ( conn != null )
348         conn.close();
349     }
350   }

352   protected LdapObject performLdapRead(String dn, String... attributes) throws PspException {
353     LOG.debug("Doing ldap read: {}", dn);
354     
355     Connection conn = getLdapConnection();
356     try {
357       conn.open();
358 
359       SearchRequest read = new SearchRequest(dn, "objectclass=*");
360       read.setSearchScope(SearchScope.OBJECT);
361       read.setReturnAttributes(attributes);
362       
363       // Turn on attribute-value paging if this is an active directory target
364       if ( config.isActiveDirectory() )
365         read.setSearchEntryHandlers(new RangeEntryHandler());
366 
367       SearchOperation searchOp = new SearchOperation(conn);
368       
369       Response<SearchResult> response = searchOp.execute(read);
370       SearchResult searchResult = response.getResult();
371       
372       LdapEntry result = searchResult.getEntry();
373       
374       if ( result == null )
375         return null;
376       return new LdapObject(result, attributes);
377     }
378     catch (LdapException e) {
379       if ( e.getResultCode() == ResultCode.NO_SUCH_OBJECT ) {
380         LOG.warn("{}: Ldap object does not exist: {}", getName(), dn);
381         return null;
382       }
383       
384       LOG.error("Problem during ldap read {}", dn, e);
385       throw new PspException("Problem during LDAP read: %s", e.getMessage());
386     }
387     finally {
388       if ( conn != null )
389         conn.close();
390     }
391   }

414   /**
415    * Note that the given {@link ProvisioningWorkItem} needs the given {@link ModifyRequest} done.
416    * 
417    * These are not done right away so that multiple modifications can be implemented together
418    * in batches. For example, LDAP servers can generally process a single ldap modification that
419    * adds 10 values to an attribute MUCH faster than processing 10 single-value Modify-Add operations.
420    * @param workItem
421    * @param operation
422    */
423   protected void scheduleLdapModification(ModifyRequest operation) {
424     ProvisioningWorkItem workItem = getCurrentWorkItem();
425     LOG.info("{}: Scheduling ldap modification: {}", getName(), operation);
426     
427     workItem.addValueToProvisioningData(LDAP_MOD_LIST, operation);
428   }

430   /**
431    * This implements the LDAP Modifications that were scheduled with schedulLdapModification.
432    * Those scheduled changes are stored within the ProvisioningWorkItems that are passed around.
433    * 
434    * In order to be fast, we first try to coalesce the changes across ProvisioningWorkItems.
435    * 
436    * If all the fancy, coalescing LDAP-implementation fails, each workItem's LDAP operations
437    *  will be done individually so problems will be tracked down to specific workItem(s).
438    */
439   @Override
440   public void finishProvisioningBatch(List<ProvisioningWorkItem> workItems) throws PspException {
441     
442     // TODO: Remove the unnecessary changes (LDAP Differencing)
443     
444     try {
445       MDC.put("step", "coalesced");
446       makeCoalescedLdapChanges(workItems);
447 
448       // They all worked, so mark them all as successful
449       for ( ProvisioningWorkItem workItem : workItems )
450         workItem.markAsSuccess("Modification complete");
451       
452     } catch (PspException e1) {
453       LOG.error("Optimized ldap provisioning failed: {}", e1.getMessage());
454       LOG.warn("Performing much slower unoptimized ldap provisioning after optimized provisioning failed");
455       
456         for ( ProvisioningWorkItem workItem : workItems ) {
457           try {
458             MDC.put("step", "ldap_retry:"+workItem.getMdcLabel());
459             makeIndividualLdapChanges(workItem);
460             workItem.markAsSuccess("Modification complete");
461           } catch (PspException e2) {
462             LOG.error("Simple ldap provisioning failed for {}: {}", workItem, e2.getMessage());
463             workItem.markAsFailure("Modification failed: %s", e2.getMessage());
464           }
465       }
466     }
467     finally {
468       MDC.remove("step");
469     }
470     
471     super.finishProvisioningBatch(workItems);
472   }

474   /**
475    * This ldap implementation is made complicated by strong desires to be fast, 
476    * specifically:
477    * + Pull changes made to common objects changed by several WorkItems together so they
478    * can be made in single ldap operations.
479    * + Chunk these changes into reasonable-sized pieces
480    * 
481    * Note, this involves the following steps:
482    * 1) Find all the ModifyRequests for a dn
483    * 2) Pull apart the ModifyRequests apart and find the AttributeModifications they contain
484    * 3) For Modify-Add and Modify-Delete operations, pull out all the added and deleted values
485    * and put them together in as few Modify-Add and Modify-Delete operations as is reasonable
486    * 4) Combine each attribute's changes into a list of values to ADD and a list of values to DELETE
487    * 5) Break long lists of attribute values into bite-sized chunks
488    * 6) Make a new Modification request for each dn that contains all the AttributeModifications
489    * for that dn
490 
491  * @param workItems
492  * @throws PspException
493  */
494   private void makeCoalescedLdapChanges(List<ProvisioningWorkItem> workItems) throws PspException {
495     LOG.debug("{}: Making coalescedLdapChanges", getName());
496 
497     // Assemble and execute all the LDAP_MOD_LIST values saved up in workItems
498     MultiMap dn2Mods = new MultiValueMap();
499     
500     // Sort all the necessary operations by the DN that they modify
501     for ( ProvisioningWorkItem workItem : workItems ) {
502       List<ModifyRequest> mods = (List) workItem.getProvisioningDataValues(LDAP_MOD_LIST);
503       
504       // Obviously there is nothing to coalesce if no mods were necessary for this work item
505       if ( mods == null ) 
506         continue;
507       
508       LOG.info("{}: WorkItem {} needs {} ldap modifications", 
509           new Object[]{getName(), workItem, mods.size()} );
510       for ( ModifyRequest mod : mods ) {
511         LOG.debug("{}: Mod for WorkItem: {}", getName(), getLoggingSummary(mod));
512         dn2Mods.put(mod.getDn(), mod);
513       }
514     }
515     
516     // Now loop through the DNs that need to be modified
517     for ( String dn : (Collection<String>) dn2Mods.keySet() ) {
518       // These are all the modifications that were assembled across our provisioning batch
519       Collection<ModifyRequest> modsForDn = (Collection<ModifyRequest>) dn2Mods.get(dn);
520 
521       // This will hold the actual operations that are necessary.
522       // This is a List of List<LDAP Modifications that should be done together>
523       //
524       // There will be more than one element in coalescedOperations when a DN has operations
525       // that are so large that they need to be broken into bite-sized chunks
526       List<List<AttributeModification>> coalescedOperations = new ArrayList<List<AttributeModification>>();
527       
528       // We know we're going to need at least one operation (since this DN was in dn2Mods), 
529       // so put an empty list of mods here to keep things simpler below
530       coalescedOperations.add(new ArrayList<AttributeModification>());
531       
532       // Sort all the attribute values modified by these modifications into one DEL & ADD
533       // list for each attribute. 
534       MultiMap attribute2ValuesToAdd = new MultiValueMap();
535       MultiMap attribute2ValuesToDel = new MultiValueMap();
536       
537       for ( ModifyRequest mod : modsForDn ) {
538         for ( AttributeModification attributeMod : mod.getAttributeModifications() ) {
539           LdapAttribute attribute = attributeMod.getAttribute();
540           
541           switch (attributeMod.getAttributeModificationType() ) {
542             case ADD: 
543               for ( String value : attribute.getStringValues() )
544                 attribute2ValuesToAdd.put(attribute.getName(), value);
545               break;
546             case REMOVE: 
547               for ( String value : attribute.getStringValues() )
548                 attribute2ValuesToDel.put(attribute.getName(), value);
549               break;
550             case REPLACE: 
551             default:
552               // We don't know how to combine these, so just do the operation as is
553               // in our first eventual operation
554               coalescedOperations.get(0).add(attributeMod);  
555           }
556         }
557       }
558       
559       int maxValues = config.getMaxValuesToChangePerOperation();
560       
561       // Create a single value-removal for each attribute that needs values removed
562       // Loop through the attributes that had values removed from them
563       //
564       // NOTE: We're doing removals first so that if an value is both added and removed
565       // (because there were 2+ workItems that conflicted with each other), the ADD will be
566       // done last and will 'stick.' If the removal is supposed to be the one that sticks, then
567       // it will be taken care of at full-sync time.
568       
569       // TODO: Figure out what workItems conflict and make sure those groups are full-sync'ed
570       // first
571       for ( String attributeName : (Collection<String>) attribute2ValuesToDel.keySet() ) {
572         Collection<String> values = (Collection<String>) attribute2ValuesToDel.get(attributeName);
573         List<List<String>> valueChunks = PspUtils.chopped(values, maxValues);
574         
575         for (int i=0; i<valueChunks.size(); i++) {
576           List<String> valueChunk = valueChunks.get(i);
577           
578           LdapAttribute attribute = new LdapAttribute(attributeName, GrouperUtil.toArray(valueChunk, String.class));
579           AttributeModification mod = new AttributeModification(AttributeModificationType.REMOVE, attribute);
580           
581           // Grow our list of operations if necessary
582           if ( coalescedOperations.size() <= i )
583             coalescedOperations.add(new ArrayList<AttributeModification>());
584           
585           coalescedOperations.get(i).add(mod);
586         }
587       }
588       
589 
590       
591       // Create a single value-add for each attribute that needs values added
592       // Loop through the attributes that had values added to them
593       for ( String attributeName : (Collection<String>) attribute2ValuesToAdd.keySet() ) {
594         Collection<String> values = (Collection<String>) attribute2ValuesToAdd.get(attributeName);
595         List<List<String>> valueChunks = PspUtils.chopped(values, maxValues);
596         
597         for (int i=0; i<valueChunks.size(); i++) {
598           List<String> valueChunk = valueChunks.get(i);
599           
600           LdapAttribute attribute = new LdapAttribute(attributeName, GrouperUtil.toArray(valueChunk, String.class));
601           AttributeModification mod = new AttributeModification(AttributeModificationType.ADD, attribute);
602           
603           // Grow our list of operations if necessary
604           if ( coalescedOperations.size() <= i )
605             coalescedOperations.add(new ArrayList<AttributeModification>());
606           
607           coalescedOperations.get(i).add(mod);
608         }
609       }
610       
611       Connection conn = getLdapConnection();
612       try {
613         for ( List<AttributeModification> operation : coalescedOperations ) {
614           ModifyRequest mod = new ModifyRequest(dn, GrouperUtil.toArray(operation, AttributeModification.class));
615           try {
616             conn.open();
617             
618             LOG.info("Performing LDAP modification: {}", getLoggingSummary(mod) );
619             conn.getProviderConnection().modify(mod);
620           } catch (LdapException e) {
621             LOG.warn("Problem doing coalesced ldap modification (THIS WILL BE RETRIED): {} / {}: {}",
622                 new Object[]{dn, mod, e.getMessage()});
623             throw new PspException("Coalesced LDAP Modification failed");
624           } 
625         }
626       } finally {
627         conn.close();
628       }
629     }
630   }

633   /**
634    * This method is a backup plan to makeCoalescedLdapChanges and takes a simple approach 
635    * to ldap provisioning. This is useful for two reasons: 1) It might work around a bug
636    * buried in the complexity of coalescing changes; 2) It tells us which workItems 
637    * have a problem because each workItem is done separately. 
638    * @param workItems
639    */
640   private void makeIndividualLdapChanges(ProvisioningWorkItem workItem) throws PspException {
641     List<ModifyRequest> mods = (List) workItem.getProvisioningDataValues(LDAP_MOD_LIST);
642     
643     if ( mods == null ) {
644       LOG.debug("No ldap changes are necessary for work item {}", workItem);
645       return;
646     }
647     
648     LOG.debug("Implementing changes for work item {}", workItem);
649     for ( ModifyRequest mod : mods ) {
650       Connection conn = getLdapConnection();
651       try {
652         conn.open();
653         conn.getProviderConnection().modify(mod);
654       } catch (LdapException e) {
655         
656         // Since we are a plan b provisioning attempt, it is possible that some of our 
657         // modifications have already been done. So, we look for the errors you get when
658         // you try to add something that already exists or try to delete something that doesn't exist
659         
660         // We're only doing this check if we have only one attribute in our modification request
661         
662         if ( mod.getAttributeModifications().length == 1 &&
663              mod.getAttributeModifications()[0].getAttributeModificationType() == AttributeModificationType.REMOVE &&
664              (e.getResultCode() == ResultCode.NO_SUCH_ATTRIBUTE || 
665               e.getResultCode() == ResultCode.NO_SUCH_OBJECT) )
666           LOG.info("Ignoring NO_SUCH_ATTRIBUTE/NO_SUCH_OBJECT error on an attribute removal operation: {}", mod);
667         
668         else if ( mod.getAttributeModifications().length == 1 &&
669             mod.getAttributeModifications()[0].getAttributeModificationType() == AttributeModificationType.ADD &&
670             e.getResultCode() == ResultCode.ATTRIBUTE_OR_VALUE_EXISTS )
671          LOG.info("Ignoring ATTRIBUTE_OR_VALUE_EXISTS error on an attribute add operation: {}", mod);
672         
673         else {
674           LOG.error("Ldap provisioning failed for {} / {}", new Object[]{workItem, mod, e});
675           throw new PspException("LDAP Provisioning failed");
676         }
677       } finally {
678         conn.close();
679       }
680     }
681   }

684   private String getLoggingSummary(ModifyRequest modForDn) {
685     if ( modForDn == null )
686       return "no changes";
687     
688     StringBuilder sb = new StringBuilder();
689     for ( AttributeModification attribute : modForDn.getAttributeModifications()) {
690       switch (attribute.getAttributeModificationType()) {
691         case ADD: sb.append(String.format("[%s: +%d value(s)]",
692                       attribute.getAttribute().getName(),
693                       attribute.getAttribute().getStringValues().size()));
694         break;
695         
696         case REMOVE: sb.append(String.format("[%s: -%d value(s)]",
697             attribute.getAttribute().getName(),
698             attribute.getAttribute().getStringValues().size()));
699         break;
700         
701         case REPLACE: sb.append(String.format("[%s: =%d value(s)]",
702             attribute.getAttribute().getName(),
703             attribute.getAttribute().getStringValues().size()));
704         break;
705       }
706     }
707   
708     return sb.toString();
709   }

724   public void ensureLdapOusExist(String dn, boolean wholeDnIsTheOu) throws PspException {
725     List<LdapAttribute> dnParts = DnParser.convertDnToAttributes(dn);
726     
727     if ( existingOUs.contains(dn) )
728       return;
729     
730     int start;
731     
732     // Start at 1 (skip rdn) if wholeDn is false
733     if (wholeDnIsTheOu)
734       start=0;
735     else
736       start=1;
737 
738     // When we're done, this will be the part of the DN that exists
739     int partThatExists=start;
740     
741     
742     while (partThatExists < dnParts.size()) {
743       String ou = DnParser.substring(dn, partThatExists);
744       
745       // If we know it exists (from previous searches), then break right out.
746       if ( existingOUs.contains(ou) )
747         break;
748       
749       LOG.debug("{}: Checking to see if ou exists: {}", getName(), ou);
750       try {
751         if ( performLdapRead(ou) == null ) 
752           partThatExists++;
753         else {
754           existingOUs.add(ou);
755           break;
756         }
757       }
758       catch (PspException e) {
759         LOG.error("{}: Unable to find existing OU ({})", getName(), ou);
760         throw new PspException("Unable to find existing OU or create new one");
761       }
762     }
763 
764     // Work our way back from partThatExists to the 2nd part of the DN
765     for (int i=partThatExists-1; i>=start; i--) {
766       String ouDn = DnParser.substring(dn,  i);
767       LdapAttribute attribute = dnParts.get(i);
768       
769       LOG.info("{}: Creating OU: {}", getName(), ouDn);
770       
771       String ldif = evaluateJexlExpression(config.getOuCreationLdifTemplate_defaultValue(), null, null, "dn", ouDn, "ou", attribute.getStringValue());
772       ldif = ldif.replaceAll("\\|\\|", "\n");
773 
774       try {
775         Reader reader = new StringReader(ldif);
776         LdifReader ldifReader = new LdifReader(reader);
777         SearchResult ldifResult = ldifReader.read();
778         LdapEntry ldifEntry = ldifResult.getEntry();
779         
780         // Add the current attribute if it is something other than OU
781         if ( ! attribute.getName().equalsIgnoreCase("ou") )
782           ldifEntry.addAttribute(attribute);
783         
784         performLdapAdd(ldifEntry);
785         existingOUs.add(ldifEntry.getDn());
786       } catch ( IOException e ) {
787         LOG.error("Problem while processing ldif to create new OU: {}", ldif, e);
788         throw new PspException("LDIF problem creating OU: %s", e.getMessage());
789       }
790     }
791   }
