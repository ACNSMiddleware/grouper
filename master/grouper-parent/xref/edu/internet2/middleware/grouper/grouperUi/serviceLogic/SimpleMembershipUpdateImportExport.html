129   /**
130    * export all fields of a group
131    * @param group
132    * @param headersCommaSeparated
133    * @param exportAllSortField
134    * @param simpleMembershipUpdateContainer
135    * @param immediateOnly
136    * @throws IOException
137    */
138   public static void exportGroupAllFieldsToBrowser(Group group, String headersCommaSeparated, String exportAllSortField, boolean immediateOnly) {
139     
140     try {
141       Set<Member> members = immediateOnly ? group.getImmediateMembers() : group.getMembers();
142       
143       Member.resolveSubjects(members, true);
144       
145       HttpServletResponse response = GrouperUiFilter.retrieveHttpServletResponse(); 
146       
147       String[] headers = GrouperUtil.splitTrim(headersCommaSeparated, ",");
148       
149       //note: isError is second to last col, error is the last column
150       boolean[] isAttribute = new boolean[headers.length];
151       int sortCol = 0;
152       int sourceIdCol = -1;
153       for (int i=0;i<headers.length;i++) {
154         isAttribute[i] = !nonAttributeCols.contains(headers[i].toLowerCase());
155   
156         if (StringUtils.equalsIgnoreCase(headers[i], exportAllSortField)) {
157           sortCol = i;
158         } else if (StringUtils.equalsIgnoreCase("sourceId", headers[i])) {
159           sourceIdCol = i;
160         }
161       }
162       
163       List<String[]> memberData = new ArrayList<String[]>(); 
164       for (Member member : members) {
165         // feed in your array (or convert your data to an array)
166         String[] entries = exportAllStringArray(member, headers, isAttribute);
167         memberData.add(entries);
168       }      
169    
170       final int SOURCE_ID_COL = sourceIdCol;
171       final int SORT_COL = sortCol;
172       //sort
173       Collections.sort(memberData, new Comparator() {
174    
175         /**
176          * 
177          * @param o1
178          * @param o2
179          * @return 1, -1, 0
180          */
181         public int compare(Object o1, Object o2) {
182           String[] first = (String[])o1;
183           String[] second = (String[])o2;
184           if (SOURCE_ID_COL != -1 && !StringUtils.equals(first[SOURCE_ID_COL], second[SOURCE_ID_COL])) {
185             return GrouperUiUtils.compare(first[SOURCE_ID_COL], second[SOURCE_ID_COL], true);
186           }
187           return GrouperUiUtils.compare(first[SORT_COL], second[SORT_COL], true);
188         }
189       });
190       
191       //say it is CSV
192       response.setContentType(HttpContentType.TEXT_CSV.getContentType());
193     
194       String groupExtensionFileName = GuiGroup.getExportAllFileNameStatic(group);
195       
196       response.setHeader ("Content-Disposition", "inline;filename=\"" + groupExtensionFileName + "\"");
197       
198       //just write some stuff
199       PrintWriter out = null;
200     
201       try {
202         out = response.getWriter();
203       } catch (Exception e) {
204         throw new RuntimeException("Cant get response.getWriter: ", e);
205       }
206       
207       CSVWriter writer = new CSVWriter(out);
208       String[] headersNew = new String[headers.length+2];
209       System.arraycopy(headers, 0, headersNew, 0, headers.length);
210       headersNew[headers.length] = "success";
211       headersNew[headers.length+1] = "errorMessage";
212       writer.writeNext(headersNew);
213       for (String[] entries: memberData) {
214         // feed in your array (or convert your data to an array)
215         writer.writeNext(entries);
216       }      
217       writer.close();
218       auditExport(group.getUuid(), group.getName(), memberData.size(), groupExtensionFileName);
219  
220     } catch (NoSessionException se) {
221       throw se;
222     } catch (Exception se) {
223       throw new RuntimeException("Error exporting all members from group: " + group.getName() 
224           + ", " + se.getMessage(), se);
225     }
226     throw new ControllerDone();
227   }

229   /**
230    * export all members
231    * @param member
232    * @param headers 
233    * @param isAttribute which indexes are attributes
234    * @return the stirng array for csv
235    */
236   public static String[] exportAllStringArray(Member member, String[] headers, boolean[] isAttribute) {
237     String[] result = new String[headers.length+2];
238     
239     //lets see what we can get from the member
240     for (int i=0;i<headers.length;i++) {
241       String header = headers[i];
242       if ("subjectId".equalsIgnoreCase(header)) {
243         result[i] = member.getSubjectId();
244       } else if ("entityId".equalsIgnoreCase(header)) {
245         result[i] = member.getSubjectId();
246       } else if ("sourceId".equalsIgnoreCase(header)) {
247         result[i] = member.getSubjectSourceId();
248       } else if ("memberId".equalsIgnoreCase(header)) {
249         result[i] = member.getUuid();
250       }
251     }
252     
253     
254     try {
255       
256       Subject subject = member.getSubject();
257       
258       //lets see what we can get from the subject
259       for (int i=0;i<headers.length;i++) {
260         String header = headers[i];
261         if ("name".equalsIgnoreCase(header)) {
262           result[i] = subject.getName();
263         } else if ("description".equalsIgnoreCase(header)) {
264           result[i] = subject.getDescription();
265         } else if ("screenLabel".equalsIgnoreCase(header)) {
266           result[i] = GrouperUiUtils.convertSubjectToLabelConfigured(subject);
267         } else if (isAttribute[i]) {
268           result[i] = subject.getAttributeValueOrCommaSeparated(header);
269         }
270       }
271       
272       result[headers.length] = "T";
273     } catch (NoSessionException se) {
274       throw se;
275     } catch (Exception e) {
276       result[headers.length] = "F";
277       String error = "error with memberId: " + member.getUuid() + ", subjectId: " + member.getSubjectId()
278         + ", " + ExceptionUtils.getFullStackTrace(e);
279       LOG.error(error);
280       result[headers.length + 1] = error;
281     }
282     return result;
283   }

285   /**
286    * export group subject ids
287    * @param group
288    * @param immediateOnly
289    */
290   public static void exportGroupSubjectIdsCsv(Group group, boolean immediateOnly) {
291     
292     try {
293   
294       Set<Member> members = immediateOnly ? group.getImmediateMembers() : group.getMembers();
295       
296       HttpServletResponse response = GrouperUiFilter.retrieveHttpServletResponse(); 
297       
298       List<String[]> memberData = new ArrayList<String[]>(); 
299       for (Member member : members) {
300         // feed in your array (or convert your data to an array)
301         String[] entries = new String[]{member.getSubjectSourceId(), member.getSubjectId()};
302         memberData.add(entries);
303       }      
304   
305       //sort
306       Collections.sort(memberData, new Comparator() {
307   
308         /**
309          * 
310          * @param o1
311          * @param o2
312          * @return 1, -1, 0
313          */
314         public int compare(Object o1, Object o2) {
315           String[] first = (String[])o1;
316           String[] second = (String[])o2;
317           if (!StringUtils.equals(first[0], second[0])) {
318             return GrouperUiUtils.compare(first[0], second[0], true);
319           }
320           return GrouperUiUtils.compare(first[1], second[1], true);
321         }
322       });
323       
324       //say it is CSV
325       response.setContentType("text/csv");
326     
327       String groupExtensionFileName = GuiGroup.getExportSubjectIdsFileNameStatic(group);
328       
329       response.setHeader ("Content-Disposition", "inline;filename=\"" + groupExtensionFileName + "\"");
330       
331       //just write some stuff
332       PrintWriter out = null;
333     
334       try {
335         out = response.getWriter();
336       } catch (Exception e) {
337         throw new RuntimeException("Cant get response.getWriter: ", e);
338       }
339       
340       CSVWriter writer = new CSVWriter(out);
341       writer.writeNext(new String[]{"sourceId", "entityId"});
342       for (String[] entries: memberData) {
343         // feed in your array (or convert your data to an array)
344         writer.writeNext(entries);
345       }      
346       writer.close();
347       
348       auditExport(group.getUuid(), group.getName(), memberData.size(), groupExtensionFileName);
349 
350       throw new ControllerDone();
351     } catch (NoSessionException se) {
352       throw se;
353     } catch (ControllerDone cd) {
354       throw cd;
355     } catch (Exception se) {
356       throw new RuntimeException("Error exporting all members from group: " + group.getName()
357           + ", " + se.getMessage(), se);
358     }
359     
360   }

432   /**
433    * import a CSV file
434    * @param httpServletRequest
435    * @param httpServletResponse
436    */
437   @SuppressWarnings("unchecked")
438   public void importCsv(HttpServletRequest httpServletRequest, HttpServletResponse httpServletResponse) {
439   
440     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
441     guiResponseJs.setAddTextAreaTag(true);
442     guiResponseJs.addAction(GuiScreenAction.newCloseModal());
443   
444     final Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn();
445     
446     GrouperSession grouperSession = null;
447   
448     Group group = null;
449     String groupName = null;
450   
451     try {
452       SimpleMembershipUpdateContainer simpleMembershipUpdateContainer = SimpleMembershipUpdateContainer.retrieveFromSession();
453       grouperSession = GrouperSession.start(loggedInSubject);
454   
455       group = new SimpleMembershipUpdate().retrieveGroup(grouperSession);
456       groupName = group.getName();
457       
458       List<Member> existingMembers = new ArrayList<Member>(GrouperUtil.nonNull(group.getImmediateMembers()));
459       
460       int existingCount = GrouperUtil.length(existingMembers);
461   
462       GrouperRequestWrapper grouperRequestWrapper = (GrouperRequestWrapper)httpServletRequest;
463       
464       Reader reader = null;
465       String fileName = null;
466       
467       GuiHideShow membershipLiteImportFileHideShow = GuiHideShow.retrieveHideShow("membershipLiteImportFile", true);
468       
469       if (membershipLiteImportFileHideShow.isShowing()) {
470         
471         FileItem importCsvFile = grouperRequestWrapper.getParameterFileItem("importCsvFile");
472         
473         fileName = StringUtils.defaultString(importCsvFile == null ? "" : importCsvFile.getName());
474     
475         String fileSize = importCsvFile == null ? "" : (" : " + FileUtils.byteCountToDisplaySize(importCsvFile.getSize()));
476         
477         fileName += fileSize;
478         
479         //validate the inputs, file is required
480         if (importCsvFile == null || importCsvFile.getSize() == 0 || 
481             importCsvFile.getName() == null 
482             || (!importCsvFile.getName().toLowerCase().endsWith(".csv")
483               &&  !importCsvFile.getName().toLowerCase().endsWith(".txt"))) {
484           
485           guiResponseJs.addAction(GuiScreenAction.newAlert("<pre>"  + simpleMembershipUpdateContainer.getText().getImportErrorNoWrongFile()
486               + fileName + "</pre>"));
487           return;
488         }
489         
490         reader = new InputStreamReader(importCsvFile.getInputStream());
491         
492       } else {
493         
494         //textarea import
495         String importCsvTextarea = grouperRequestWrapper.getParameter("importCsvTextarea");
496         
497         if (StringUtils.isBlank(importCsvTextarea)) {
498           guiResponseJs.addAction(GuiScreenAction.newAlert("<pre>" 
499               + simpleMembershipUpdateContainer.getText().getImportErrorBlankTextarea()
500               + "</pre>"));
501           return;
502           
503         }
504         
505         fileName = "textareaInput";
506         
507         reader = new StringReader(importCsvTextarea);
508       }
509       
510       
511       boolean importReplaceMembers = grouperRequestWrapper.getParameterBoolean("importReplaceMembers", false);
512       
513       //convert the import file to subjects
514       List<String> subjectErrors = new ArrayList<String>();
515       
516       List<Subject> importedSubjectWrappers = parseCsvImportFile(
517           reader, fileName, subjectErrors, new LinkedHashMap<String, Integer>(),
518           membershipLiteImportFileHideShow.isShowing());
519 
520       final List<String> addErrors = new ArrayList<String>();
521 
522       //filter out the require group
523       final String requireGroup = simpleMembershipUpdateContainer.configValue(
524           "simpleMembershipUpdate.subjectSearchRequireGroup", false);
525       
526       if (!StringUtils.isBlank(requireGroup)) {
527 
528         final List<Subject> SUBJECTS = importedSubjectWrappers;
529         
530         GrouperSession.callbackGrouperSession(
531             grouperSession.internal_getRootSession(), new GrouperSessionHandler() {
532           
533           @Override
534           public Object callback(GrouperSession rootGrouperSession) throws GrouperSessionException {
535             
536             Group groupFilter = GroupFinder.findByName(rootGrouperSession, requireGroup, true);
537             
538             Iterator<Subject> iterator = SUBJECTS.iterator();
539             
540             while (iterator.hasNext()) {
541               
542               Subject subject = iterator.next();
543               
544               if (!groupFilter.hasMember(subject)) {
545                 iterator.remove();
546                 String error = "Subject " + ((ImportSubjectWrapper)subject).errorLabelForError() 
547                   + " not in required group " + groupFilter.getName();
548                 addErrors.add(error);
549               }
550               
551               
552             }
553             return null;
554           }
555         });
556       }
557       
558       GrouperUiUtils.removeOverlappingSubjects(existingMembers, importedSubjectWrappers);
559       
560       int addedCount = 0;
561       
562       //first lets add some members
563       for (int i=0;i<importedSubjectWrappers.size();i++) {
564         
565         ImportSubjectWrapper importedSubjectWrapper = (ImportSubjectWrapper)importedSubjectWrappers.get(i);
566         Subject wrappedSubject = null;
567         try {
568           wrappedSubject = importedSubjectWrapper.wrappedSubject();
569         } catch (Exception e) {
570           //ignore
571         }
572         try {
573           if (wrappedSubject != null) {
574             
575             group.addMember(wrappedSubject, false);
576           } else {
577             
578             group.addMember(importedSubjectWrapper, false);
579           }
580           
581           addedCount++;
582         } catch (Exception e) {
583           String error = "Error adding subject from " + importedSubjectWrapper.errorLabelForError() + ", " + e.getMessage();
584           LOG.warn(error, e);
585           addErrors.add(error);
586         }
587   
588       }
589   
590       boolean didntImportDueToSubjects = false;
591       int deletedCount = 0;
592       List<String> deleteErrors = new ArrayList<String>();
593   
594       //remove the ones which are already there
595       if (importReplaceMembers) {
596         
597         if (GrouperUtil.length(subjectErrors) == 0) {
598           
599           for (Member existingMember : existingMembers) {
600             
601             try {
602               group.deleteMember(existingMember, false);
603               deletedCount++;
604             } catch (Exception e) {
605               String error = "Error deleting subject " + SubjectHelper.getPretty(existingMember) + e.getMessage();
606               LOG.warn(error, e);
607               deleteErrors.add(error);
608             }
609           }
610         } else {
611           didntImportDueToSubjects = true;
612         }
613         
614         
615       }
616       
617       //this might be a little wasteful, but I think it is a good sanity check
618       int newSize = group.getImmediateMembers().size();
619       
620       StringBuilder result = new StringBuilder();
621       //result.append("File: " + importCsvFile.getName() + ", " + importCsvFile.getFieldName() 
622       //    + ", isFormField: " + importCsvFile.isFormField() + ", inMemory: " 
623       //    + importCsvFile.isInMemory() + "\n\n" + importCsvFile.getString() + "\n\n");
624       //result.append("importReplaceMembers: " + importReplaceMembers + "\n\n");
625       
626       //first of all, was it successful?
627       int errorSize = GrouperUtil.length(subjectErrors) + GrouperUtil.length(addErrors) 
628         + GrouperUtil.length(deleteErrors);
629       boolean hasError = errorSize > 0;
630       if (!hasError) {
631         
632         result.append("<b>").append(simpleMembershipUpdateContainer.getText().getImportSuccessSummary()).append("</b><br /><br />\n");
633         
634       } else {
635         
636         result.append("<b>").append(simpleMembershipUpdateContainer.getText().getImportErrorSummary(errorSize)).append("</b><br /><br />\n");
637         
638       }
639       
640       //give general summary
641       result.append(simpleMembershipUpdateContainer.getText().getImportSizeSummary(existingCount, newSize)).append("<br />\n");
642   
643       if (didntImportDueToSubjects) {
644         result.append(simpleMembershipUpdateContainer.getText().getImportErrorSubjectProblems()).append("<br />\n");
645       }
646       
647       //adds, deletes
648       result.append(simpleMembershipUpdateContainer.getText().getImportAddsDeletesSummary(addedCount, deletedCount)).append("<br />\n");
649       
650       if (GrouperUtil.length(subjectErrors) > 0) {
651         result.append("<br /><b>").append(simpleMembershipUpdateContainer.getText().getImportSubjectErrorsLabel())
652             .append("</b><br />\n");
653         for (String error: subjectErrors) {
654           result.append(error).append("<br />\n");
655         }
656       }
657       
658       if (GrouperUtil.length(addErrors) > 0) {
659         result.append("<br /><b>").append(simpleMembershipUpdateContainer.getText().getImportAddErrorsLabel())
660             .append("</b><br />\n");
661         for (String error: addErrors) {
662           result.append(error).append("<br />\n");
663         }
664       }
665   
666       if (GrouperUtil.length(deleteErrors) > 0) {
667         result.append("<br /><b>").append(simpleMembershipUpdateContainer.getText().getImportRemoveErrorsLabel())
668           .append("</b><br />\n");
669         for (String error: deleteErrors) {
670           result.append(error).append("<br />\n");
671         }
672       }
673       
674       guiResponseJs.addAction(GuiScreenAction.newAlert(result.toString(), false));
675   
676     } catch (NoSessionException se) {
677       throw se;
678     } catch (ControllerDone cd) {
679       throw cd;
680     } catch (Exception se) {
681       throw new RuntimeException("Error importing members to group: " + groupName 
682           + ", " + se.getMessage(), se);
683     } finally {
684       GrouperSession.stopQuietly(grouperSession); 
685     }
686     
687     //refresh list... since it probably changed
688     new SimpleMembershipUpdate().retrieveMembers(httpServletRequest, httpServletResponse);
689     
690   }

695   @SuppressWarnings("serial")
696   public static class GrouperImportException extends RuntimeException {
697 
698     /**
699      * 
700      */
701     public GrouperImportException() {
702       super();
703     }
704 
705     /**
706      * @param message
707      * @param cause
708      */
709     public GrouperImportException(String message, Throwable cause) {
710       super(message, cause);
711     }
712

714      * @param message
715      */
716     public GrouperImportException(String message) {
717       super(message);
718     }
719 
720     /**
721      * @param cause
722      */
723     public GrouperImportException(Throwable cause) {
724       super(cause);
725     }
726     
727     
728   }

730   /**
731    * Note, this will close the reader passed in
732    * @param originalReader
733    * @param fileName
734    * @param subjectErrors pass in a list and errors will be put in here
735    * @param errorSubjectIdsOnRow is a map where the key is subjectId and the value if the line number
736    * @param isFileUpload true if file upload, as opposed to a textarea
737    * @return the list, never null
738    * @throws GrouperImportException for messages to the screen
739    */
740   @SuppressWarnings("unchecked")
741   public static List<Subject> parseCsvImportFile(Reader originalReader, String fileName, List<String> subjectErrors, 
742       Map<String, Integer> errorSubjectIdsOnRow, boolean isFileUpload) throws GrouperImportException {
743     
744     //convert from CSV to 
745     CSVReader reader = null;
746     
747     try {
748       //note, the first row is the title
749       List<String[]> csvEntries = null;
750     
751       try {
752         reader = new CSVReader(originalReader);
753         csvEntries = reader.readAll();
754       } catch (IOException ioe) {
755         throw new GrouperImportException("Error processing file: " + fileName, ioe);
756       }
757       
758       List<Subject> uploadedSubjects = new ArrayList<Subject>();
759       
760       //lets get the headers
761       int sourceIdColumn = -1;
762       int subjectIdColumn = -1;
763       int subjectIdentifierColumn = -1;
764       int subjectIdOrIdentifierColumn = -1;
765       
766       //must have lines
767       if (GrouperUtil.length(csvEntries) == 0) {
768         if (isFileUpload) {
769           throw new GrouperImportException(GrouperUiUtils.message("simpleMembershipUpdate.importErrorNoWrongFile"));
770         }
771         throw new GrouperImportException(GrouperUiUtils.message("simpleMembershipUpdate.importErrorBlankTextarea"));
772       }
773       
774       //lets go through the headers
775       String[] headers = csvEntries.get(0);
776       int headerSize = headers.length;
777       boolean foundHeader = false;
778       for (int i=0;i<headerSize;i++) {
779         if ("sourceId".equalsIgnoreCase(headers[i])) {
780           foundHeader = true;
781           sourceIdColumn = i;
782         }
783         if ("subjectId".equalsIgnoreCase(headers[i]) || "entityId".equalsIgnoreCase(headers[i])) {
784           foundHeader = true;
785           subjectIdColumn = i;
786         }
787         if ("subjectIdentifier".equalsIgnoreCase(headers[i]) || "entityIdentifier".equalsIgnoreCase(headers[i])) {
788           foundHeader = true;
789           subjectIdentifierColumn = i;
790         }
791         if ("subjectIdOrIdentifier".equalsIgnoreCase(headers[i]) || "entityIdOrIdentifier".equalsIgnoreCase(headers[i])) {
792           foundHeader = true;
793           subjectIdOrIdentifierColumn = i;
794         }
795       }
796       
797       //normally start on index 1, if the first row is header
798       int startIndex = 1;
799       
800       //must pass in an id
801       if (subjectIdColumn == -1 && subjectIdentifierColumn == -1 && subjectIdOrIdentifierColumn == -1) {
802         if (!foundHeader && headerSize == 1) {
803           //there was no header, so pretend like it was subjectIdOrIdentifier
804           subjectIdOrIdentifierColumn = 0;
805           startIndex = 0;
806         } else {
807           throw new GrouperImportException(TextContainer.retrieveFromRequest().getText().get("simpleMembershipUpdate.importErrorNoIdCol"));
808         }
809       }
810       
811       //ok, lets go through the rows, start after the headers
812       for (int i=startIndex;i<csvEntries.size();i++) {
813         String[] csvEntry = csvEntries.get(i);
814         int row = i+1;
815         
816         //try catch each one and see where we get
817         String subjectId = null;
818         String subjectIdentifier = null;
819         String subjectIdOrIdentifier = null;
820         try {
821           String sourceId = null;
822     
823           sourceId = sourceIdColumn == -1 ? null : csvEntry[sourceIdColumn]; 
824           subjectId = subjectIdColumn == -1 ? null : csvEntry[subjectIdColumn]; 
825           subjectIdentifier = subjectIdentifierColumn == -1 ? null : csvEntry[subjectIdentifierColumn]; 
826           subjectIdOrIdentifier = subjectIdOrIdentifierColumn == -1 ? null : csvEntry[subjectIdOrIdentifierColumn]; 
827           
828           ImportSubjectWrapper importSubjectWrapper = 
829             new ImportSubjectWrapper(row, sourceId, subjectId, subjectIdentifier, subjectIdOrIdentifier, csvEntry);
830           uploadedSubjects.add(importSubjectWrapper);
831           
832         } catch (Exception e) {
833           LOG.info(e);
834           subjectErrors.add("Error on " + ImportSubjectWrapper.errorLabelForRowStatic(row, csvEntry) + ": " +    e.getMessage());
835           
836           String errorSubjectId = StringUtils.defaultIfEmpty(subjectId, subjectIdentifier);
837           errorSubjectId = StringUtils.defaultIfEmpty(errorSubjectId, subjectIdOrIdentifier);
838           
839           errorSubjectIdsOnRow.put(errorSubjectId, row);
840         }
841       
842       }
843       
844       return uploadedSubjects;
845       
846     } finally {
847       if (reader != null) {
848         try {
849           reader.close();
850         } catch (Exception e) {
851           LOG.warn("error", e);
852         }
853       }
854     }
855     
856   }
