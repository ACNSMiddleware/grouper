167   /**
168    * @see javax.servlet.http.HttpServlet#service(javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse)
169    */
170   @SuppressWarnings({ "unchecked" })
171   @Override
172   protected void doGet(HttpServletRequest request, HttpServletResponse response)
173       throws ServletException, IOException {
174 
175     GrouperUiFilter.assignHttpServlet(this);
176     
177     String uri = request.getRequestURI();
178     
179     if (!uri.endsWith("/UiV2Main.index")
180         && !uri.endsWith("/UiV2Public.index")) {
181       RequestContainer.retrieveFromRequest().setAjaxRequest(true);
182     }
183     //initialize the bean
184     GrouperRequestContainer.retrieveFromRequestOrCreate();
185 
186     List<String> urlStrings = extractUrlStrings(request);
187     
188     GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
189     
190     String appStateString = request.getParameter("appState");
191     JSONObject jsonObject = JSONObject.fromObject(appStateString);
192 
193     AppState appState = (AppState)JSONObject.toBean( jsonObject, AppState.class ); 
194 
195     //app state isnt there e.g. for ajax components
196     if (appState == null) {
197       appState = new AppState();
198     }
199     
200     appState.initRequest();
201     
202     try {
203       initGui();
204     } catch (NoUserAuthenticatedException nuae) {
205       //redirect to the error screen if no loginid could be found
206       response.sendRedirect(GrouperUiFilter.retrieveServletContext() + "/grouperExternal/public/UiV2Public.index?operation=UiV2Public.postIndex&function=UiV2Public.error&code=noUserAuthenticated");
207       return;
208     } catch (SubjectNotFoundException nuae) {
209       //redirect to the error screen if loginid did not resolve to a subject
210       response.sendRedirect(GrouperUiFilter.retrieveServletContext() + "/grouperExternal/public/UiV2Public.index?operation=UiV2Public.postIndex&function=UiV2Public.error&code=authenticatedSubjectNotFound");
211       return;
212     } catch (ControllerDone cd) {
213       guiResponseJs.printToScreen();
214       return;
215     }
216     
217     boolean printToScreen = true;
218 
219     //this is just the filename of the export, for the browser to save correctly if right click
220     if (GrouperUtil.length(urlStrings) == 3
221         && StringUtils.equals("app", urlStrings.get(0))
222         && (StringUtils.equals("SimpleMembershipUpdateImportExport.exportSubjectIdsCsv", urlStrings.get(1))
223             || StringUtils.equals("SimpleMembershipUpdateImportExport.exportAllCsv", urlStrings.get(1)))) {
224       //strip off the filename
225       urlStrings = GrouperUtil.toList(urlStrings.get(0), urlStrings.get(1));
226     }
227 
228     if (GrouperUtil.length(urlStrings) == 5
229         && StringUtils.equals("app", urlStrings.get(0))
230         && StringUtils.equals(UiV2GroupImport.class.getSimpleName() + ".groupExportSubmit", urlStrings.get(1))) {
231 
232       //strip off the filename
233       urlStrings = GrouperUtil.toList(urlStrings.get(0), urlStrings.get(1));
234     }
235 
236     Subject loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn(true, response);
237     if (loggedInSubject == null && !StringUtils.equals(UiV2Public.class.getSimpleName() + ".index", urlStrings.get(1))
238         && !StringUtils.equals(UiV2Public.class.getSimpleName() + ".postIndex", urlStrings.get(1))
239         && !StringUtils.defaultString(urlStrings.get(1)).startsWith("ExternalSubjectSelfRegister.")) {
240 
241       response.sendRedirect(GrouperUiFilter.retrieveServletContext() + "/grouperExternal/public/UiV2Public.index?operation=UiV2Public.postIndex&function=UiV2Public.error&code=anonymousSessionNotAllowed");
242       return;
243 
244     }
245     
246     //see what operation we are doing
247     if (GrouperUtil.length(urlStrings) == 2 
248         && (StringUtils.equals("public", urlStrings.get(0))
249             || StringUtils.equals("app", urlStrings.get(0)))) {
250 
251       String classAndMethodName = urlStrings.get(1);
252 
253       //if there is a hash, then toss that part out
254       if (classAndMethodName.contains("#")) {
255         classAndMethodName = GrouperUtil.prefixOrSuffix(classAndMethodName, "#", true);
256       }
257       
258       //lets do some simple validation, text dot text
259       if (!classAndMethodName.matches("^[a-zA-Z0-9]+\\.[a-zA-Z0-9]+$")) {
260         throw new RuntimeException("Invalid class and method name: '" + classAndMethodName + "'");
261       }
262 
263       //I think we are all post all the time, right?
264       if (!StringUtils.equalsIgnoreCase("post", request.getMethod() )) {
265         if (!operationsOkGet.contains(classAndMethodName)) {
266           String errorMessage = "Cant process method: " + request.getMethod() + " for operation: " + classAndMethodName;
267           throw new RuntimeException(errorMessage);
268         }
269       }
270       
271       String className = GrouperUtil.prefixOrSuffix(classAndMethodName, ".", true);
272       String methodName = GrouperUtil.prefixOrSuffix(classAndMethodName, ".", false);
273       
274       //not very scientific, oh well
275       boolean uiv2 = className.toLowerCase().startsWith("uiv2");
276       
277       //now lets call some simple reflection, must be public static void and take a request and response
278       className = "edu.internet2.middleware.grouper.grouperUi.serviceLogic." + className;
279       
280       Object instance = null;
281       
282       try {
283         instance = GrouperUtil.newInstance(GrouperUtil.forName(className));
284       } catch (RuntimeException re) {
285         GrouperUtil.injectInException(re, "Problem calling class and method: " + className + "." + methodName);
286         throw re;
287       }
288       
289       try {
290         
291         GrouperUtil.callMethod(instance.getClass(), instance, methodName, 
292             new Class<?>[]{HttpServletRequest.class, HttpServletResponse.class}, 
293             new Object[]{request, response}, true, false);
294 
295       } catch (NoSessionException nse) {
296 
297         if (uiv2) {
298           guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
299               TextContainer.retrieveFromRequest().getText().get("guiMiscNoSessionError")));
300         } else {
301 
302           boolean addTextArea = guiResponseJs.isAddTextAreaTag();
303           
304           //print out error message for user, a new one
305           guiResponseJs = new GuiResponseJs();
306           guiResponseJs.addAction(GuiScreenAction.newCloseModal());
307           guiResponseJs.setAddTextAreaTag(addTextArea);
308           
309           UiSection uiSection = GrouperUiFilter.uiSectionForRequest();
310           
311           String startOverKey = "simpleMembershipUpdate.startOver";
312           
313           if (uiSection == UiSection.EXTERNAL) {
314             startOverKey = "externalSubjectSelfRegister.startOver";
315           }
316             
317           guiResponseJs.addAction(GuiScreenAction.newAlert(GrouperUiUtils.message(startOverKey)));
318         }
319       } catch (ControllerDone cd) {
320         printToScreen = cd.isPrintGuiReponseJs();
321         //do nothing, this is ok
322       } catch (RuntimeException re) {
323         String error = "Problem calling reflection from URL: " + className + "." + methodName + "\n\n" + ExceptionUtils.getFullStackTrace(re);
324         LOG.error(error);
325         GrouperUiUtils.appendErrorToRequest(error);
326         
327         if (uiv2) {
328           guiResponseJs.addAction(GuiScreenAction.newMessage(GuiMessageType.error, 
329               TextContainer.retrieveFromRequest().getText().get("guiMiscErrorPrefix")
330               + " " + GrouperUiUtils.escapeHtml(re.getMessage(), true)));
331         } else {
332           //if adding text area for form file submits, make sure to do that in errors too
333           boolean addTextArea = guiResponseJs.isAddTextAreaTag();
334           
335           //print out error message for user, a new one
336           guiResponseJs = new GuiResponseJs();
337           guiResponseJs.addAction(GuiScreenAction.newCloseModal());
338           guiResponseJs.setAddTextAreaTag(addTextArea);
339           guiResponseJs.addAction(GuiScreenAction.newAlert("Error: " + GrouperUiUtils.escapeHtml(re.getMessage(), true)));
340         }        
341       }
342     } else {
343       //print out error message for user, a new one
344       guiResponseJs = new GuiResponseJs();
345       String error = "Cant find logic for URL: " 
346         + GrouperUtil.toStringForLog(urlStrings);
347       guiResponseJs.addAction(GuiScreenAction.newAlert("Error: " + error));
348       LOG.error(error);
349       GrouperUiUtils.appendErrorToRequest(error);
350     }
351 
352     if (printToScreen) {
353       guiResponseJs.printToScreen();
354     }
355   }

357   /**
358    * init stuff on gui
359    */
360   public static void initGui() {
361     GuiSettings guiSettings = new GuiSettings();
362     
363     guiSettings.storeToRequest();
364     
365     guiSettings.setAuthnKey(GrouperUuid.getUuid());
366 
367     // lets see where the templates are: assume grouperUiText.properties is in WEB-INF/classes,
368     // and templates are WEB-INF/templates
369     AppState appState = AppState.retrieveFromRequest();
370     if (!appState.isInitted()) {
371 
372       GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
373       guiResponseJs.addAction(GuiScreenAction.newAssign("allObjects.guiSettings", guiSettings));  
374       
375       guiResponseJs.addAction(GuiScreenAction.newAssign("allObjects.appState.initted", true));
376     }
377 
378     Subject loggedInSubject = null;
379     
380     loggedInSubject = GrouperUiFilter.retrieveSubjectLoggedIn(true, null);
381     if (loggedInSubject != null) {
382       guiSettings.setLoggedInSubject(new GuiSubject(loggedInSubject));
383     }
384     
385   }

387   /**
388    * for error messages, get a detailed report of the request
389    * @param request
390    * @return the string of descriptive result
391    */
392   public static String requestDebugInfo(HttpServletRequest request) {
393     StringBuilder result = new StringBuilder();
394     result.append(" uri: ").append(request.getRequestURI());
395     result.append(", method: ").append(request.getMethod());
396     result.append(", decoded url strings: ");
397     List<String> urlStrings = extractUrlStrings(request);
398     int urlStringsLength = GrouperUtil.length(urlStrings);
399     if (urlStringsLength == 0) {
400       result.append("[none]");
401     } else {
402       for (int i = 0; i < urlStringsLength; i++) {
403         result.append(i).append(": '").append(urlStrings.get(i)).append("'");
404         if (i != urlStringsLength - 1) {
405           result.append(", ");
406         }
407       }
408     }
409     return result.toString();
410   }

421     return extractUrlStrings(requestResourceFull);

424   /**
425    * TODO change to GrouperRestServlet in next release
426    * <pre>
427    * take a request uri and break up the url strings not including the app name or servlet
428    * this does not include the url params (if applicable)
429    * if the input is: grouper-ws/servicesRest/xhtml/v1_3_000/groups/members
430    * then the result is a list of size 2: {"group", "members"}
431    * 
432    * </pre>
433    * @param requestResourceFull
434    * @return the url strings
435    */
436   private static List<String> extractUrlStrings(String requestResourceFull) {
437     String[] requestResources = StringUtils.split(requestResourceFull, '/');
438     List<String> urlStrings = new ArrayList<String>();
439     //loop through and decode
440     int index = 0;
441     for (String requestResource : requestResources) {
442       //skip the app name and lite servlet
443       if (index++ < 2) {
444         continue;
445       }
446       //unescape the url encoding
447       urlStrings.add(GrouperUtil.escapeUrlDecode(requestResource));
448     }
449     return urlStrings;
450   }
