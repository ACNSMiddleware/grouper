38    /**
39     * find group inherit rules by stem name
40     * @param stem
41     * @return the rules
42     */
43    public static Set<RuleDefinition> findGroupPrivilegeInheritRules(Stem stem) {
44      
45      RuleEngine ruleEngine = RuleEngine.ruleEngine();
46      
47      //handle root stem... hmmm
48      //we need to simulate a child object here
49      String groupName = stem.isRootStem() ? "qwertyuiopasdfghjkl:b" : (stem.getName() + ":b");
50      
51      RuleCheck ruleCheck = new RuleCheck(RuleCheckType.groupCreate.name(), null, groupName, null, null, null);
52      Set<RuleDefinition> ruleDefinitions = new HashSet<RuleDefinition>(GrouperUtil.nonNull(ruleEngine.ruleCheckIndexDefinitionsByNameOrIdInFolder(ruleCheck)));
53      
54      Iterator<RuleDefinition> iterator = ruleDefinitions.iterator();
55      
56      while (iterator.hasNext()) {
57        
58        RuleDefinition ruleDefinition = iterator.next();
59        RuleThen ruleThen = ruleDefinition.getThen();
60        RuleThenEnum ruleThenEnum = ruleThen.thenEnum();
61        if (ruleThenEnum != RuleThenEnum.assignGroupPrivilegeToGroupId) {
62          iterator.remove();
63        }
64      }
65      
66      return ruleDefinitions;
67  
68    }

70    /**
71     * find subject inherit rules by stem name.  Note, the calling subject must be able to see the rules
72     * @param secure
73     * @return the rules
74     */
75    public static Set<RuleDefinition> findPrivilegeInheritRules(boolean secure) {
76      RuleEngine ruleEngine = RuleEngine.ruleEngine();
77  
78      Set<RuleDefinition> ruleDefinitions = new HashSet<RuleDefinition>();
79      
80      //go through rules and see which are about creating objects, and assigning privileges
81  
82      Subject grouperSessionSubject = GrouperSession.staticGrouperSession().getSubject();
83  
84      boolean wheelOrRoot = PrivilegeHelper.isWheelOrRoot(GrouperSession.staticGrouperSession().getSubject());
85  
86      for (RuleDefinition ruleDefinition : ruleEngine.getRuleDefinitions()) {
87        switch (ruleDefinition.getCheck().checkTypeEnum()) {
88          case attributeDefCreate:
89          case groupCreate:
90          case stemCreate:
91            switch (ruleDefinition.getThen().thenEnum()) {
92              case assignAttributeDefPrivilegeToAttributeDefId:
93              case assignGroupPrivilegeToGroupId:
94              case assignStemPrivilegeToStemId:
95  
96                //get the stem
97                Stem stem = ruleDefinition.getAttributeAssignType().getOwnerStemFailsafe();
98                if (stem == null) {
99                  //why?
100                 continue;
101               }
102 
103               boolean privilegeOk = false;
104               if (wheelOrRoot || !secure) {
105                 privilegeOk = true;
106               }
107               if (!privilegeOk) {
108 
109                 if (!stem.canHavePrivilege(grouperSessionSubject, NamingPrivilege.STEM_ADMIN.getName(), false)) {
110                   privilegeOk = true;
111                 }
112               }
113               if (!privilegeOk) {
114                 Scope scope = Scope.valueOfIgnoreCase(ruleDefinition.getCheck().getCheckStemScope(), false);
115                 
116                 if (scope == null) {
117                   continue;
118                 }
119                 
120                 switch (ruleDefinition.getCheck().checkTypeEnum()) {
121                   case attributeDefCreate:
122 
123                     //is there a membership for attribute def?
124                     Membership membership = new MembershipFinder().assignScope(stem.getName()).assignStemScope(scope)
125                       .assignField(AttributeDefPrivilege.ATTR_ADMIN.getField())
126                       .assignQueryOptionsForAttributeDef(new QueryOptions().paging(1, 1, false)).findMembership(false);
127                     
128                     privilegeOk = membership != null;
129                     
130                     break;
131                   case groupCreate:
132 
133                     //is there a membership for group?
134                     membership = new MembershipFinder().assignScope(stem.getName()).assignStemScope(scope)
135                       .assignField(AccessPrivilege.ADMIN.getField())
136                       .assignQueryOptionsForGroup(new QueryOptions().paging(1, 1, false)).findMembership(false);
137                     
138                     privilegeOk = membership != null;
139 
140                     break;
141                   case stemCreate:
142                     
143                     //is there a membership for stem?
144                     membership = new MembershipFinder().assignScope(stem.getName()).assignStemScope(scope)
145                       .assignField(NamingPrivilege.STEM_ADMIN.getField())
146                       .assignQueryOptionsForStem(new QueryOptions().paging(1, 1, false)).findMembership(false);
147                     
148                     privilegeOk = membership != null;
149 
150                     break;
151                   default:
152                     
153                 }
154               }
155               if (!privilegeOk) {
156                 continue;
157               }
158               ruleDefinitions.add(ruleDefinition);
159               break;
160             default: 
161                 
162           }
163           default:
164       }
165     }
166     
167     return ruleDefinitions;
168 
169   }
