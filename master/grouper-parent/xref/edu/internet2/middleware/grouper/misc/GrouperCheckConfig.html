131   /**
132    * verify that a group exists by name (dont throw exceptions)
133    * @param grouperSession (probably should be root session)
134    * @param groupName
135    * @param logError 
136    * @param autoCreate if auto create, or null, for grouper.properties setting
137    * @param logAutocreate 
138    * @param displayExtension optional, dislpay extension if creating
139    * @param groupDescription group description if auto create
140    * @param propertyDescription for logging explaning to the user how to fix the problem
141    * @param groupResult put in an array of size one to get the group back
142    * @return if group exists or not or was created
143    */
144   public static CheckGroupResult checkGroup(GrouperSession grouperSession, String groupName, 
145       boolean logError, Boolean autoCreate, 
146       boolean logAutocreate, String displayExtension, String groupDescription, String propertyDescription,
147       Group[] groupResult) {
148 
149     if (configCheckDisabled()) {
150       return CheckGroupResult.DIDNT_CHECK;
151     }
152     try {
153       Group group = GroupFinder.findByName(grouperSession, groupName, true);
154       if (group != null) {
155         if (GrouperUtil.length(groupResult) >= 1) {
156           groupResult[0] = group;
157         }
158         return CheckGroupResult.EXISTS;
159       }
160     } catch (Exception e) {
161       
162     }
163     
164     if (logError) {
165       String error = "cannot find group from config: " + propertyDescription + ": " + groupName;
166       System.err.println("Grouper warning: " + error);
167       LOG.warn(error);
168     }
169     
170     //get auto create from config
171     if (autoCreate == null) {
172       Properties properties = GrouperConfig.retrieveConfig().properties();
173       autoCreate = GrouperUtil.propertiesValueBoolean(properties, GrouperConfig.retrieveConfig().propertiesOverrideMap(), 
174           "configuration.autocreate.system.groups", false);
175     }
176     
177     if (autoCreate) {
178       try {
179         Group group = Group.saveGroup(grouperSession, null, null, groupName, displayExtension, groupDescription, null, true);
180         if (GrouperUtil.length(groupResult) >= 1) {
181           groupResult[0] = group;
182         }
183         if (logAutocreate) {
184           String error = "auto-created " + propertyDescription + ": " + groupName;
185           System.err.println("Grouper note: " + error);
186           LOG.warn(error);
187         }
188         return CheckGroupResult.CREATED;
189       } catch (Exception e) {
190         System.err.println("Grouper error: " + groupName + ", " + ExceptionUtils.getFullStackTrace(e));
191         LOG.error("Problem with group: " + groupName, e);
192         return CheckGroupResult.ERROR_CREATING;
193       }
194     }
195     
196     return CheckGroupResult.DOESNT_EXIST;
197   }

199   /**
200    * check a jar
201    * @param name name of the jar from grouper
202    * @param size that the jar should be
203    * @param sampleClassName inside the jar
204    * @param manifestVersion in the manifest file, which version we are expecting
205    */
206   public static void checkJar(String name, long size, String sampleClassName, String manifestVersion) {
207     
208     if (configCheckDisabled()) {
209       return;
210     }
211     
212     Class sampleClass = null;
213     try {
214       sampleClass = Class.forName(sampleClassName);
215     } catch (ClassNotFoundException cnfe) {
216       String error = "cannot find class " + sampleClassName + ", perhaps you are missing jar: " + name;
217       System.err.println("Grouper warning: " + error);
218       LOG.warn(error);
219       return;
220     }
221     String jarFileFullName = null;
222     String jarFileName = null;
223     String jarVersion = null;
224     long jarFileSize = -1;
225     try {
226       
227       File jarFile = GrouperUtil.jarFile(sampleClass, true);
228       jarFileFullName = jarFile.getCanonicalPath();
229       jarFileName = jarFile.getName();
230       jarFileSize = jarFile.length();
231       //in case null
232       jarVersion = jarVersion(sampleClass) + "";
233       
234       if (size == jarFileSize && StringUtils.equals(manifestVersion, jarVersion)
235           && StringUtils.equals(name, jarFile.getName())) {
236         LOG.debug("Found jarfile: " + jarFileFullName + " with correct size " + size + " and version: " + manifestVersion);
237         return;
238       }
239       
240     } catch (Exception e) {
241       //LOG.error("Error finding jar: " + name, e);
242       //e.printStackTrace();
243       //having problems
244     }
245     
246     //dont penalize activation.jar if the class is found...  sometimes its in java
247     //xpp3 just doesnt work
248     if (!StringUtils.equals("activation.jar", name) && !StringUtils.equals("xpp3_min.jar", name)) {
249       String error = "jarfile mismatch, expecting name: '" + name + "' size: " + size 
250         + " manifest version: " + manifestVersion + ".  However the jar detected is: " 
251         + jarFileFullName + ", name: " + jarFileName + " size: " + jarFileSize
252         + " manifest version: " + jarVersion;
253       System.err.println("Grouper warning: " + error);
254       LOG.warn(error);
255     }
256   }

258   /**
259    * make sure a resource is on the resource path
260    * @param resourcePath
261    * @return false if problem or if not checking configs
262    */
263   public static boolean checkResource(String resourcePath) {
264     if (configCheckDisabled()) {
265       return false;
266     }
267     try {
268       URL url = GrouperUtil.computeUrl(resourcePath, false);
269       if (url != null) {
270         LOG.debug("Found resource: " + url);
271         return true;
272       }
273     } catch (Exception e) {
274       //this means it cant be found
275     }
276     String error = "Cant find required resource on classpath: " + resourcePath;
277     //this is serious, lets go out and error
278     System.err.println("Grouper error: " + error);
279     LOG.error(error);
280     return false;
281   }

290   public static boolean configCheckDisabled() {
291     if (disableConfigCheck == null) {
292       //see if we shouldnt do this (but dont use ApiConfig API)
293       try {
294         Properties properties = GrouperConfig.retrieveConfig().properties();
295         String detectErrorsKey = "configuration.detect.errors";
296         String detectErrors = GrouperUtil.propertiesValue(properties, detectErrorsKey);
297         if (!GrouperUtil.booleanValue(detectErrors, true)) {
298           String warning = "Not checking configuration integrity due to grouper.properties: " 
299               + detectErrorsKey;
300           System.err.println("Grouper warning: " + warning);
301           LOG.warn(warning);
302           disableConfigCheck = true;
303         }
304       } catch (Exception e) {
305         //cant read grouper properties
306       }
307       if (disableConfigCheck == null) {
308         disableConfigCheck = false;
309       }
310     }
311     return disableConfigCheck;
312   }

453   /**
454    * make sure configured groups are there 
455    */
456   public static void checkGroups() {
457     
458     boolean wasInCheckConfig = inCheckConfig;
459     if (!wasInCheckConfig) {
460       inCheckConfig = true;
461     }
462     
463     //groups auto-create
464     //#configuration.autocreate.group.name.0 = etc:uiUsers
465     //#configuration.autocreate.group.description.0 = users allowed to log in to the UI
466     //#configuration.autocreate.group.subjects.0 = johnsmith
467     int i=0;
468     
469     GrouperSession grouperSession = null;
470     boolean startedGrouperSession = false;
471     try {
472       grouperSession = GrouperSession.staticGrouperSession(false);
473 
474       if (grouperSession == null) {
475         grouperSession = GrouperSession.startRootSession();
476         startedGrouperSession = true;
477       }
478       
479       while(true) {
480         String groupName = null;
481         try {
482           String groupNameKey = "configuration.autocreate.group.name." + i;
483           groupName = GrouperConfig.retrieveConfig().propertyValueString(groupNameKey);
484           
485           if (StringUtils.isBlank(groupName)) {
486             break;
487           }
488           
489           String groupDescription = GrouperConfig.retrieveConfig().propertyValueString("configuration.autocreate.group.description." + i);
490           String subjectsKey = "configuration.autocreate.group.subjects." + i;
491           String subjects = GrouperConfig.retrieveConfig().propertyValueString(subjectsKey);
492     
493           Group[] theGroup = new Group[1];
494           //first the group
495           checkGroup(grouperSession, groupName, wasInCheckConfig, true, wasInCheckConfig, null, groupDescription, "grouper.properties key " + groupNameKey, theGroup);
496           //now the subjects
497           if (!StringUtils.isBlank(subjects)) {
498             String[] subjectArray = GrouperUtil.splitTrim(subjects, ",");
499             for (String subjectId : subjectArray) {
500               
501               try {
502                 Subject subject = SubjectFinder.findByIdOrIdentifier(subjectId, false);
503                 boolean added = theGroup[0].addMember(subject, false);
504                 if (added && wasInCheckConfig) {
505                   String error = "auto-added subject " + subjectId + " to group: " + theGroup[0].getName();
506                   System.err.println("Grouper warning: " + error);
507                   LOG.warn(error);
508                 }
509               } catch (MemberAddException mae) {
510                 throw new RuntimeException("this should never happen", mae);
511               } catch (InsufficientPrivilegeException snfe) {
512                 throw new RuntimeException("this should never happen", snfe);
513               } catch (SubjectNotFoundException snfe) {
514                 throw new RuntimeException("this should never happen", snfe);
515               } catch (SubjectNotUniqueException snue) {
516                 String error = "subject not unique from grouper.properties key: " + subjectsKey + ", " + subjectId;
517                 System.err.println("Grouper error: " + error);
518                 LOG.error(error, snue);
519               }
520             }
521               
522           }
523         } catch (RuntimeException re) {
524           GrouperUtil.injectInException(re, ", problem with auto-create group: " + groupName);
525         }
526         i++;
527       }
528       {
529         boolean useWheel = GrouperConfig.retrieveConfig().propertyValueBoolean("groups.wheel.use", false);
530         if (useWheel) {
531           String wheelName = GrouperConfig.retrieveConfig().propertyValueString("groups.wheel.group");
532           if (StringUtils.isBlank(wheelName) && wasInCheckConfig) {
533             String error = "grouper.properties property groups.wheel.group should not be blank if groups.wheel.use is true";
534             System.err.println("Grouper error: " + error);
535             LOG.warn(error);
536           } else {
537             checkGroup(grouperSession, wheelName, wasInCheckConfig, null, wasInCheckConfig, null, "system administrators with all privileges", 
538                 "wheel group from grouper.properties key: groups.wheel.group", null);
539           }
540         }
541       }      
542       {
543         boolean useViewonlyWheel = GrouperConfig.retrieveConfig().propertyValueBoolean("groups.wheel.viewonly.use", false);
544         if (useViewonlyWheel) {
545           String wheelViewonlyName = GrouperConfig.retrieveConfig().propertyValueString("groups.wheel.viewonly.group");
546           if (StringUtils.isBlank(wheelViewonlyName) && wasInCheckConfig) {
547             String error = "grouper.properties property groups.wheel.viewonly.group should not be blank if groups.wheel.viewonly.use is true";
548             System.err.println("Grouper error: " + error);
549             LOG.warn(error);
550           } else {
551             checkGroup(grouperSession, wheelViewonlyName, wasInCheckConfig, null, wasInCheckConfig, null, "system administrators with view privileges", 
552                 "viewonly wheel group from grouper.properties key: groups.wheel.viewonly.group", null);
553           }
554         }
555       }      
556       {
557         boolean useReadonlyWheel = GrouperConfig.retrieveConfig().propertyValueBoolean("groups.wheel.readonly.use", false);
558         if (useReadonlyWheel) {
559           String wheelReadonlyName = GrouperConfig.retrieveConfig().propertyValueString("groups.wheel.readonly.group");
560           if (StringUtils.isBlank(wheelReadonlyName) && wasInCheckConfig) {
561             String error = "grouper.properties property groups.wheel.readonly.group should not be blank if groups.wheel.readonly.use is true";
562             System.err.println("Grouper error: " + error);
563             LOG.warn(error);
564           } else {
565             checkGroup(grouperSession, wheelReadonlyName, wasInCheckConfig, null, wasInCheckConfig, null, "system administrators with read privileges", 
566                 "readonly wheel group from grouper.properties key: groups.wheel.readonly.group", null);
567           }

1449             break OUTER;

1451         }

1455

1458           LOG.info(jarFile.getName() + ", " + key + ": " + attributes.getValue((Name)key));
