185 	public ActionForward grouperExecute(ActionMapping mapping, ActionForm form,
186 			HttpServletRequest request, HttpServletResponse response,
187 			HttpSession session, GrouperSession grouperSession)
188 			throws Exception {
189 		DynaActionForm groupForm = (DynaActionForm) form;
190 		boolean groupExists = false;
191 		String curNode = (String)groupForm.get("groupId");
192 
193 		if (curNode == null || "".equals(curNode)) {
194 			//Need to find the stem in which new group wuill be created
195 			
196 			curNode = getBrowseNode(session);
197 		} else {
198 			groupExists = true;
199 		}
200 		if (curNode == null || "".equals(curNode)) {
201 			String defaultStem = getDefaultRootStemName(session);
202 			Stem root = StemFinder.findByName(grouperSession, defaultStem, true);
203 			curNode = root.getUuid();
204 		}
205 
206 		//TODO: should be checked by the API and exception thrown?
207 		String groupName = (String) groupForm.get("groupName");
208 		if (!groupExists && !groupName.matches("[^ \"<>\\*]+")) {
209 			request.setAttribute("message", new Message(
210 					"groups.message.error.invalid-char", true));
211 			if (groupExists) {
212 				return mapping.findForward(FORWARD_EditAgain);
213 			} else {
214 				return mapping.findForward(FORWARD_CreateAgain);
215 			}
216 		}
217 		if ("".equals(groupForm.get("groupDisplayName")))
218 			groupForm.set("groupDisplayName", groupName);
219 		Group group = null;
220 		
221 		String extension = (String) groupForm.get("groupName");
222 		String displayExtension = (String) groupForm.get("groupDisplayName");
223 		String alternateName = (String) groupForm.get("groupAlternateName");
224 
225 		if(isEmpty(displayExtension))displayExtension=extension;
226 		//TODO: should be transactional - so add map or List of attributes
227 		Map assignedPrivs=null;
228 		Subject grouperAll = SubjectFinder.findById("GrouperAll", true);
229 		if (groupExists) {
230 			group = GroupFinder.findByUuid(grouperSession, curNode, true);
231 			Set<GroupType> removableTypes = group.getRemovableTypes();
232 			group.setDisplayExtension(displayExtension);
233 			group.setExtension(extension);
234 
235 			String oldAlternateName = null;
236 			Iterator<String> alternateNames = group.getAlternateNames().iterator();
237 			if (alternateNames.hasNext()) {
238 				oldAlternateName = alternateNames.next();
239 			}
240 
241 			if (isEmpty(alternateName) && oldAlternateName != null) {
242 				group.deleteAlternateName(oldAlternateName);
243 			} else if (!isEmpty(alternateName) && (oldAlternateName == null || !oldAlternateName.equals(alternateName))) {
244 				try {
245 					group.addAlternateName(alternateName);
246 				} catch (GroupModifyException e) {
247 				  setTransactionRollback(true);
248 					request.setAttribute("message", new Message(
249 						"groups.message.error.alternate-name-problem", true));
250 					return mapping.findForward(FORWARD_EditAgain);
251 				}
252 			}
253 
254                         try {
255 			group.store();
256                         } catch (GroupModifyAlreadyExistsException e) {
257                           setTransactionRollback(true);
258                           request.setAttribute("message", new Message("groups.message.error.update-problem-already-exists", true));
259                           return mapping.findForward(FORWARD_EditAgain);
260                         }
261 
262 			//do this after the store, in case there were types added in the hook...
263 			doTypes(group,request, removableTypes);
264 
265 			Map selectedPrivs = GrouperHelper.getImmediateHas(grouperSession,GroupOrStem.findByGroup(grouperSession,group),MemberFinder.findBySubject(grouperSession,grouperAll, true));
266 			assignedPrivs=new HashMap();
267 			Map.Entry entry;
268 			String key;
269 			Iterator it = selectedPrivs.entrySet().iterator();
270 			while(it.hasNext()) {
271 				entry = (Map.Entry)it.next();
272 				key = (String)entry.getKey();
273 				assignedPrivs.put(key,entry.getValue());
274 			}
275 		} else {
276 			GroupOrStem curGos = GroupOrStem.findByID(grouperSession,curNode);
277 			Stem parent=null;
278 			if(curGos.isStem()) {
279 				parent = curGos.getStem();
280 			}else{
281 				parent = curGos.getGroup().getParentStem();
282 			}
283 			try{
284 				group = parent.addChildGroup(extension,displayExtension );
285 
286 			}catch(HookVeto hookVeto) {
287 			  setTransactionRollback(true);
288 			  //this action was vetoed, put explanation on screen, and go back
289 			  Message.addVetoMessageToScreen(request, hookVeto);
290 		    return mapping.findForward(FORWARD_CreateAgain);
291 			  
292 			}catch(GroupAddException e) {
293 			  setTransactionRollback(true);
294 				String name = parent.getName() + GrouperHelper.HIER_DELIM + extension;
295 				request.setAttribute("message", new Message(
296 							"groups.message.error.add-problem",new String[] {e.getMessage()}, true));
297 				return mapping.findForward(FORWARD_CreateAgain);
298 			}
299 			doTypes(group,request, new HashSet<GroupType>());
300 			groupForm.set("groupId", group.getUuid());
301 			assignedPrivs = GrouperHelper.getDefaultAccessPrivsForGrouperAPI();
302 		}
303 		String [] privileges = request.getParameterValues("privileges");
304 		
305 		Map selectedPrivs = new HashMap();
306 		if(privileges==null) privileges = new String[]{};
307 		
308 			
309 		for(int i=0;i<privileges.length;i++) {
310 			selectedPrivs.put(privileges[i],Boolean.TRUE);
311 			//Check if priv was automatically assigned...
312 			if(!Boolean.TRUE.equals(assignedPrivs.get(privileges[i]))){
313 				group.grantPriv(grouperAll,Privilege.getInstance(privileges[i]));
314 			}
315 		}
316 		//if(0==0) throw new RuntimeException("Forced error for transaction testing");
317 		Map.Entry entry = null;
318 		Iterator it = assignedPrivs.entrySet().iterator();
319 		while(it.hasNext()) {
320 			entry = (Map.Entry)it.next();
321 			if(!Boolean.TRUE.equals(selectedPrivs.get(entry.getKey()))) {
322 				group.revokePriv(grouperAll,Privilege.getInstance((String)entry.getKey()));
323 			}
324 		}
325 		
326 		
327 		
328 		
329 		//TODO: are both these necessary?
330 		request.setAttribute("groupId", group.getUuid());
331 		groupForm.set("groupId",group.getUuid());
332 		try {
333 			if("true".equals(GrouperUiFilter.retrieveSessionMediaResourceBundle().getString("put.in.session.updated.groups"))) {
334 				addSavedSubject(session,group.toSubject());
335 			}
336 		}catch(Exception e){}
337 		String val  =(String) groupForm.get("groupDescription");
338 		if("".equals(val)) val=null;
339 
340 		if(val!=null) {
341 		  group.setDescription(val);
342 		  group.store();
343 		}
344 
345 		request.setAttribute("message", new Message(
346 				"groups.message.group-saved", group.getDisplayExtension()));
347 		
348 		//TODO: more sophistication in determining where to go?
349 		String submit = request.getParameter("submit.save");
350 
351 		if (submit != null) {
352 			return mapping.findForward(FORWARD_GroupSummary);
353 		}
354 		submit=request.getParameter("submit.saveAndAddComposite");
355 		if (submit != null) {
356 			return mapping.findForward(FORWARD_AddComposite);
357 		}
358 		session.setAttribute("findForNode", group.getUuid());
359 		return mapping.findForward(FORWARD_FindMembers);
360 
361 	}
362 	
363 	private void doTypes(Group group,HttpServletRequest request, Set<GroupType> curGroupTypes) throws Exception {
364 		
365 		String[] selectedGroupTypes = request.getParameterValues("groupTypes");
366 		Set selected = new HashSet();
367 		GroupType type;
368 		if(selectedGroupTypes!=null) {
369 			for(int i=0;i<selectedGroupTypes.length;i++) {
370 				type = GroupTypeFinder.find(selectedGroupTypes[i], true);
371 				selected.add(type);
372 				if(!curGroupTypes.contains(type)) group.addType(type);
373 			}
374 		}
375 		Iterator it = curGroupTypes.iterator();
376 		while(it.hasNext()) {
377 			type=(GroupType)it.next();
378 			if(!selected.contains(type) && !"base".equals(type.getName())) group.deleteType(type);
379 		}
380 	}
