68  /**
69   * Controller for menu that reads files configured through the media.resources key menu.resource.files.
70   * See resources/grouper/menu-items.xml for the xml format.
71   * The order of menu items configured through key 'menu.order'
72   * menu.cache determines if menu is cached for users - use false for development
73   * if changing source xml files
74   * 
75   * Since 1.2.1 a new mechanism has been put in place to control which users get which menu items. The
76   * media.resources key menu.filters defines a space separated list of MenuFilter. Each filter has a chance
77   * to veto a menu item. Currently, two MenuFilter implementations are provided:
78   * <ul><li>edu.internet2.middleware.grouper.ui.RootMenuFilter</li>
79   * <li>edu.internet2.middleware.grouper.ui.GroupMembershipMenuFilter</li></ul>
80   * The latter is configured through an XML configuration file - 
81  @see edu.internet2.middleware.grouper.ui.GroupMembershipMenuFilter
82   * 
83  
84   * <p/>
85   * 
86   * @author Gary Brown.
87   * @version $Id: PrepareMenuAction.java,v 1.7 2009-08-12 04:52:14 mchyzer Exp $
88   */
89  public class PrepareMenuAction extends LowLevelGrouperCapableAction {
90  	protected static final Log LOG = LogFactory.getLog(PrepareMenuAction.class);
91  
92    /**
93     * 
94     * @see edu.internet2.middleware.grouper.ui.actions.LowLevelGrouperCapableAction#grouperExecute(org.apache.struts.action.ActionMapping, org.apache.struts.action.ActionForm, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse, javax.servlet.http.HttpSession, edu.internet2.middleware.grouper.GrouperSession)
95     */
96  	@Override
97    public ActionForward grouperExecute(ActionMapping mapping, ActionForm form,
98  			HttpServletRequest request, HttpServletResponse response,
99  			HttpSession session, GrouperSession grouperSession)
100 			throws Exception {
101 		
102 		ResourceBundle mediaResources = GrouperUiFilter.retrieveSessionMediaResourceBundle();
103 		String menuFiles = mediaResources.getString("menu.resource.files");
104 		String menuOrder = mediaResources.getString("menu.order");
105 		String menuCache = mediaResources.getString("menu.cache");
106 		if(LOG.isDebugEnabled()) {
107 			LOG.debug("menu.resource.files=" + menuFiles);
108 			LOG.debug("menu.order=" + menuOrder);
109 			LOG.debug("menu.cache=" + menuCache);
110 		}
111 		boolean useCache=false;
112 		if("true".equals(menuCache)) {
113 			Object cachedMenu = session.getAttribute("cachedMenu");
114 			if(cachedMenu!=null) {
115 				request.setAttribute("menuItems",cachedMenu);
116 				return null;
117 			}
118 			useCache=true;
119 		}
120 		List<Map<String, String>> menu = new ArrayList<Map<String, String>>();
121 		Map<String, Map<String, String>> menuItems = new LinkedHashMap<String, Map<String, String>>();
122 		String[] inputResources = menuFiles.split(" ");
123 		Document menuDom = null;
124 		Element itemElement = null;
125 		Map<String, String> itemMap = null;
126 		Iterator menuIterator = null;
127 		Collection menuItemElements = null;
128 		Attr attribute = null;
129 		for(int i=0;i<inputResources.length;i++) {
130 			LOG.debug("Reading menu file: " + inputResources[i]);
131 			menuDom = DOMHelper.getDomFromResourceOnClassPath(inputResources[i]);
132 			menuItemElements = DOMHelper.getImmediateElements(menuDom.getDocumentElement(),"item");
133 			menuIterator= menuItemElements.iterator();
134 			while(menuIterator.hasNext()) {
135 				itemElement = (Element)menuIterator.next();
136 				itemMap = new HashMap<String, String>();
137 				NamedNodeMap attributes = itemElement.getAttributes();
138 				for(int j=0;j<attributes.getLength();j++) {
139 					attribute = (Attr)attributes.item(j);
140 					if(attribute.getName().equals("functionalArea")) {
141 						menuItems.put(attribute.getValue(),itemMap);
142 					}
143 					itemMap.put(attribute.getName(),attribute.getValue());
144 				}
145 			}
146 		}
147 		String[] order = menuOrder.split(" ");
148 		Map<String, String> orderedItem = null;
149 		for(int i=0;i<order.length;i++) {
150 			orderedItem = menuItems.get(order[i]);
151 			if(orderedItem!=null && isValidMenuItem(orderedItem,grouperSession,request)) {
152 				LOG.debug(order[i] + " added to menu");
153 				menu.add(orderedItem);
154 			}
155 		}
156 		request.setAttribute("menuItems",menu);
157 		if(useCache) {
158 		  session.setAttribute("cachedMenu",menu);
159 		}
160 		//store in session whether cached or not
161     session.setAttribute("menuMetaBean",new MenuMetaBean(menu));
162 		return null;
163 	}
164 	
165 	/**
166 	 * 
167 	 * @param item
168 	 * @param grouperSession
169 	 * @param request
170 	 * @return if
171 	 */
172 	protected boolean isValidMenuItem(Map<String, String> item,GrouperSession grouperSession,HttpServletRequest request) {
173 		if(item.containsKey("mediaKeyMustBeTrue")) {
174 	    if (!GrouperUiConfig.retrieveConfig().propertyValueBoolean(item.get("mediaKeyMustBeTrue"), false)) {
175         LOG.debug("Discarding " + item.get("functionalArea") + " since " + item.get("mediaKeyMustBeTrue") + "is not 'true'");
176         return false;
177 			}
178 		}
179 		
180 		Set menuFilters = SessionInitialiser.getMenuFilters(request.getSession());
181 		if(menuFilters.isEmpty()) return true;
182 		Iterator it = menuFilters.iterator();
183 		MenuFilter filter;
184 		while(it.hasNext()) {
185 			filter=(MenuFilter)it.next();
186 			if(!filter.isValid(grouperSession, item, request)) {
187 				LOG.debug("Discarding " + item.get("functionalArea") + " - rejected by " + filter.getClass().getSimpleName());
188 				return false;
189 			}
190 		}
191 		//if((isActiveWheelGroupMember(request.getSession())||"GrouperSystem".equals(SessionInitialiser.getGrouperSession(request.getSession()).getSubject().getId())) && "false".equals(item.get("forAdmin"))) return false;
192 		return true;
193 	}
