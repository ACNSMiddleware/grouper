38  import javax.servlet.http.HttpServletResponse;
39  import javax.servlet.http.HttpSession;

41  import org.apache.struts.action.ActionForm;
42  import org.apache.struts.action.ActionForward;
43  import org.apache.struts.action.ActionMapping;
44  
45  
46  import edu.internet2.middleware.grouper.GrouperSession;
47  import edu.internet2.middleware.grouper.ui.Message;
48  
49  /**
50   * Top level Strut's action which launches JSP editor configured through 
51   * populateDebugPrefs. Should only be used on local development system. 
52   * <p />
53   * 
54   * @author Gary Brown.
55   * @version $Id: EditJSPAction.java,v 1.3 2009-03-02 13:44:42 isgwb Exp $
56   */
57  public class EditJSPAction extends GrouperCapableAction {
58  
59  	//------------------------------------------------------------ Local
60  	// Forwards
61  	static final private String FORWARD_Editor = "Editor";
62  	static final private String FORWARD_DebugNotAllowed = "NotAllowed";
63  
64  	//------------------------------------------------------------ Action
65  	// Methods
66  
67  	public ActionForward grouperExecute(ActionMapping mapping, ActionForm form,
68  			HttpServletRequest request, HttpServletResponse response,
69  			HttpSession session, GrouperSession grouperSession)
70  			throws Exception {
71  		
72  		session.removeAttribute("subtitle");
73  		session.setAttribute("title", "prefs.debug.title");
74  		if(!Boolean.TRUE.equals(session.getAttribute("enableHtmlEditor"))){
75  			addMessage(new Message("debug.error.editor-not-allowed",true), request);
76  			return mapping.findForward(FORWARD_DebugNotAllowed);
77  		}
78  		Map prefs = (Map)session.getAttribute("debugPrefs");
79  		
80  		String jsp=request.getParameter("jsp");
81  		//Launches editor
82  		Runtime.getRuntime().exec(new String[]{(String)prefs.get("JSPEditor"),jsp});
83  		//Web page should be launched in new window which closes itself
84  		return mapping.findForward(FORWARD_Editor);
85  	}
86  
87  }
