233 public class PopulateGroupMemberAction extends GrouperCapableAction {
234 	protected static final Log LOG = LogFactory.getLog(PopulateGroupMemberAction.class);
235 	//------------------------------------------------------------ Local
236 	// Forwards
237 	static final private String FORWARD_GroupMembers = "GroupMembers";
238 
239 	static final private String FORWARD_GroupMember = "GroupMember";
240 
241 	static final private String FORWARD_StemMembers = "StemMembers";
242 
243 	static final private String FORWARD_StemMember = "StemMember";
244 
245 	//------------------------------------------------------------ Action
246 	// Methods
247 
248 	public ActionForward grouperExecute(ActionMapping mapping, ActionForm form,
249 			HttpServletRequest request, HttpServletResponse response,
250 			HttpSession session, GrouperSession grouperSession)
251 			throws Exception {
252 		NavExceptionHelper neh=getExceptionHelper(session);
253 		//We are using same class for stems and groups, but action is different
254 		//we distinguish by checking parameter
255 		
256 		
257 		boolean forStems = "stems".equals(mapping.getParameter());
258 		DynaActionForm groupOrStemMemberForm = (DynaActionForm) form;
259 		saveAsCallerPage(request,groupOrStemMemberForm,"findForNode");
260 		String asMemberOf = (String) groupOrStemMemberForm.get("asMemberOf");
261 		String contextGroupId = (String) groupOrStemMemberForm.get("contextGroup");
262 		GroupOrStem contextGroup=null;
263 		
264 		String listField = request.getParameter("listField");
265 		String membershipField = "members";
266 		
267 		if(!isEmpty(listField)) {
268 			membershipField=listField;
269 			request.setAttribute("listField",listField);
270 		}
271 		Field mField = null;
272 		try {
273 			mField=FieldFinder.find(membershipField, true);
274 		}catch(SchemaException e) {
275 			LOG.error("Error retrieving " + membershipField,e);
276 			throw new UnrecoverableErrorException("error.group-member.bad-field",e,membershipField);
277 		}
278 		if(!isEmpty(contextGroupId)) {
279 			contextGroup = GroupOrStem.findByID(grouperSession,contextGroupId);
280 		}
281 		if(isEmpty(asMemberOf) && !isEmpty(contextGroupId)) asMemberOf = contextGroupId;
282 		
283 //		If not set explicitly, default to findForNode
284 		if (isEmpty(asMemberOf)) {
285 			asMemberOf = (String) session.getAttribute("findForNode");
286 			groupOrStemMemberForm.set("asMemberOf",asMemberOf);
287 		}
288 		//If not set explicitly, default to current browse location
289 		if (isEmpty(asMemberOf)) {
290 			asMemberOf = getBrowseNode(session);
291 			groupOrStemMemberForm.set("asMemberOf",asMemberOf);
292 		}
293 		
294 		if(isEmpty(asMemberOf)) {
295 			String msg = neh.missingAlternativeParameters(contextGroupId,"contextGroupId",asMemberOf,"asMemberOf",asMemberOf,"findForNode");
296 			LOG.error(msg);
297 			throw new UnrecoverableErrorException("error.group-member.missing-grouporstem-id");
298 		}
299 		Group group = null;
300 		Stem stem = null;
301 		GroupOrStem groupOrStem = null;
302 		try{
303 			groupOrStem=GroupOrStem.findByID(grouperSession,asMemberOf);
304 		}catch(MissingGroupOrStemException e) {
305 			LOG.error(e);
306 			throw new UnrecoverableErrorException("error.group-member.bad-id",asMemberOf);
307 		}
308 		Map groupOrStemMap = null;
309 		//In UI we need to display set of possible privileges - with effective ones
310 		//pre-selected
311 		String[] possiblePrivs = null;
312 		String[] possibleEffPrivs = null;
313 		
314 		if(forStems) {
315 			stem= groupOrStem.getStem();
316 			groupOrStemMap=GrouperHelper.stem2Map(grouperSession,stem);
317 			session.setAttribute("subtitle","stems.action.edit-member");
318 			possiblePrivs = GrouperHelper.getStemPrivs(grouperSession);
319 			possibleEffPrivs=possiblePrivs;
320 		}else{
321 			group=groupOrStem.getGroup();
322 			groupOrStemMap=GrouperHelper.group2Map(grouperSession,group);
323 			session.setAttribute("subtitle","groups.action.edit-member");
324 			possiblePrivs = GrouperHelper.getGroupPrivs(grouperSession);
325 			possibleEffPrivs = GrouperHelper.getGroupPrivsWithMember(grouperSession);
326 		}
327 		
328 		
329 		request.setAttribute("subtitleArgs", new Object[] { groupOrStemMap.get("displayExtension")});
330 
331 		
332 		String subjectId = (String) groupOrStemMemberForm.get("subjectId");
333 		String subjectType = (String) groupOrStemMemberForm.get("subjectType");
334 		String sourceId = (String) groupOrStemMemberForm.get("sourceId");
335 		String msg = neh.missingParameters(subjectId,"subjectId",subjectType,"subjectTypeId",sourceId,"sourceId");
336 		if(msg!=null) {
337 			LOG.error(msg);
338 			throw new UnrecoverableErrorException("error.group-member.missing-subject-parameter");
339 		}
340 		//Retrieve subject whose privileges we are displaying and make
341 		//available to view
342 		Map subjectMap =null;
343 		try {
344 			subjectMap=GrouperHelper.subject2Map(grouperSession, subjectId,
345 					subjectType,sourceId);
346 		}catch(Exception e) {
347 			LOG.error("Error retrieving subject id=" + subjectId,e);
348 			throw new UnrecoverableErrorException("error.group-member.retrieve-subject",e,subjectId);
349 		}
350 		 
351 		
352 		//Retrieve privileges current user, and selected subject have over
353 		//current group/stem
354 				Member member = null;
355 		try {
356 			member=MemberFinder.findBySubject(grouperSession,SubjectFinder.findById(subjectId,subjectType,sourceId, true), true);
357 		}catch(SubjectNotFoundException e) {
358 			Subject unresolvable = new UnresolvableSubject(subjectId,subjectType,sourceId);
359 			member = MemberFinder.findBySubject(grouperSession,unresolvable, true);
360 		}
361     Map authUserPrivs = GrouperHelper.hasAsMap(grouperSession, groupOrStem,false);
362 		Map privs = GrouperHelper.hasAsMap(grouperSession, groupOrStem, member,mField); 
363 		Map extendedPrivs = GrouperHelper.getExtendedHas(grouperSession,groupOrStem,member,mField);
364 		Map immediatePrivs = GrouperHelper.getImmediateHas(grouperSession,groupOrStem,member,mField);
365 		if(!forStems) {
366 			UIGroupPrivilegeResolver resolver = 
367 				UIGroupPrivilegeResolverFactory.getInstance(grouperSession, 
368 				    GrouperUiFilter.retrieveSessionMediaResourceBundle(), 
369 					                                    	group, grouperSession.getSubject());
370 			request.setAttribute("groupPrivResolver", resolver.asMap());
371 		}
372 		
373 		//The actual privileges the subject has
374 		String privileges[] = new String[immediatePrivs.size()];
375 		Iterator it = immediatePrivs.keySet().iterator();
376 		String privilege;
377 		int pos = 0;
378 		while (it.hasNext()) {
379 			privilege = (String) it.next();
380 			privileges[pos++] = privilege;
381 		}
382 		groupOrStemMemberForm.set("privileges", privileges);
383 		//Set up attributes required by view
384 		request.setAttribute("subject", subjectMap);
385 		request.setAttribute("authUserPriv", authUserPrivs);
386 		request.setAttribute("subjectPriv", privs);
387 		request.setAttribute("extendedSubjectPriv", extendedPrivs);
388 		
389 		request.setAttribute("possiblePrivs", possiblePrivs);
390 		request.setAttribute("possibleEffectivePrivs", possibleEffPrivs);
391 		
392 		if(contextGroup==null) {
393 			request.setAttribute("browseParent", groupOrStemMap);
394 		}else{
395 			request.setAttribute("browseParent", GrouperHelper.group2Map(grouperSession,contextGroup));
396 		}
397 		
398 		if (forStems) {
399 			return mapping.findForward(FORWARD_StemMember);
400 		}
401 
402 		return mapping.findForward(FORWARD_GroupMember);
403 	}
