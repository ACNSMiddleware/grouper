29    /**
30     * convert a subject to string safely
31     * @param subject
32     * @return the string value of subject (might be null)
33     */
34    public static String subjectToString(Subject subject) {
35      if (subject == null) {
36        return null;
37      }
38      try {
39        return "Subject id: " + subject.getId() + ", sourceId: " + subject.getSource().getId();
40      } catch (RuntimeException e) {
41        //might be subject not found if lazy subject
42        return subject.toString();
43      }
44    }

46    /**
47     * Null safe array length or map
48     *
49     * @param arrayOrCollection
50     * @return the length of the array (0 for null)
51     */
52    public static int length(Object arrayOrCollection) {
53      if (arrayOrCollection == null) {
54        return 0;
55      }
56      if (arrayOrCollection.getClass().isArray()) {
57        return Array.getLength(arrayOrCollection);
58      }
59      if (arrayOrCollection instanceof Collection) {
60        return ((Collection) arrayOrCollection).size();
61      }
62      if (arrayOrCollection instanceof Map) {
63        return ((Map) arrayOrCollection).size();
64      }
65      // simple non array non collection object
66      return 1;
67    }

69    /**
70     * return a set of objects from varargs.
71     *
72     * @param <T> template type of the objects
73     * @param objects
74     * @return the set
75     */
76    public static <T> Set<T> toSet(T... objects) {
77      if (objects == null) {
78        return null;
79      }
80      Set<T> result = new LinkedHashSet<T>();
81      for (T object : objects) {
82        result.add(object);
83      }
84      return result;
85    }

87    /**
88     * if there is no driver class specified, then try to derive it from the URL
89     * @param connectionUrl
90     * @param driverClassName
91     * @return the driver class
92     */
93    public static String convertUrlToDriverClassIfNeeded(String connectionUrl, String driverClassName) {
94      //default some of the stuff
95      if (StringUtils.isBlank(driverClassName)) {
96        
97        if (isHsql(connectionUrl)) {
98          driverClassName = "org.hsqldb.jdbcDriver";
99        } else if (isMysql(connectionUrl)) {
100         driverClassName = "com.mysql.jdbc.Driver";
101       } else if (isOracle(connectionUrl)) {
102         driverClassName = "oracle.jdbc.driver.OracleDriver";
103       } else if (isPostgres(connectionUrl)) { 
104         driverClassName = "org.postgresql.Driver";
105       } else if (isSQLServer(connectionUrl)) {
106         driverClassName = "com.microsoft.sqlserver.jdbc.SQLServerDriver";
107       } else {
108         
109         //if this is blank we will figure it out later
110         if (!StringUtils.isBlank(connectionUrl)) {
111         
112           String error = "Cannot determine the driver class from database URL: " + connectionUrl;
113           System.err.println(error);
114           log.error(error);
115           return null;
116         }
117       }
118     }
119     return driverClassName;
120   
121   }

173   /**
174      * retrieve a batch by 0 index. Will return an array of size batchSize or
175      * the remainder. the array will be full of elements. Note, this requires an
176      * ordered input (so use linkedhashset not hashset if doing sets)
177      * @param <T> template type
178      * @param collection
179      * @param batchSize
180      * @param batchIndex
181      * @return the list
182      *         This never returns null, only empty list
183      */
184     @SuppressWarnings("unchecked")
185     public static <T> List<T> batchList(List<T> collection, int batchSize,
186         int batchIndex) {
187   
188       int numberOfBatches = batchNumberOfBatches(collection, batchSize);
189       int arraySize = length(collection);
190   
191       // short circuit
192       if (arraySize == 0) {
193         return new ArrayList<T>();
194       }
195   
196       List<T> theBatchObjects = new ArrayList<T>();
197   
198       // lets get the type of the first element if possible
199   //    Object first = get(arrayOrCollection, 0);
200   //
201   //    Class theType = first == null ? Object.class : first.getClass();
202   
203       // if last batch
204       if (batchIndex == numberOfBatches - 1) {
205   
206         // needs to work to 1-n
207         //int thisBatchSize = 1 + ((arraySize - 1) % batchSize);
208   
209         int collectionIndex = 0;
210         for (T t : collection) {
211           if (collectionIndex++ < batchIndex * batchSize) {
212             continue;
213           }
214           //just copy the rest
215           //if (collectionIndex >= (batchIndex * batchSize) + arraySize) {
216           //  break;
217           //}
218           //we are in the copy mode
219           theBatchObjects.add(t);
220         }
221   
222       } else {
223         // if non-last batch
224         //int newIndex = 0;
225         int collectionIndex = 0;
226         for (T t : collection) {
227           if (collectionIndex < batchIndex * batchSize) {
228             collectionIndex++;
229             continue;
230           }
231           //done with batch
232           if (collectionIndex >= (batchIndex + 1) * batchSize) {
233             break;
234           }
235           theBatchObjects.add(t);
236           collectionIndex++;
237         }
238       }
239       return theBatchObjects;
240     }

255    * If batching this is the number of batches
256    * @param count is size of set
257    * @param batchSize
258    * @return the number of batches
259    */
260   public static int batchNumberOfBatches(int count, int batchSize) {
261     //not sure why this would be 0...
262     if (batchSize == 0) {
263       return 0;
264     }
265     int batches = 1 + ((count - 1) / batchSize);
266     return batches;
267   
268   }

271    * convert a collection of strings (no parens) to an in clause
272    * @param collection
273    * @return the string of in clause (without parens)
274    */
275   public static String convertToInClauseForSqlStatic(Collection<String> collection) {
276     
277     StringBuilder result = new StringBuilder();
278     int collectionSize = collection.size();
279     for (int i = 0; i < collectionSize; i++) {
280       result.append("?");
281   
282       if (i < collectionSize - 1) {
283         result.append(", ");
284       }
285     }
286     return result.toString();
287   }

331   /**
332    * make sure a array is non null.  If null, then return an empty array.
333    * @param <T>
334    * @param array
335    * @param theClass to make array from
336    * @return the list or empty list if null
337    */
338   @SuppressWarnings({ "unchecked", "cast" })
339   public static <T> T[] nonNull(T[] array, Class<?> theClass) {
340     if (int.class.equals(theClass)) {
341       return (T[])(Object)new int[0];
342     }
343     if (float.class.equals(theClass)) {
344       return (T[])(Object)new float[0];
345     }
346     if (double.class.equals(theClass)) {
347       return (T[])(Object)new double[0];
348     }
349     if (short.class.equals(theClass)) {
350       return (T[])(Object)new short[0];
351     }
352     if (long.class.equals(theClass)) {
353       return (T[])(Object)new long[0];
354     }
355     if (byte.class.equals(theClass)) {
356       return (T[])(Object)new byte[0];
357     }
358     if (boolean.class.equals(theClass)) {
359       return (T[])(Object)new boolean[0];
360     }
361     if (char.class.equals(theClass)) {
362       return (T[])(Object)new char[0];
363     }
364     return array == null ? ((T[])Array.newInstance(theClass, 0)) : array;
365   }
