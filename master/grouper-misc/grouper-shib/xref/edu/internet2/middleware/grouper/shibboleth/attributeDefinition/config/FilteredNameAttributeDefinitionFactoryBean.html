30  package edu.internet2.middleware.grouper.shibboleth.attributeDefinition.config;
31  
32  import edu.internet2.middleware.grouper.shibboleth.attributeDefinition.FilteredNameAttributeDefinition;
33  import edu.internet2.middleware.grouper.shibboleth.filter.Filter;
34  import edu.internet2.middleware.shibboleth.common.config.attribute.resolver.attributeDefinition.BaseAttributeDefinitionFactoryBean;
35  
36  /** Spring factory bean for {@link FilteredNameAttributeDefinitionFactoryBean}s. */
37  public class FilteredNameAttributeDefinitionFactoryBean extends BaseAttributeDefinitionFactoryBean {
38  
39    /** Filters the values returned. */
40    private Filter<String> filter;
41  
42    /**
43     * Get the filter.
44     * 
45     * @return the filter
46     */
47    public Filter<String> getFilter() {
48      return filter;
49    }
50  
51    /**
52     * Set the filter.
53     * 
54     * @param filter the Filter
55     */
56    public void setFilter(Filter filter) {
57      this.filter = filter;
58    }
59  
60    /** {@inheritDoc} */
61    protected Object createInstance() throws Exception {
62      FilteredNameAttributeDefinition definition = new FilteredNameAttributeDefinition();
63      populateAttributeDefinition(definition);
64      definition.setFilter(filter);
65      return definition;
66    }
67  
68    /** {@inheritDoc} */
69    public Class getObjectType() {
70      return FilteredNameAttributeDefinition.class;
71    }
72  
73  }
