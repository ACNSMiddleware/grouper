52      /**
53       * How to find member objects (users, accounts, etc), EL based on a subject.
54       * 
55       * If either userSearchBaseDn or userSearchFilter  is null (which is the default), 
56       * then users will not be fetched. This is okay for ldap provisioning that only
57       * needs subject attributes. For instance, posixGroup provisioning only needs the
58       * usernames of the accounts, which might be the SubjectId or SubjectIdentifier.
59       * 
60       */
61      private String userSearchBaseDn;
62      protected String userSearchBaseDn_defaultValue = null;
63      private String userSearchFilter;
64      protected String userSearchFilter_defaultValue = null;
65      
66      // Where to create users if createMissingUsers is enabled
67      private String userCreationBaseDn;
68      protected String userCreationBaseDn_defaultValue = null;
69  
70      /** 
71       * LDIF used to create a new user. The DN of this LDIF should be relative
72       * to the userCreationBaseDn. 
73       */
74      private String userCreationLdifTemplate;
75      protected String userCreationLdifTemplate_defaultValue = null;

77      
78      private String ouCreationLdifTemplate;
79      protected String ouCreationLdifTemplate_defaultValue = "dn: ${dn}||ou: ${ou}||objectclass: organizationalunit";
80      
81      /** An array of attributes needed by the userSearchFilter. 
82       * These are used to make sure the cached user LdapObjects have the necessary data
83       * to run the userSearchFilter in memory after a bunch of users are found via bulk searching.
84       * 
85       * TODO: Parse the filter to see what attributes it uses.
86       * 
87       * By default this will be uid,mail,samAccountName
88       */
89      private String userSearchAttributes[];
90      protected String userSearchAttributes_defaultValue = "cn,uid,uidNumber,mail,samAccountName,objectclass";

104     /**
105      * How many ldap values should be changed (added or deleted) in each

120     @Override
121     public void readConfiguration() {
122       super.readConfiguration();
123       final String qualifiedParameterNamespace = PARAMETER_NAMESPACE + provisionerName + ".";
124 
125       LOG.debug("Ldap Provisioner - Setting properties for {} consumer/provisioner.", provisionerName);
126 
127       isActiveDirectory = GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "isActiveDirectory", isActiveDirectory_defaultValue);
128       LOG.debug("Ldap Provisioner {} - Setting isActiveDirectory to {}", provisionerName, isActiveDirectory);
129 
130       ldapPoolName = GrouperLoaderConfig.retrieveConfig().propertyValueStringRequired(qualifiedParameterNamespace + "ldapPoolName");
131       LOG.debug("Ldap Provisioner {} - Setting ldapPoolName to {}", provisionerName, ldapPoolName);
132       
133       maxValuesToChangePerOperation =
134               GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "maxValuesToChangePerOperation", maxValuesToChangePerOperation_defaultValue);
135       LOG.debug("Ldap Provisioner {} - Setting maxValuesToChangePerOperation to {}", provisionerName, maxValuesToChangePerOperation);
136 
137       userSearchBaseDn =
138           GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "userSearchBaseDn" , userSearchBaseDn_defaultValue);
139       LOG.debug("Ldap Attribute Provisioner {} - Setting userSearchBaseDn to {}", provisionerName, userSearchBaseDn);
140 
141       userSearchFilter =
142           GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "userSearchFilter" , userSearchFilter_defaultValue);
143       LOG.debug("Ldap Attribute Provisioner {} - Setting userSearchFilter to {}", provisionerName, userSearchFilter);
144 
145       String userSearchAttributeString =
146           GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "userSearchAttributes" , userSearchAttributes_defaultValue);
147       userSearchAttributes = userSearchAttributeString.trim().split(" *, *");
148       LOG.debug("Ldap Attribute Provisioner {} - Setting userSearchAttributes to {}", provisionerName, userSearchAttributes);
149       
150       userCreationBaseDn =
151           GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "userCreationBaseDn" , userCreationBaseDn_defaultValue);
152       LOG.debug("Ldap Attribute Provisioner {} - Setting userCreationBaseDn to {}", provisionerName, userCreationBaseDn);
153 
154       userCreationLdifTemplate =
155           GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "userCreationLdifTemplate" , userCreationLdifTemplate_defaultValue);
156       LOG.debug("Ldap Attribute Provisioner {} - Setting userCreationLdifTemplate to {}", provisionerName, userCreationLdifTemplate);
157       
158       ouCreationLdifTemplate =
159           GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "ouCreationLdifTemplate" , ouCreationLdifTemplate_defaultValue);
160       LOG.debug("Ldap Attribute Provisioner {} - Setting ouCreationLdifTemplate to {}", provisionerName, ouCreationLdifTemplate);
161       
162       ldapUserCacheTime_secs =
163           GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "ldapUserCacheTime_secs", ldapUserCacheTime_secs_defaultValue);
164       LOG.debug("Ldap Provisioner {} - Setting ldapUserCacheTime_secs to {}", provisionerName, ldapUserCacheTime_secs);
165   
166       ldapUserCacheSize =
167           GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "ldapUserCacheSize", ldapUserCacheSize_defaultValue);
168       LOG.debug("Ldap Provisioner {} - Setting ldapUserCacheSize to {}", provisionerName, ldapUserCacheSize);
169     }

171     public BlockingConnectionPool getLdapPool() {
172       if ( ldapPool != null )
173         return ldapPool;
174       
175       // We don't have a pool setup yet. Synchronize so we're sure we only make one pool.
176       synchronized(this) {
177         // Check if another thread has created our pool while we were waiting for the semaphore
178         if ( ldapPool != null )
179           return ldapPool;
180         
181        ldapPool = buildLdapConnectionPool();
182       }
183 
184       return ldapPool;
185     }
186 
187 
188     private BlockingConnectionPool buildLdapConnectionPool() {
189       BlockingConnectionPool result;
190 
191       String ldapPropertyPrefix = "ldap." + ldapPoolName.toLowerCase() + ".";
192       
193       for (String propName : GrouperLoaderConfig.retrieveConfig().propertyNames()) {
194         if ( propName.toLowerCase().startsWith(ldapPropertyPrefix) ) {
195           String propValue = GrouperLoaderConfig.getPropertyString(propName);
196           
197           // Get the part of the property after ldapPropertyPrefix 'ldap.person.'
198           String propNameTail = propName.substring(ldapPropertyPrefix.length());
199           ldaptiveProperties.put("org.ldaptive." + propNameTail, propValue);
200           
201           // Some compatibility between old vtldap properties and ldaptive versions
202           // url (vtldap) ==> ldapUrl
203           if ( propNameTail.equalsIgnoreCase("url") )
204             ldaptiveProperties.put("org.ldaptive.ldapUrl", propValue);
205           // tls (vtldap) ==> useStartTls
206           if ( propNameTail.equalsIgnoreCase("tls") )
207             ldaptiveProperties.put("org.ldaptive.useStartTLS", propValue);
208           // user (vtldap) ==> bindDn
209           if ( propNameTail.equalsIgnoreCase("user") )
210             ldaptiveProperties.put("org.ldaptive.bindDn", propValue);
211           // pass (vtldap) ==> bindCredential
212           if ( propNameTail.equalsIgnoreCase("pass") )
213             ldaptiveProperties.put("org.ldaptive.bindCredential", propValue);
214         }
215       }
216       
217       // Setup ldaptive ConnectionConfig
218       ConnectionConfig connConfig = new ConnectionConfig();
219       ConnectionConfigPropertySource ccpSource = new ConnectionConfigPropertySource(connConfig, ldaptiveProperties);
220       ccpSource.initialize();
221 
222       //GrouperLoaderLdapServer grouperLoaderLdapProperties 
223       //  = GrouperLoaderConfig.retrieveLdapProfile(ldapPoolName);
224       
225       /////////////
226       // Binding
227       BindConnectionInitializer binder = new BindConnectionInitializer();
228 
229       BindConnectionInitializerPropertySource bcip = new BindConnectionInitializerPropertySource(binder, ldaptiveProperties);
230       bcip.initialize();
231 
232       // I'm not sure if SaslRealm and/or SaslAuthorizationId can be used independently
233       // Therefore, we'll initialize gssApiConfig when either one of them is used.
234       // And, then, we'll attach the gssApiConfig to the binder if there is a gssApiConfig
235       GssApiConfig gssApiConfig = null;
236       String val = (String) ldaptiveProperties.get("org.ldaptive.saslRealm");
237       if (!StringUtils.isBlank(val)) {
238         if ( gssApiConfig == null )
239           gssApiConfig = new GssApiConfig();
240         gssApiConfig.setRealm(val);
241         
242       }
243       
244       val = (String) ldaptiveProperties.get("org.ldaptive.saslAuthorizationId");
245       if (!StringUtils.isBlank(val)) {
246         if ( gssApiConfig == null )
247           gssApiConfig = new GssApiConfig();
248         gssApiConfig.setAuthorizationId(val);
249       }
250 
251       // If there was a sasl/gssapi attribute, then save the gssApiConfig
252       if ( gssApiConfig != null )
253         binder.setBindSaslConfig(gssApiConfig);
254       
255       
256       /////////////
257       // PoolConfig
258       
259       PoolConfig ldapPoolConfig = new PoolConfig();
260       PoolConfigPropertySource pcPropertySource = new PoolConfigPropertySource(ldapPoolConfig, ldaptiveProperties);
261 
262       DefaultConnectionFactory factory = new DefaultConnectionFactory(connConfig);
263       factory.setProvider(new UnboundIDProvider());
264       result = new BlockingConnectionPool(factory);
265       
266       result.initialize();
267       return result;
268     }

290     public void addUserSearchAttribute(String attribute) {
291       for ( String a : userSearchAttributes )
292         if ( a.equalsIgnoreCase(attribute))
293           return;
294 
295       userSearchAttributes = Arrays.copyOf(userSearchAttributes, userSearchAttributes.length+1);
296       userSearchAttributes[userSearchAttributes.length-1] = attribute;
297     }
