20  import java.util.Map;
21  
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import edu.internet2.middleware.grouper.app.loader.GrouperLoaderConfig;

27  /**
28   * Collects all the various properties and makes them available to the provisioner.
29   *
30   * @author Bert Bee-Lindgren
31   */
32  public class ProvisionerConfiguration {
33  
34      private static final Logger LOG = LoggerFactory.getLogger(ProvisionerConfiguration.class);
35      public static final String PARAMETER_NAMESPACE = "changeLog.consumer.";
36      protected final String provisionerName;
37  
38      /**
39       * Should this provisioner be active, or should it simply report 
40       * success when changelog entries occur and skip full-sync requests?
41       */
42      private boolean enabled;
43      protected boolean enabled_defaultValue = true;
44      
45      /**
46       * Should groups in the groupSearchBaseDn/allGroupSearchFilter be removed 
47       * if they no longer exist in Grouper?
48       */
49      private boolean grouperIsAuthoritative;
50      protected boolean grouperIsAuthoritative_defaultValue = false;
51  
52      private int sleepTimeAfterError_ms;
53      protected int sleepTimeAfterError_ms_defaultValue = 1000;
54      
55      private int grouperDataCacheTime_secs;
56      protected int grouperGroupCacheTime_secs_defaultValue = 600;
57      
58      private int grouperGroupCacheSize, grouperSubjectCacheSize;
59      protected int grouperGroupCacheSize_defaultValue = 10000;
60      protected int grouperSubjectCacheSize_defaultValue = 10000;
61  
62      protected String groupSelectionExpression;
63      
64  
65  
66      // This expression says that the provisionerName has to be in a group or stem provision_to attribute
67      // and NOT in neither a group or stem do_not_provision_to attribute
68  
69      protected String groupSelectionExpression_defaultValue
70        = "${utils.containedWithin(provisionerName, stemAttributes['etc:pspng:provision_to'], groupAttributes['etc:pspng:provision_to']) " 
71            + "&& !utils.containedWithin(provisionerName, stemAttributes['etc:pspng:do_not_provision_to'], groupAttributes['etc:pspng:do_not_provision_to'])}";
72  
73      
74      // Does the provisioning process need User information from the target system or
75      // can all the changes be implemented with Subject attributes from Grouper?
76      private boolean needsTargetSystemUsers;
77      protected boolean needsTargetSystemUsers_defaultValue = false;
78  
79      private int userSearch_batchSize;
80      protected int userSearch_batchSize_defaultValue = 50;
81      
82      private boolean createMissingUsers;
83      protected boolean createMissingUsers_defaultValue = false;
84      
85      // Does the provisioning process need Group information from the target system or
86      // can all the changes be implemented with Group attributes from Grouper?
87      private boolean needsTargetSystemGroups;
88      protected boolean needsTargetSystemGroups_defaultValue = false;
89      
90      private boolean supportsEmptyGroups;
91      protected boolean supportsEmptyGroups_defaultValue = true;
92  
93      private int groupSearch_batchSize;
94      protected int groupSearch_batchSize_defaultValue = 50;
95  
96  
97      public ProvisionerConfiguration(String provisionerName) {
98        this.provisionerName = provisionerName;
99      }
100     
101     /**
102      * Populate the various settings from the configuration source(s). This is separated from our 
103      * constructor so that our subclasses can override default values. In other words, a subclass
104      * might construct us, change a _defaultValue ivar and then ask us to do 'readConfiguration'.
105      */
106     public void readConfiguration() {
107         final String qualifiedParameterNamespace = PARAMETER_NAMESPACE + provisionerName + ".";
108 
109         LOG.debug("Ldap Group Provisioner - Setting properties for {} consumer/provisioner.", provisionerName);
110 
111         enabled =
112             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "enabled", enabled_defaultValue);
113         LOG.debug("Provisioner {} - Setting enabled to {}", provisionerName, enabled);
114 
115         grouperIsAuthoritative =
116             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "grouperIsAuthoritative", grouperIsAuthoritative_defaultValue);
117         LOG.debug("Provisioner {} - Setting grouperIsAuthoritative to {}", provisionerName, grouperIsAuthoritative);
118 
119         grouperDataCacheTime_secs =
120             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "grouperGroupCacheTime_secs", grouperGroupCacheTime_secs_defaultValue);
121         LOG.debug("Provisioner {} - Setting grouperGroupCacheTime_secs to {}", provisionerName, grouperDataCacheTime_secs);
122     
123         sleepTimeAfterError_ms =
124             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "sleepTimeAfterError_ms", sleepTimeAfterError_ms_defaultValue);
125         LOG.debug("Provisioner {} - Setting sleepTimeAfterError_ms to {}", provisionerName, sleepTimeAfterError_ms);
126     
127         grouperGroupCacheSize =
128                 GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "grouperGroupCacheSize", grouperGroupCacheSize_defaultValue);
129         LOG.debug("Provisioner {} - Setting grouperGroupCacheSize to {}", provisionerName, grouperGroupCacheSize);
130 
131         grouperSubjectCacheSize =
132             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "grouperSubjectCacheSize", grouperSubjectCacheSize_defaultValue);
133         LOG.debug("Provisioner {} - Setting grouperSubjectCacheSize to {}", provisionerName, grouperSubjectCacheSize);
134         
135         createMissingUsers =
136             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "createMissingUsers", createMissingUsers_defaultValue);
137         LOG.debug("Provisioner {} - Setting createMissingUsers to {}", provisionerName, createMissingUsers);
138         
139         needsTargetSystemUsers =
140             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "needsTargetSystemUsers", needsTargetSystemUsers_defaultValue);
141         LOG.debug("Provisioner {} - Setting needsTargetSystemUsers to {}", provisionerName, needsTargetSystemUsers);
142 
143         needsTargetSystemGroups =
144             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "needsTargetSystemGroups", needsTargetSystemGroups_defaultValue);
145         LOG.debug("Provisioner {} - Setting needsTargetSystemGroups to {}", provisionerName, needsTargetSystemGroups);
146 
147         supportsEmptyGroups =
148             GrouperLoaderConfig.retrieveConfig().propertyValueBoolean(qualifiedParameterNamespace + "supportsEmptyGroups", supportsEmptyGroups_defaultValue);
149         LOG.debug("Provisioner {} - Setting supportsEmptyGroups to {}", provisionerName, supportsEmptyGroups);
150 
151         groupSelectionExpression =
152             GrouperLoaderConfig.retrieveConfig().propertyValueString(qualifiedParameterNamespace + "groupSelectionExpression", groupSelectionExpression_defaultValue);
153         LOG.debug("Ldap Provisioner {} - Setting groupSelectionExpression to {}", provisionerName, groupSelectionExpression);
154 
155         userSearch_batchSize =
156             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "userSearch_batchSize", userSearch_batchSize_defaultValue);
157         LOG.debug("Ldap Provisioner {} - Setting userSearch_batchSize to {}", provisionerName, userSearch_batchSize);
158 
159         groupSearch_batchSize =
160             GrouperLoaderConfig.retrieveConfig().propertyValueInt(qualifiedParameterNamespace + "groupSearch_batchSize", groupSearch_batchSize_defaultValue);
161         LOG.debug("Ldap Provisioner {} - Setting groupSearch_batchSize to {}", provisionerName, groupSearch_batchSize);
162     }
163 
164 
165     public boolean isEnabled() {
166     	return enabled;
167     }
168

170       return grouperDataCacheTime_secs;
171     }
172 
173     public int getGrouperGroupCacheSize() {
174       return grouperGroupCacheSize;
175     }
176 
177     public int getGrouperSubjectCacheSize() {
178       return grouperSubjectCacheSize;
179     }
180     
181     public boolean isCreatingMissingUsersEnabled() {
182       return createMissingUsers;
183     }
184     
185     public boolean areEmptyGroupsSupported() {
186       return supportsEmptyGroups;
187     }
188     
189     public int getUserSearch_batchSize() {
190       return userSearch_batchSize;
191     }
192     
193     public int getGroupSearch_batchSize() {
194       return groupSearch_batchSize;
195     }
196     
197     public boolean needsTargetSystemUsers() {
198       return needsTargetSystemUsers;
199     }
200 
201     public boolean needsTargetSystemGroups() {
202       return needsTargetSystemGroups;
203     }
204     
205     public String getGroupSelectionExpression() {
206       return groupSelectionExpression;
207     }
208 
209     public void populateElMap(Map<String, Object> variableMap) {
210       variableMap.put("provisionerName", provisionerName);
211     }
212 
213     public long getSleepTimeAfterError_ms() {
214       return sleepTimeAfterError_ms;
215     }
216     
217     public boolean isGrouperAuthoritative() {
218     	return grouperIsAuthoritative;
219     }
