219   /**
220    * pass in a grouper session
221    * @param grouperSession 
222    * @param exceptionOnNull excdeption if nothing in stem
223    */
224   public void retrieveStemIfNeeded(GrouperSession grouperSession, boolean exceptionOnNull) {
225     //see if we already retrieved
226     if (this.stemFindResult != null) {
227       return;
228     }
229     try {
230       //assume success (set otherwise if there is a problem)
231       this.stemFindResult = StemFindResult.SUCCESS;
232 
233       boolean hasUuid = !StringUtils.isBlank(this.uuid);
234 
235       boolean hasName = !StringUtils.isBlank(this.stemName);
236 
237       boolean hasIdIndex = !StringUtils.isBlank(this.idIndex);
238 
239       //maybe nothing and thats ok
240       if (!exceptionOnNull && !hasUuid && !hasName && !hasIdIndex) {
241         return;
242       }
243 
244       //must have a name or uuid
245       if (!hasUuid && !hasName && !hasIdIndex) {
246         this.stemFindResult = StemFindResult.INVALID_QUERY;
247         String logMessage = "Invalid query: " + this;
248         LOG.warn(logMessage);
249         return;
250       }
251 
252       Stem theStem = null;
253       
254       if (hasName) {
255         theStem = StemFinder.findByName(grouperSession, this.stemName, true, new QueryOptions().secondLevelCache(false));
256       } else if (hasUuid) {
257         theStem = StemFinder.findByUuid(grouperSession, this.uuid, true, new QueryOptions().secondLevelCache(false));
258       } else if (hasIdIndex) {
259         theStem = StemFinder.findByIdIndex(GrouperUtil.longValue(this.idIndex), true, new QueryOptions().secondLevelCache(false));
260       }
261 
262       //make sure matches
263       if ((hasUuid && !StringUtils.equals(this.uuid, theStem.getUuid()))
264           || (hasName && !StringUtils.equals(this.stemName, theStem.getName()))
265           || (hasIdIndex && !GrouperUtil.equals(GrouperUtil.longValue(this.idIndex), theStem.getIdIndex()))){
266         this.stemFindResult = StemFindResult.STEM_UUID_NAME_IDINDEX_DONT_MATCH;
267         String error = "Stem name '" + this.stemName + "', uuid '" + this.uuid
268             + "', idIndex: " + this.idIndex + " do not match";
269         String logMessage = "Invalid query: " + error + ", " + this;
270         LOG.warn(logMessage);
271       }
272       
273       this.stem = theStem;
274       
275     } catch (StemNotFoundException gnf) {
276       this.stemFindResult = StemFindResult.STEM_NOT_FOUND;
277     }
278

329   /**
330    * convert stem lookups to stem ids
331    * @param grouperSession
332    * @param wsStemLookups
333    * @param errorMessage
334    * @param lookupCount is an array of size one int where 1 will be added if there are records, and no change if not
335    * @return the stem ids
336    */
337   public static Set<String> convertToStemIds(GrouperSession grouperSession, WsStemLookup[] wsStemLookups, StringBuilder errorMessage, int[] lookupCount) {
338     //get all the stems
339     //we could probably batch these to get better performance.
340     Set<String> stemIds = null;
341     if (!GrouperServiceUtils.nullArray(wsStemLookups)) {
342       
343       stemIds = new LinkedHashSet<String>();
344       int i=0;
345 
346       boolean foundRecords = false;
347       
348       for (WsStemLookup wsStemLookup : wsStemLookups) {
349         
350         if (wsStemLookup == null || !wsStemLookup.hasData()) {
351           continue;
352         }
353 
354         if (!foundRecords) {
355           lookupCount[0]++;
356           foundRecords = true;
357         }
358 
359         wsStemLookup.retrieveStemIfNeeded(grouperSession, false);
360         Stem stem = wsStemLookup.retrieveStem();
361         if (stem != null) {
362           stemIds.add(stem.getUuid());
363         } else {
364           
365           if (errorMessage.length() > 0) {
366             errorMessage.append(", ");
367           }
368 
369           errorMessage.append("Error on stem index: " + i + ", " + wsStemLookup.retrieveStemFindResult() + ", " + wsStemLookup.toStringCompact());
370         }
371         
372         i++;
373       }
374       
375     }
376     return stemIds;
377   }

379   /**
380    * make sure this is an explicit toString
381    * @return return a compact to string
382    */
383   public String toStringCompact() {
384     if (!StringUtils.isBlank(this.stemName)) {
385       return "name: " + this.stemName;
386     }
387     if (!StringUtils.isBlank(this.uuid)) {
388       return "id: " + this.uuid;
389     }
390     if (!StringUtils.isBlank(this.idIndex)) {
391       return "idIndex: " + this.idIndex;
392     }
393     return "blank";
394   }
