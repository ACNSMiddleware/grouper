76    /**
77     * Get a Grouper configuration parameter.
78     * 
79     * <pre class="eg">
80     * String wheel = GrouperConfig.getProperty(&quot;groups.wheel.group&quot;);
81     * </pre>
82     * 
83     * @param property to lookup
84     * @param defaultValue is the value if the property isnt found
85     * @return Value of configuration parameter or the default value (will trim the value)
86     * @since 1.1.0
87     * @deprecated use GrouperWsConfig.retrieveConfig().propertyValueString instead
88     */
89    @Deprecated
90    public static String getPropertyString(String property, String defaultValue) {
91      String value = retrieveConfig().propertyValueString(property, defaultValue);
92      if (StringUtils.isBlank(value)) {
93        return defaultValue;
94      }
95      return value;
96    }

274   /**
275    * name of param
276    * # Web service users who are in the following group can use the actAs field to act as someone else
277    * # You can put multiple groups separated by commas.  e.g. a:b:c, e:f:g
278    * # You can put a single entry as the group the calling user has to be in, and the grouper the actAs has to be in
279    * # separated by 4 colons
280    * # e.g. if the configured values is:       a:b:c, e:f:d :::: r:e:w, x:e:w
281    * # then if the calling user is in a:b:c or x:e:w, then the actAs can be anyone
282    * # if not, then if the calling user is in e:f:d, then the actAs must be in r:e:w.  If multiple rules, then 
283    * # if one passes, then it is a success, if they all fail, then fail.
284    *
285    * field to act as someone else ws.act.as.group = aStem:aGroup
286    */
287   public static final String WS_ACT_AS_GROUP = "ws.act.as.group";
