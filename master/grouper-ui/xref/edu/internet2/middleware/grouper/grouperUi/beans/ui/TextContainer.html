44    /**
45     * retrieve the container from the request or create a new one if not there
46     * @return the container
47     */
48    public static TextContainer retrieveFromRequest() {
49      
50      HttpServletRequest httpServletRequest = GrouperUiFilter.retrieveHttpServletRequest();
51  
52      
53      TextContainer textContainer = 
54          (TextContainer)httpServletRequest.getAttribute("textContainer");
55      
56      if (textContainer == null) {
57        textContainer = new TextContainer();
58        httpServletRequest.setAttribute(
59            "textContainer", textContainer);
60      }
61      
62      return textContainer;
63      
64    }

89    /**
90     * massage text with substitutions etc
91     * @param key
92     * @param value
93     * @param errorIfNotFound true if error text if not found
94     * @return the text
95     */
96    public static String massageText(String key, String value, boolean errorIfNotFound) {
97      if (StringUtils.isBlank(value)) {
98        if (!errorIfNotFound) {
99          return null;
100       }
101       LOG.error("Cant find text for variable: '" + key + "'");
102       return "$$not found: " + key + "$$";
103     }
104     
105     //if there might be a scriptlet
106     if (value.contains("${")) {
107       Map<String, Object> substituteMap = new LinkedHashMap<String, Object>();
108       substituteMap.put("grouperRequestContainer", GrouperRequestContainer.retrieveFromRequestOrCreate());
109       substituteMap.put("request", GrouperUiFilter.retrieveHttpServletRequest());
110       substituteMap.put("textContainer", TextContainer.retrieveFromRequest());
111       value = GrouperUtil.substituteExpressionLanguage(value, substituteMap, true, false, true);
112     }
113     return value;
114   }

151   /**
152    * convert text to tooltip
153    * @param key
154    * @param value
155    * @return the tooltip html
156    */
157   public String convertTextToTooltip(String key, String value) {
158     String targettedTooltipKey = GrouperUiTextConfig.TOOLTIP_TARGETTED_PREFIX + key;
159     String targettedTooltipRefKey = GrouperUiTextConfig.TOOLTIP_TARGETTED_REF_PREFIX + key;
160 
161     GrouperUiTextConfig grouperUiTextConfig = GrouperUiTextConfig.retrieveTextConfig();
162 
163     String targettedTooltipValue = null;
164 
165     if (grouperUiTextConfig != null) {
166       String targettedTooltipRefValue = grouperUiTextConfig
167           .propertyValueString(targettedTooltipRefKey);
168 
169       targettedTooltipValue = grouperUiTextConfig
170           .propertyValueString(targettedTooltipKey);
171 
172       if (StringUtils.isNotBlank(targettedTooltipRefValue)
173           && StringUtils.isNotBlank(targettedTooltipValue)) {
174         LOG.warn("Duplicate tooltip target and ref in text config: "
175             + targettedTooltipKey
176             + ", " + targettedTooltipRefKey);
177       }
178 
179       if (StringUtils.isBlank(targettedTooltipValue)) {
180 
181         //first priority is a targetted ref, next priority is a target.  not sure why both would be there
182         targettedTooltipValue = StringUtils.isBlank(targettedTooltipRefValue) ?
183             targettedTooltipValue : grouperUiTextConfig
184                 .propertyValueString(targettedTooltipRefValue);
185 
186         //if there is a ref, but it doesnt exist, that is a problem
187         if (StringUtils.isNotBlank(targettedTooltipRefValue)
188             && StringUtils.isBlank(targettedTooltipValue)) {
189           LOG.error("Missing tooltip targetted ref in nav.properties: "
190               + targettedTooltipRefValue);
191         }
192 
193       }
194     }
195     boolean hasTooltip = StringUtils.isNotBlank(targettedTooltipValue);
196 
197     if (LOG.isDebugEnabled()) {
198       LOG.debug("Tooltip key: " + targettedTooltipKey + " has tooltip? " + hasTooltip);
199     }
200 
201     if (hasTooltip && !StringUtils.equals(targettedTooltipValue, value)) {
202 
203       //replace the whole message with tooltip
204       value = convertTooltipTextToHtml(targettedTooltipValue, value, false);
205 
206     } else {
207 
208       //CH 20080129 at this point we need to make the tooltip substitutions
209       value = substituteTooltips(value, false);
210 
211     }
212     return value;
213   }

215   /**
216    * substitute tooltips
217    * @param message
218    * @param isIgnoreTooltipStyle true if should ignore tooltip style
219    * @return the substituted strings
220    */
221   public String substituteTooltips(String message, boolean isIgnoreTooltipStyle) {
222     
223     //first step, get the map of substitutions
224     GrouperUiTextConfig grouperUiTextConfig = GrouperUiTextConfig.retrieveTextConfig();
225     List<String> theTooltipKeys = grouperUiTextConfig.tooltipKeys();
226     List<String> theTooltipValues = grouperUiTextConfig.tooltipValues();
227     
228     //substitute, and remove if replaced
229     String result = GrouperUtil.replace(message, theTooltipKeys, theTooltipValues, false, true);
230     
231     //maybe take out styles
232     if (isIgnoreTooltipStyle) {
233       result = StringUtils.replace(result, "class=\"grouperTooltip\" ", "");
234     }
235     
236     return result;
237   }

239   /**
240    * convert tooltip text to html
241    * @param tooltipText 
242    * @param term
243    * @param isIgnoreTooltipStyle if tooltip style should be ignored
244    * @return the html tooltip text
245    */
246   public static String convertTooltipTextToHtml(String tooltipText, String term, 
247       boolean isIgnoreTooltipStyle) {
248     //substitute single quotes for javascript
249     //tooltipText = tooltipText.replace("'", "\\'");
250     //put it in the tooltip code
251     //the class="tooltip" is substituted later, so if this is changed, change in other places as well
252     String escapedTooltipText = StringUtils.replace(tooltipText, "'", "&#39;");
253     escapedTooltipText = GrouperUiUtils.escapeHtml(escapedTooltipText, true, true);
254     tooltipText = "<span " + (isIgnoreTooltipStyle ? "" : "class=\"grouperTooltip\" ") 
255       + "onmouseover=\"grouperTooltip('" 
256       + escapedTooltipText + "');\" onmouseout=\"UnTip()\">" + term + "</span>";
257     return tooltipText;
258   }
