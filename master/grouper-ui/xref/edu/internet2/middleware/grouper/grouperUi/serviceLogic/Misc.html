43    /**
44     * index page of application
45     * @param request
46     * @param response
47     */
48    public void index(HttpServletRequest request, HttpServletResponse response) {
49      GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
50      
51      guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#topDiv", 
52          "/WEB-INF/grouperUi/templates/common/commonTop.jsp"));
53  
54      guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#bodyDiv", 
55          "/WEB-INF/grouperUi/templates/misc/index.jsp"));
56  
57    }
58    
59    /**
60     * 
61     * @param request
62     * @param response
63     */
64    public void logout(HttpServletRequest request, HttpServletResponse response) {
65  
66      if (!LogoutAction.logoutLogic(request, response, request.getSession(false), true)) {
67        //not sure why this would happen, we are in ajax
68        //throw new ControllerDone();
69      }
70  
71      GuiResponseJs guiResponseJs = GuiResponseJs.retrieveGuiResponseJs();
72      guiResponseJs.addAction(GuiScreenAction.newInnerHtmlFromJsp("#bodyDiv", 
73          "/WEB-INF/grouperUi/templates/misc/logout.jsp"));
74      
75      guiResponseJs.addAction(GuiScreenAction.newInnerHtml("#topDiv", ""));
76    
77    }
