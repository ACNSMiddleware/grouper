118   /**

123   static Set<AttributeDefName> filterSecurity(Set<AttributeDefName> attributeDefNames) {
124     Set<AttributeDefName> result = new LinkedHashSet<AttributeDefName>();
125     if (attributeDefNames != null) {
126       for (AttributeDefName attributeDefName : attributeDefNames) {
127         attributeDefName = filterSecurity(attributeDefName);
128         if (attributeDefName != null) {
129           result.add(attributeDefName);
130         }
131       }
132     }
133     return result;
134   }

293    * @see edu.internet2.middleware.grouper.internal.dao.AttributeDefNameDAO#findAllSecure(java.lang.String, java.util.Set, QueryOptions)
294    */
295   public Set<AttributeDefName> findAllSecure(String searchField,
296       Set<String> searchInAttributeDefIds, QueryOptions queryOptions) {
297 
298     {
299       String searchFieldNoPercents = StringUtils.replace(StringUtils.defaultString(searchField), "%", "");
300       
301       if (StringUtils.isBlank(searchFieldNoPercents) || searchFieldNoPercents.length() < 2) {
302         throw new RuntimeException("Need to pass in a searchField of at least 2 chars");
303       }
304     }
305     
306     String searchFieldLower = StringUtils.defaultString(searchField).toLowerCase();
307 
308     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
309     
310     Subject grouperSessionSubject = grouperSession.getSubject();
311     
312     StringBuilder sqlTables = new StringBuilder("from AttributeDefName as attributeDefName ");
313     ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
314     
315     StringBuilder sqlWhereClause =  new StringBuilder(" (lower(attributeDefName.extensionDb) like :searchField " +
316       "or lower(attributeDefName.displayExtensionDb) like :searchField " +
317       "or lower(attributeDefName.description) like :searchField) ");
318 
319     grouperSession.getAttributeDefResolver().hqlFilterAttrDefsWhereClause(
320         grouperSessionSubject, byHqlStatic, 
321         sqlTables, sqlWhereClause, "attributeDefName.attributeDefId", AttributeDefPrivilege.ATTR_READ_PRIVILEGES);
322 
323     StringBuilder sql;
324     sql = sqlTables.append(" where ").append(sqlWhereClause);
325     
326     if (GrouperUtil.length(searchInAttributeDefIds) > 0) {
327       sql.append(" and attributeDefName.attributeDefId in (");
328       sql.append(HibUtils.convertToInClause(searchInAttributeDefIds, byHqlStatic));
329       sql.append(") ");
330     }
331     
332     Set<AttributeDefName> attributeDefNames = byHqlStatic
333       .createQuery(sql.toString()).options(queryOptions)
334       .setCacheable(true)
335       .setCacheRegion(KLASS + ".FindAll")
336       .setString("searchField", searchFieldLower)
337       .listSet(AttributeDefName.class);
338 
339     int maxSize = GrouperConfig.retrieveConfig().propertyValueInt("findAllAttributeDefNames.maxResultSize", 30000);
340     if (maxSize > -1) {
341       if (maxSize < attributeDefNames.size()) {
342         throw new AttributeDefNameTooManyResults("Too many results: " 
343             + attributeDefNames.size() + ", '" + searchField + "'");
344       }
345     }
346     
347     return attributeDefNames;
348 
349   }

374   /**
375    * if there are sort fields, go through them, and replace name with nameDb, etc,
376    * extension for extensionDb, displayName with displayNameDb, and displayExtension with displayExtensionDb
377    * @param querySort
378    */
379   private static void massageSortFields(QuerySort querySort) {
380     if (querySort == null) {
381       return;
382     }
383     for (QuerySortField querySortField : GrouperUtil.nonNull(querySort.getQuerySortFields())) {
384       if (StringUtils.equals("extension", querySortField.getColumn())) {
385         querySortField.setColumn("theAttributeDefName.extensionDb");
386       }
387       if (StringUtils.equals("name", querySortField.getColumn())) {
388         querySortField.setColumn("theAttributeDefName.nameDb");
389       }
390       if (StringUtils.equals("displayExtension", querySortField.getColumn())
391           || StringUtils.equals("display_extension", querySortField.getColumn())) {
392         querySortField.setColumn("theAttributeDefName.displayExtensionDb");
393       }
394       if (StringUtils.equals("displayName", querySortField.getColumn())
395           || StringUtils.equals("display_name", querySortField.getColumn())) {
396         querySortField.setColumn("theAttributeDefName.displayNameDb");
397       }
398       if (StringUtils.equals("description", querySortField.getColumn())) {
399         querySortField.setColumn("theAttributeDefName.description");
400       }
401     }
402 
403   }

405   /**
406    * 
407    * @param scope 
408    * @param grouperSession 
409    * @param attributeDefId 
410    * @param subject 
411    * @param privileges if privileges is null and filterByServicesCanView is true then set the privileges to the view set 
412    * @param queryOptions 
413    * @param splitScope 
414    * @param attributeAssignType
415    * @param attributeDefType 
416    * @param filterByServicesCanView if true then only return services the user can view based on memberships which have the 
417    * service, or folders which have the service, or attribute definitions which have the service,
418    * or the view privilege set
419    * @param serviceRole if filtering by services the user has in a certain role
420    * @param anyServiceRole true if services should be returned where the user has any role
421    * @param parentStemId stem id of the parent of ancestor of this object
422    * @param stemScope is SUB or ONE
423    * @param findByUuidOrName 
424    * @return  attribute def names
425    * 
426    */
427   private Set<AttributeDefName> findAllAttributeNamesSecureHelper(String scope,
428       GrouperSession grouperSession, String attributeDefId, Subject subject, Set<Privilege> privileges,
429       QueryOptions queryOptions, boolean splitScope, AttributeAssignType attributeAssignType,
430       AttributeDefType attributeDefType, ServiceRole serviceRole, boolean anyServiceRole, String parentStemId,
431       Scope stemScope, boolean findByUuidOrName) {
432     
433     if (queryOptions == null) {
434       queryOptions = new QueryOptions();
435     }
436     if (queryOptions.getQuerySort() == null) {
437       queryOptions.sortAsc("theAttributeDefName.displayNameDb");
438     }
439   
440     Member member = subject == null ? null : MemberFinder.findBySubject(grouperSession, subject, true);
441 
442     StringBuilder sql = new StringBuilder(
443         "select distinct theAttributeDefName from AttributeDefName theAttributeDefName, AttributeDef theAttributeDef ");
444 
445     if (serviceRole != null || anyServiceRole) {
446       
447       if (attributeDefType != null && attributeDefType != AttributeDefType.service) {
448         throw new RuntimeException("You cant look for services and not have AttributeDefType of service: " + attributeDefType);
449       }
450       //CH not sure this is needed
451       //attributeDefType = AttributeDefType.service;
452       sql.append(", ServiceRoleView theServiceRoleView ");
453     }
454 
455     if (!StringUtils.isBlank(parentStemId) || stemScope != null) {
456       
457       if (StringUtils.isBlank(parentStemId) || stemScope == null) {
458         throw new RuntimeException("If you are passing in a parentStemId or a stemScope, then you need to pass both of them: " + parentStemId + ", " + stemScope);
459       }
460       
461       if (stemScope == Scope.SUB) {
462         sql.append(", StemSet theStemSet ");
463       }
464     }      
465 
466     
467     ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
468   
469     StringBuilder whereClause = new StringBuilder(" theAttributeDefName.attributeDefId = theAttributeDef.id ");
470 
471     if (serviceRole != null || anyServiceRole) {
472       if (attributeDefType == null) {
473         attributeDefType = AttributeDefType.service;
474       }
475       if (attributeDefType != AttributeDefType.service) {
476         throw new RuntimeException("Why are you filtering by serviceRole: " + serviceRole + " but you have the " +
477         		"attributeDefType not equal to AttributeDefType.service????  (or can be null): " + attributeDefType);
478       }
479       if (attributeAssignType == null) {
480         attributeAssignType = AttributeAssignType.stem;
481       }
482       if (attributeAssignType != AttributeAssignType.stem) {
483         throw new RuntimeException("Why are you filtering by serviceRole: " + serviceRole + " but you have the " +
484             "attributeAssignType not equal to AttributeAssignType.stem????  (or can be null): " + attributeAssignType);
485       }
486       
487     }
488     
489     if (attributeDefType != null) {
490       whereClause.append(" and ");
491       whereClause.append(" theAttributeDef.attributeDefTypeDb = :theAttributeDefType ");
492       byHqlStatic.setString("theAttributeDefType", attributeDefType.name());
493     }
494 
495     if (!StringUtils.isBlank(parentStemId) || stemScope != null) {
496       switch(stemScope) {
497         case ONE:
498           
499           whereClause.append(" and theAttributeDefName.stemId = :theStemId ");
500           byHqlStatic.setString("theStemId", parentStemId);
501           break;
502         case SUB:
503           
504           whereClause.append(" and theAttributeDefName.stemId = theStemSet.ifHasStemId " +
505           		" and theStemSet.thenHasStemId = :theStemId ");
506           byHqlStatic.setString("theStemId", parentStemId);
507           
508           break;
509         
510       }
511     }
512 
513     if (!StringUtils.isBlank(attributeDefId)) {
514       
515       whereClause.append(" and ");
516       whereClause.append(" theAttributeDefName.attributeDefId = :theAttributeDefId ");
517       byHqlStatic.setString("theAttributeDefId", attributeDefId);
518       
519     }
520 
521     if (attributeAssignType != null) {
522       whereClause.append(" and ");
523       switch (attributeAssignType) {
524         case any_mem:
525           whereClause.append(" theAttributeDef.assignToEffMembershipDb = 'T' ");
526           break;
527         case any_mem_asgn:
528           whereClause.append(" theAttributeDef.assignToEffMembershipAssnDb = 'T' ");
529           break;
530         case attr_def:
531           whereClause.append(" theAttributeDef.assignToAttributeDefDb = 'T' ");
532           break;
533         case attr_def_asgn:
534           whereClause.append(" theAttributeDef.assignToAttributeDefAssnDb = 'T' ");
535           break;
536         case group:
537           whereClause.append(" theAttributeDef.assignToGroupDb = 'T' ");
538           break;
539         case group_asgn:
540           whereClause.append(" theAttributeDef.assignToGroupAssnDb = 'T' ");
541           break;
542         case imm_mem:
543           whereClause.append(" theAttributeDef.assignToImmMembershipDb = 'T' ");
544           break;
545         case imm_mem_asgn:
546           whereClause.append(" theAttributeDef.assignToImmMembershipAssnDb = 'T' ");
547           break;
548         case member:
549           whereClause.append(" theAttributeDef.assignToMemberDb = 'T' ");
550           break;
551         case mem_asgn:
552           whereClause.append(" theAttributeDef.assignToMemberAssnDb = 'T' ");
553           break;
554         case stem:
555           whereClause.append(" theAttributeDef.assignToStemDb = 'T' ");
556           break;
557         case stem_asgn:
558           whereClause.append(" theAttributeDef.assignToStemAssnDb = 'T' ");
559           break;
560         default:
561           throw new RuntimeException("Not expecting attribute assign type: " + attributeAssignType);
562       }
563     }
564     
565 
566     //see if there is a scope
567     if (!StringUtils.isBlank(scope)) {
568       scope = scope.toLowerCase();
569 
570       String[] scopes = splitScope ? GrouperUtil.splitTrim(scope, " ") : new String[]{scope};
571 
572       whereClause.append(" and ");
573 
574       if (GrouperUtil.length(scopes) == 1) {
575         whereClause.append(" ( theAttributeDefName.id = :theAttributeDefNameIdScope or ( ");
576         byHqlStatic.setString("theAttributeDefNameIdScope", scope);
577       } else {
578         whereClause.append(" ( ( ");
579       }
580 
581       int index = 0;
582       for (String theScope : scopes) {
583         if (index != 0) {
584           whereClause.append(" and ");
585         }
586 
587         if (findByUuidOrName) {
588           whereClause.append(" ( theAttributeDefName.nameDb = :scope" + index + " or theAttributeDefName.alternateNameDb = :scope" + index 
589               + " or theAttributeDefName.displayNameDb = :scope" + index + " ) ");
590           byHqlStatic.setString("scope" + index, theScope);
591         } else {
592           whereClause.append(" ( lower(theAttributeDefName.nameDb) like :scope" + index 
593               + " or lower(theAttributeDefName.displayNameDb) like :scope" + index 
594               + " or lower(theAttributeDefName.description) like :scope" + index + " ) ");
595           if (splitScope) {
596             theScope = "%" + theScope + "%";
597           } else if (!theScope.endsWith("%")) {
598             theScope += "%";
599           }
600           byHqlStatic.setString("scope" + index, theScope.toLowerCase());
601 
602         }        
603 
604         index++;
605       }
606       whereClause.append(" ) ) ");
607     }
608   
609     boolean changedQuery = false;
610     
611     if (serviceRole != null || anyServiceRole) {
612       
613       if (serviceRole != null && anyServiceRole) {
614         throw new RuntimeException("Cant set serviceRole and anyServiceRole, they are mutually exclusive");
615       }
616       
617       if (member == null) {
618         throw new RuntimeException("If filtering by serviceRole: " + serviceRole + ", then pass in a subject who can see the services");
619       }
620       
621       //must be a service type
622       whereClause.append(" and theAttributeDefName.id = theServiceRoleView.serviceNameId ");
623 
624       changedQuery = grouperSession.getAccessResolver().hqlFilterGroupsWhereClause(
625           grouperSession.getSubject(), byHqlStatic, 
626           sql, "theServiceRoleView.groupId", AccessPrivilege.READ_PRIVILEGES);
627       
628       //fields for the service role
629       HibUtils.convertFieldsToSqlInString(
630           anyServiceRole ? ServiceRole.allFieldsForGroupQuery() : serviceRole.fieldsForGroupQuery(), 
631           byHqlStatic, whereClause, "theServiceRoleView.fieldId");
632 
633       whereClause.append(" and theServiceRoleView.memberId = :groupMemberId ");
634       byHqlStatic.setString("groupMemberId", member.getUuid());
635 
636       //no need to do security for services, right????
637       
638     } else {
639 
640       //see if we are adding more to the query
641       if (GrouperUtil.length(privileges) > 0) {
642         
643         if (subject == null) {
644           subject = grouperSession.getSubject();
645         }
646         
647         grouperSession.getAttributeDefResolver().hqlFilterAttrDefsWhereClause(subject, byHqlStatic,
648             sql, whereClause, "theAttributeDefName.attributeDefId", privileges);
649       }
650     }
651     
652     if (changedQuery) {
653       sql.append(" and ");      
654     } else {
655       sql.append(" where ");
656     }
657     
658     sql.append(whereClause);
659     
660     if (queryOptions != null) {
661       massageSortFields(queryOptions.getQuerySort());
662     }
663 
664     Set<AttributeDefName> attributeDefNames = byHqlStatic.createQuery(sql.toString())
665       .setCacheable(false)
666       .setCacheRegion(KLASS + ".GetAllAttributeDefNamesSecure")
667       .options(queryOptions)
668       .listSet(AttributeDefName.class);
669     
670     //if find by uuid or name, try to narrow down to one...
671     if (findByUuidOrName) {
672       
673       //get the one with uuid
674       for (AttributeDefName attributeDefName : attributeDefNames) {
675         if (StringUtils.equals(scope, attributeDefName.getId())) {
676           return GrouperUtil.toSet(attributeDefName);
677         }
678       }
679       
680       //get the one with name
681       for (AttributeDefName attributeDefName : attributeDefNames) {
682         if (StringUtils.equals(scope, attributeDefName.getName())) {
683           return GrouperUtil.toSet(attributeDefName);
684         }
685       }
686     }
687     return attributeDefNames;
688   
689   }

812   /**
813    * @see AttributeDefNameDAO#findByIdsSecure(Collection, QueryOptions)
814    */
815   @Override
816   public Set<AttributeDefName> findByIdsSecure(Collection<String> ids,
817       QueryOptions queryOptions) {
818     
819     Set<AttributeDefName> attributeDefNames = new HashSet<AttributeDefName>();
820     
821     if (GrouperUtil.length(ids) == 0) {
822       return attributeDefNames;
823     }
824     
825     GrouperSession grouperSession = GrouperSession.staticGrouperSession();
826     
827     int numberOfBatches = GrouperUtil.batchNumberOfBatches(ids, 180);
828     
829     List<String> idsList = GrouperUtil.listFromCollection(ids);
830     
831     for (int i=0;i<numberOfBatches;i++) {
832       
833       List<String> uuidsBatch = GrouperUtil.batchList(idsList, 180, i);
834       
835       ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
836 
837       StringBuilder sql = new StringBuilder("select distinct theAttributeDefName from AttributeDef as theAttributeDef, " +
838       		"AttributeDefName as theAttributeDefName ");
839       
840       StringBuilder whereClause = new StringBuilder();
841       
842       //see if we are adding more to the query
843       grouperSession.getAttributeDefResolver().hqlFilterAttrDefsWhereClause(
844           grouperSession.getSubject(), byHqlStatic, 
845           sql, whereClause, "theAttributeDef.id", AttributeDefPrivilege.ATTR_VIEW_PRIVILEGES);
846 
847       sql.append(" where ").append(whereClause);
848       
849       if (whereClause.length() > 0) {
850         sql.append(" and ");
851       }
852       
853       sql.append(" theAttributeDef.id = theAttributeDefName.attributeDefId and ");
854       
855       sql.append(" theAttributeDefName.id in ( ");
856       
857       sql.append(HibUtils.convertToInClause(uuidsBatch, byHqlStatic)).append(" ) ");
858       
859       byHqlStatic
860         .createQuery(sql.toString())
861         .setCacheable(true)
862         .options(queryOptions)
863         .setCacheRegion(KLASS + ".FindByUuidsSecure");
864       
865       Set<AttributeDefName> attributeDefNamesBatch = byHqlStatic.listSet(AttributeDefName.class);
866       
867       attributeDefNames.addAll(GrouperUtil.nonNull(attributeDefNamesBatch));
868       
869     }
870     
871     return attributeDefNames;
872   }
