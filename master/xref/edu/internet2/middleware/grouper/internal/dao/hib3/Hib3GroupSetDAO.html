94              //http://bugs.mysql.com/bug.php?id=15746
95              hibernateHandlerBean.getHibernateSession().byHql().createQuery(
96                  "update GroupSet set parentId = null where id = :id")
97                  .setString("id", groupSet.getId()).executeUpdate();
98              */
99              
100             
101             hibernateHandlerBean.getHibernateSession().byObject().delete(groupSet);
102             return null;
103 
104           }
105       
106     });

195   /**
196    * reset group set
197    * @param hibernateSession
198    */
199   public static void reset(HibernateSession hibernateSession) {
200     Session hs = hibernateSession.getSession();
201 
202     // Find the root stem
203     Stem rootStem = GrouperDAOFactory.getFactory().getStem().findByName(Stem.ROOT_INT, true);
204     
205     // get the max depth
206     int maxDepth = HibernateSession.byHqlStatic().createQuery("select max(depth) from GroupSet").uniqueResult(Integer.class);
207     
208     // delete the rows that don't have self reference by the depth desc
209     for (int i = maxDepth; i > 0; i--) {
210       hs.createQuery("delete from GroupSet where depth = :depth").setInteger("depth", i).executeUpdate();
211     }
212     
213     // set parent_id to null due to mysql bug -- http://bugs.mysql.com/bug.php?id=15746
214     // also note that parent_id is part of a unique constraint, but now that we have only self groupSets, that doesn't matter.
215     hs.createQuery("update GroupSet set parentId = null where ownerStemId not like :owner or ownerStemId is null")
216         .setString("owner", rootStem.getUuid())
217         .executeUpdate();
218     
219     hs.createQuery("delete from GroupSet where parentId = null")
220         .executeUpdate();
221   }

306    */
307   public Set<GroupSet> findAllChildren(GroupSet groupSet) {
308     Set<GroupSet> allChildren = new LinkedHashSet<GroupSet>();
309     Set<GroupSet> children = HibernateSession
310         .byHqlStatic()
311         .createQuery("select gs from GroupSet as gs where gs.parentId = :parent and gs.type = 'effective'")
312         .setCacheable(false).setCacheRegion(KLASS + ".FindAllChildren")
313         .setString("parent", groupSet.getId())
314         .listSet(GroupSet.class);
315     
316     Iterator<GroupSet> iter = children.iterator();
317     
318     while (iter.hasNext()) {
319       GroupSet child = iter.next();
320       allChildren.addAll(findAllChildren(child));
321       allChildren.add(child);
322     }
323     
324     return allChildren;
325   }

514     		         "groupTypeAssign.ownerGroupId = g.id and " +

516                  "and not exists " +
517                  "(select gs.ownerGroupId from GroupSet as gs where gs.ownerGroupId = g.id and gs.fieldId = f.uuid and gs.depth='0')";
