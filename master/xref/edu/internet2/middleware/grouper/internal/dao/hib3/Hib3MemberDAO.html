112    * @since   @HEAD@
113    */
114   public boolean exists(String uuid) 
115     throws  GrouperDAOException
116   {
117     if ( getExistsCache().containsKey(uuid) ) {
118       return getExistsCache().get(uuid).booleanValue();
119     }
120     Object id = null;
121     try {
122       id = HibernateSession.byHqlStatic()
123         .createQuery("select m.id from Member as m where m.uuid = :uuid")
124         .setCacheable(false)
125         .setCacheRegion(KLASS + ".Exists")
126         .setString("uuid", uuid).uniqueResult(Object.class);
127     } catch (GrouperDAOException gde) {
128       Throwable throwable = gde.getCause();
129       //CH 20080218 this was legacy error handling
130       if (throwable instanceof HibernateException) {
131         LOG.fatal( throwable.getMessage() );
132       }
133       throw gde;
134     }
135     boolean rv  = false;
136     if ( id != null ) {
137       rv = true; 
138     }
139     getExistsCache().put(uuid, rv);
140     return rv;
141   }

205   /**
206    * if there are sort fields, go through them, and replace name with m.subjectIdDb, etc,
207    * sort_string0 for m.sortString0, etc
208    * @param querySort
209    */
210   public static void massageMemberSortFields(QuerySort querySort) {
211     if (querySort == null) {
212       return;
213     }
214 
215     for (QuerySortField querySortField : GrouperUtil.nonNull(querySort.getQuerySortFields())) {
216       if (StringUtils.equalsIgnoreCase("uuid", querySortField.getColumn())
217           || StringUtils.equalsIgnoreCase("id", querySortField.getColumn())) {
218         querySortField.setColumn("m.uuid");
219       }
220       if (StringUtils.equalsIgnoreCase("subjectIdDb", querySortField.getColumn())
221           || StringUtils.equalsIgnoreCase("subject_id", querySortField.getColumn())
222           || StringUtils.equalsIgnoreCase("subjectId", querySortField.getColumn())) {
223         querySortField.setColumn("m.subjectIdDb");
224       }
225       if (StringUtils.equalsIgnoreCase("subjectSourceIdDb", querySortField.getColumn())
226           || StringUtils.equalsIgnoreCase("subject_source", querySortField.getColumn())
227           || StringUtils.equalsIgnoreCase("sourceId", querySortField.getColumn())) {
228         querySortField.setColumn("m.subjectSourceIdDb");
229       }
230       if (StringUtils.equalsIgnoreCase("subjectTypeId", querySortField.getColumn())
231           || StringUtils.equalsIgnoreCase("subject_type", querySortField.getColumn())) {
232         querySortField.setColumn("m.subjectTypeId");
233       }
234       if (StringUtils.equalsIgnoreCase("sortString0", querySortField.getColumn())
235           || StringUtils.equalsIgnoreCase("sort_string0", querySortField.getColumn())) {
236         if (!SortStringEnum.newInstance(0).hasAccess()) {
237           throw new RuntimeException("Not allowed to access " + querySortField.getColumn());
238         }
239         querySortField.setColumn("m.sortString0");
240       }
241       if (StringUtils.equalsIgnoreCase("sortString1", querySortField.getColumn())
242           || StringUtils.equalsIgnoreCase("sort_string1", querySortField.getColumn())) {
243         if (!SortStringEnum.newInstance(1).hasAccess()) {
244           throw new RuntimeException("Not allowed to access " + querySortField.getColumn());
245         }
246         querySortField.setColumn("m.sortString1");
247       }
248       if (StringUtils.equalsIgnoreCase("sortString2", querySortField.getColumn())
249           || StringUtils.equalsIgnoreCase("sort_string2", querySortField.getColumn())) {
250         if (!SortStringEnum.newInstance(2).hasAccess()) {
251           throw new RuntimeException("Not allowed to access " + querySortField.getColumn());
252         }
253         querySortField.setColumn("m.sortString2");
254       }
255       if (StringUtils.equalsIgnoreCase("sortString3", querySortField.getColumn())
256           || StringUtils.equalsIgnoreCase("sort_string3", querySortField.getColumn())) {
257         if (!SortStringEnum.newInstance(3).hasAccess()) {
258           throw new RuntimeException("Not allowed to access " + querySortField.getColumn());
259         }
260         querySortField.setColumn("m.sortString3");
261       }
262       if (StringUtils.equalsIgnoreCase("sortString4", querySortField.getColumn())
263           || StringUtils.equalsIgnoreCase("sort_string4", querySortField.getColumn())) {
264         if (!SortStringEnum.newInstance(4).hasAccess()) {
265           throw new RuntimeException("Not allowed to access " + querySortField.getColumn());
266         }
267         querySortField.setColumn("m.sortString4");
268       }
269       if (StringUtils.equalsIgnoreCase("name", querySortField.getColumn())) {
270         querySortField.setColumn("m.name");
271       }
272       if (StringUtils.equalsIgnoreCase("description", querySortField.getColumn())) {
273         querySortField.setColumn("m.description");
274       }
275 
276     }

281   /**
282    * 
283    * @param id
284    * @param src
285    * @param type
286    * @param exceptionIfNull
287    * @param queryOptions
288    * @return member
289    * @throws GrouperDAOException
290    * @throws MemberNotFoundException
291    */
292   public Member findBySubject(String id, String src, String type, boolean exceptionIfNull, QueryOptions queryOptions)
293       throws GrouperDAOException, MemberNotFoundException {
294     Member member = null;
295     
296     member = HibernateSession.byHqlStatic()
297       .createQuery("from Member as m where "
298         + "     m.subjectIdDb       = :sid    "  
299         + "and  m.subjectSourceIdDb = :source "
300         + "and  m.subjectTypeId   = :type")
301         .setCacheable(true)
302         .setCacheRegion(KLASS + ".FindBySubject")
303         .options(queryOptions)
304         .setString( "sid",    id   )
305         .setString( "type",   type )
306         .setString( "source", src  )
307         .uniqueResult(Member.class);
308       
309     if (exceptionIfNull && member == null) {
310       throw new MemberNotFoundException();
311     }
312     
313     //dont cache this
314     if (!exceptionIfNull && member == null) {
315       return null;
316     }
317     
318     getUuid2dtoCache().put( member.getUuid(), member );
319     return member;
320   }

343       }

348         .setCacheable(false) // but i probably should - or at least permit it

351       } catch (GrouperDAOException gde) {

353         //CH 20080218 this was legacy error handling
354         if (throwable instanceof HibernateException) {
355           LOG.fatal( throwable.getMessage() );
356         }
357         throw gde;
358       }
359     }
360     if (memberDto == null && exceptionIfNull) {
361       throw new MemberNotFoundException();
362     }
363     if (memberDto != null && uuid != null) {
364       getUuid2dtoCache().put(uuid, memberDto);
365     }
366     return memberDto;

423    *

425   public Member findBySubject(String subjectId, boolean exceptionIfNull)
426       throws GrouperDAOException, MemberNotFoundException, MemberNotUniqueException {
427     List<Member> members = HibernateSession.byHqlStatic().createQuery(
428         "from Member as m where m.subjectIdDb       = :sid")
429         .setCacheable(true)
430         .setCacheRegion(KLASS + ".FindBySubjectId").setString("sid", subjectId)
431         .list(Member.class);
432     if (GrouperUtil.length(members) ==0) {
433       if (exceptionIfNull) {
434         throw new MemberNotFoundException("Cant find member with subjectId: '" + subjectId + "'");
435       }
436       return null;
437     }
438     if (GrouperUtil.length(members) > 1) {
439       throw new MemberNotUniqueException("Subject id '" + subjectId + "' is not unique in the members table");
440     }
441     Member member = members.get(0);
442     getUuid2dtoCache().put(member.getUuid(), member);
443     return member;
444   }

571   /**
572    * find members by subjects and create if not exist possibly
573    * @param subjects
574    * @param createIfNotExists
575    * @return the members
576    */
577   public Set<Member> findBySubjects(
578       Collection<Subject> subjects, boolean createIfNotExists) {
579     
580     Set<Member> result = new TreeSet<Member>();
581     
582     if (GrouperUtil.length(subjects) == 0) {
583       return result;
584     }
585     
586     //lets do this in batches
587     int numberOfBatches = GrouperUtil.batchNumberOfBatches(subjects, MEMBER_SUBJECT_BATCH_SIZE);
588 
589     List<Subject> subjectsList = subjects instanceof List ? (List)subjects : new ArrayList<Subject>(subjects);
590     
591     for (int i=0;i<numberOfBatches;i++) {
592 
593       List<Subject> subjectBatch = GrouperUtil.batchList(subjectsList, MEMBER_SUBJECT_BATCH_SIZE, i);
594 
595 //      select distinct gm.* 
596 //      from grouper_members gm
597 //      where (gm.subject_id = '123' and gm.subject_source = 'jdbc') 
598 //          or (gm.subject_id = '234' and gm.subject_source = 'jdbc' )
599 
600       //lets turn the subjects into subjectIds
601       if (GrouperUtil.length(subjectBatch) == 0) {
602         continue;
603       }
604 
605       ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
606       StringBuilder query = new StringBuilder("select distinct gm " +
607           "from Member gm " +
608           "where ");
609       
610       //add all the uuids
611       query.append(HibUtils.convertToSubjectInClause(subjectBatch, byHqlStatic, "gm"));
612       List<Member> currentMemberList = byHqlStatic.createQuery(query.toString())
613         .list(Member.class);
614       result.addAll(currentMemberList);
615       
616       if (createIfNotExists) {
617         
618         //see which ones we've got
619         Set<MultiKey> subjectsRetrieved = new HashSet<MultiKey>();
620         
621         for (Member member : currentMemberList) {
622           MultiKey multiKey = new MultiKey(member.getSubjectSourceId(), member.getSubjectId());
623           subjectsRetrieved.add(multiKey);
624         }
625         
626         //loop through what we were supposed to get
627         for (Subject subject : subjectBatch) {

633           }
634         }

638     }

647    * convert a set of subjects to a set of members
648    * @param grouperSession 
649    * @param subjects to convert to members
650    * @param group that subjects must be in
651    * @param field that they must be in in the group (null will default to eh members list
652    * @param membershipType that they must be in in the group or null for any
653    * @return the members in the group
654    * @see MemberDAO#findBySubjectsInGroup(GrouperSession, Set, Group, Field, MembershipType)
655    */
656   public Set<Member> findBySubjectsInGroup(GrouperSession grouperSession,
657       Set<Subject> subjects, Group group, Field field, MembershipType membershipType) {
658     if (field == null) {
659       field = Group.getDefaultList();
660     }
661 
662     Set<Member> result = new TreeSet<Member>();
663     
664     if (GrouperUtil.length(subjects) == 0) {
665       return result;
666     }
667     
668     //lets do this in batches
669     List<Subject> subjectsList = GrouperUtil.listFromCollection(subjects);
670     int numberOfBatches = GrouperUtil.batchNumberOfBatches(subjectsList, MEMBER_SUBJECT_BATCH_SIZE);
671     
672     if (!StringUtils.equals("list", field.getTypeString())) {
673       throw new RuntimeException("Can only call this method with a list field: " + field);
674     }
675     
676     //check security once
677     if (!PrivilegeHelper.canViewMembers(grouperSession, group, field)) {
678       throw new InsufficientPrivilegeException("subject " + grouperSession.getSubject() + " cannot read group: " + group);
679     }
680 
681     for (int i=0;i<numberOfBatches;i++) {
682 
683       List<Subject> subjectBatch = GrouperUtil.batchList(subjectsList, MEMBER_SUBJECT_BATCH_SIZE, i);
684 
685 //      select distinct gm.* 
686 //      from grouper_members gm, grouper_memberships gms
687 //      where gm.id = gms.member_id
688 //      and gms.field_id = 'abc' and gms.owner_id = '123'
689 //      and ((gm.subject_id = '123' and gm.subject_source = 'jdbc') 
690 //          or (gm.subject_id = '234' and gm.subject_source = 'jdbc' ))
691 //      and mship_type = 'immediate'
692 
693       //lets turn the subjects into subjectIds
694       if (GrouperUtil.length(subjectBatch) == 0) {
695         continue;
696       }
697 
698       ByHqlStatic byHqlStatic = HibernateSession.byHqlStatic();
699       StringBuilder query = new StringBuilder("select distinct gm " +
700           "from Member gm, MembershipEntry gms " +
701           "where gm.uuid = gms.memberUuid " +
702           "and gms.enabledDb = 'T' " +
703           "and gms.fieldId = :fieldId " +
704           "and gms.ownerGroupId = :ownerId " +
705           (membershipType == null ? "" : (" and gms.type " + membershipType.queryClause() + " ")) +
706           "and ");
707       
708       //add all the uuids
709       byHqlStatic.setString("fieldId", field.getUuid());
710       byHqlStatic.setString("ownerId", group.getUuid());
711       query.append(HibUtils.convertToSubjectInClause(subjectBatch, byHqlStatic, "gm"));
712       List<Member> currentMemberList = byHqlStatic.createQuery(query.toString())
713         .list(Member.class);
714       result.addAll(currentMemberList);
715     }
