43     * find an attributeDef by id.  This is a secure method, a GrouperSession must be open
44     * @param id of attributeDef
45     * @param exceptionIfNull true if exception should be thrown if null
46     * @return the attribute def or null
47     * @throws AttributeDefNotFoundException
48     */
49    public static AttributeDef findById(String id, boolean exceptionIfNull) {
50      
51      
52      AttributeDef attributeDef = GrouperDAOFactory.getFactory().getAttributeDef().findByIdSecure(id, exceptionIfNull);
53      
54      return attributeDef;
55      
56    }

114   /**
115    * Find an attributeDef within the registry by ID index.
116    * @param idIndex id index of attributeDef to find.
117    * @param exceptionIfNotFound true if exception if not found
118    * @param queryOptions 
119    * @return  A {@link AttributeDef}
120    * @throws AttributeDefNotFoundException if not found an exceptionIfNotFound is true
121    */
122   public static AttributeDef findByIdIndexSecure(Long idIndex, boolean exceptionIfNotFound,  QueryOptions queryOptions) 
123       throws AttributeDefNotFoundException {
124     //note, no need for GrouperSession inverse of control
125     GrouperSession.validate(GrouperSession.staticGrouperSession());
126     AttributeDef a = GrouperDAOFactory.getFactory().getAttributeDef().findByIdIndexSecure(idIndex, exceptionIfNotFound, queryOptions);
127     return a;
128   }

172   /**
173    * add a privilege to filter by that the subject has on the attribute definition
174    * @param privilege should be AttributeDefPrivilege
175    * @return this for chaining
176    */
177   public AttributeDefFinder addPrivilege(Privilege privilege) {
178     
179     if (this.privileges == null) {
180       this.privileges = new HashSet<Privilege>();
181     }
182     
183     this.privileges.add(privilege);
184     
185     return this;
186   }

258   /**
259    * find all the attribute defs
260    * @return the set of attribute defs or the empty set if none found
261    */
262   public Set<AttributeDef> findAttributes() {
263     return GrouperDAOFactory.getFactory().getAttributeDef()
264       .findAllAttributeDefsSecure(this.scope, this.splitScope, 
265           this.subject, this.privileges, 
266           this.queryOptions, this.parentStemId, this.stemScope, this.findByUuidOrName, this.attributeDefIds);
267   }

272    * @return the attribute def finder
273    */
274   public AttributeDefFinder assignFindByUuidOrName(boolean theFindByUuidOrName) {
275     
276     this.findByUuidOrName = theFindByUuidOrName;
277     
278     return this;
279   }
280 
281   /**
282    * find the attributeDef
283    * @return the attributeDef or null
284    */
285   public AttributeDef findAttribute() {
286     Set<AttributeDef> attributeDefs = this.findAttributes();
287     
288     return GrouperUtil.setPopOne(attributeDefs);
289   }
