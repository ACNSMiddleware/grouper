20  package edu.internet2.middleware.subject.provider;
21  
22  import java.sql.SQLException;
23  import java.util.Properties;
24  
25  import edu.internet2.middleware.subject.SourceUnavailableException;
26  
27  
28  /**
29   * gives connections and allows returns.  e.b. dbcp pooling, c3p0 pooling,
30   * grouper built in db
31   */
32  public interface JdbcConnectionProvider {
33  
34    /**
35     * if the provider requires jdbc data in the sources.xml
36     * @return true if provider requires data in sources.xml
37     */
38    public boolean requiresJdbcConfigInSourcesXml();
39    
40    /** 
41     * init the pool 
42     * @param properties 
43     * @param sourceId mainly for logging
44     * @param driver driver
45     * @param maxActive max active connection if pool
46     * @param defaultMaxActive if max active is needed, and blank, then use this
47     * @param maxIdle max idle connections if pool
48     * @param defaultMaxIdle if max is needed, and blank, then use this
49     * @param maxWaitSeconds max wait if pool empty in seconds.  -1 means long time
50     * @param defaultMaxWaitSeconds if max wait is needed, and blank, then use this, -1 mean long time
51     * @param dbUrl jdbc url of the database
52     * @param dbUser user to login to the database
53     * @param dbPassword password for the database (unencrypted)
54     * @param readOnly if conn should be readonly
55     * @param readOnlyDefault default if not specified and needed
56     * @throws SourceUnavailableException if something is not right or not available
57     */
58    public void init(Properties properties, String sourceId, String driver, Integer maxActive, int defaultMaxActive, Integer maxIdle, int defaultMaxIdle,
59        Integer maxWaitSeconds, int defaultMaxWaitSeconds, String dbUrl, String dbUser, 
60        String dbPassword, Boolean readOnly, boolean readOnlyDefault) throws SourceUnavailableException;
61    
62    /**
63     * get a connection (dont close this when done, just call "doneWithConnection()"
64     * @return the connection
65     * @throws SQLException if there is a problem
66     */
67    public JdbcConnectionBean connectionBean() throws SQLException;
68    
69  }
