50  /**
51   * Decorator that provides <i>Wheel</i> privilege resolution for {@link NamingResolver}.
52   * <p/>
53   * @author  blair christensen.
54   * @version $Id: WheelNamingResolver.java,v 1.20 2009-09-21 06:14:26 mchyzer Exp $
55   * @since   1.2.1
56   */
57  public class WheelNamingResolver extends NamingResolverDecorator {
58  
59    /**
60     * @see edu.internet2.middleware.grouper.privs.NamingResolver#flushCache()
61     */
62    public void flushCache() {
63      WheelCache.flush();
64      super.getDecoratedResolver().flushCache();
65    }
66  
67    // TODO 20070820 DRY w/ access resolution
68  
69    /** if use wheel group */
70    private boolean useWheel = false;
71  
72    /** wheel group */
73    private Group wheelGroup;
74  
75    /** wheel session */
76    private GrouperSession wheelSession = null;
77  
78    /** logger */
79    private static final Log LOG = GrouperUtil.getLog(WheelNamingResolver.class);
80  
81    /**
82     * @param resolver 
83     * @since   1.2.1
84     */
85    public WheelNamingResolver(NamingResolver resolver) {
86      super(resolver);
87  
88  
89      // TODO 20070816 this is ugly
90      this.useWheel = Boolean.valueOf(
91          GrouperConfig.retrieveConfig().propertyValueString(GrouperConfig.PROP_USE_WHEEL_GROUP)).booleanValue();
92      // TODO 20070816 and this is even worse
93      if (this.useWheel) {
94        String wheelGroupName = "";
95        try {
96          wheelGroupName = GrouperConfig.retrieveConfig().propertyValueString(GrouperConfig.PROP_WHEEL_GROUP);
97          this.wheelSession = GrouperSession.start(SubjectFinder.findRootSubject(), false);
98          this.wheelGroup = GroupFinder.findByName(
99                      //dont replace the current grouper session
100             this.wheelSession,
101             wheelGroupName, true
102             );
103       } catch (Exception e) {
104 
105         //OK, so wheel group does not exist. Not fatal...
106         String error = "Cant find wheel group '" + wheelGroupName + "': "
107             + e.getClass().getSimpleName();
108 
109         if (!loggedWheelNotThere && !GrouperCheckConfig.inCheckConfig) {
110           LOG.error(error);
111           loggedWheelNotThere = true;
112         }
113         //exception stack is not that interesting
114         LOG.debug(error, e);
115         this.useWheel = false;
116       }
117     }
118   }
119 
120   /** */
121   private static boolean loggedWheelNotThere = false;
122 
123   /**
124    * Put boolean into cache for <code>isWheelMember(...)</code>.
125    * @param subj 
126    * @param rv 
127    * @since   1.2.1
128    */
129   private void putInHasPrivilegeCache(Subject subj, Boolean rv) {
130     WheelCache.putInHasPrivilegeCache(subj, rv);
131   }
132 
133   /**
134    * Put boolean into cache for <code>isWheelMember(...)</code>.
135    * @param subj 
136    * @return  if wheel member
137    * @since   1.2.1
138    */
139   private boolean isWheelMember(final Subject subj) {
140     Boolean rv = getFromIsWheelMemberCache(subj);
141     if (rv == null) {
142 
143       rv = (Boolean) GrouperSession.callbackGrouperSession(this.wheelSession,
144           new GrouperSessionHandler() {
145 
146             public Object callback(GrouperSession grouperSession)
147                 throws GrouperSessionException {
148               return WheelNamingResolver.this.wheelGroup.hasMember(subj);
149             }
150                     });
151 
152       putInHasPrivilegeCache(subj, rv);
153     }
154     return rv;
155   }
156 
157   /**
158    * @see     NamingResolver#hasPrivilege(Stem, Subject, Privilege)
159    * @since   1.2.1
160    */
161   public boolean hasPrivilege(Stem stem, final Subject subject, Privilege privilege)
162       throws IllegalArgumentException {
163     if (this.isAndUseWheel(subject)) {
164       return true;
165     }
166     return super.getDecoratedResolver().hasPrivilege(stem, subject, privilege);
167   }
168 
169   /**
170    * 
171    * @param subject
172    * @return true if this is wheel and if using wheel
173    */
174   private boolean isAndUseWheel(Subject subject) {
175     if (this.getGrouperSession().isConsiderIfWheelMember()) {
176       if (this.useWheel) {
177         if (isWheelMember(subject)) {
178           return true;
179         }
180       }
181     }
182     return false;
183   }
184 
185   /**
186    * @see edu.internet2.middleware.grouper.privs.NamingResolver#hqlFilterStemsWhereClause(edu.internet2.middleware.subject.Subject, edu.internet2.middleware.grouper.hibernate.HqlQuery, java.lang.StringBuilder, java.lang.String, java.util.Set)
187    */
188   public boolean hqlFilterStemsWhereClause(Subject subject, HqlQuery hqlQuery,
189       StringBuilder hql, String stemColumn, Set<Privilege> privInSet) {
190     //Wheel can see all stems
191     if (this.isAndUseWheel(subject)) {
192       return false;
193     }
194     NamingResolver decoratedResolver = super.getDecoratedResolver();
195     //CachingNamingResolver
196     return decoratedResolver.hqlFilterStemsWhereClause(subject, hqlQuery, hql,
197         stemColumn, privInSet);
198   }
199 
200   /**
201    * @see edu.internet2.middleware.grouper.privs.NamingResolver#postHqlFilterStems(java.util.Set, edu.internet2.middleware.subject.Subject, java.util.Set)
202    */
203   public Set<Stem> postHqlFilterStems(Set<Stem> stems, Subject subject,
204       Set<Privilege> privInSet) {
205 
206     //Wheel can see all groups
207     if (this.isAndUseWheel(subject)) {
208       return stems;
209     }
210     Set<Stem> filteredStems = super.getDecoratedResolver().postHqlFilterStems(stems,
211         subject, privInSet);
212 
213     //return filtered groups
214     return filteredStems;
215   }
216 
217   /**
218    * Retrieve boolean from cache for <code>isWheelMember(...)</code>.
219    * @param subj 
220    * @return  Cached return value or null.
221    * @since   1.2.1
222    */
223   private Boolean getFromIsWheelMemberCache(Subject subj) {
224     return WheelCache.getFromIsWheelMemberCache(subj);
225   }
226 
227   /**
228    * 
229    * @see edu.internet2.middleware.grouper.privs.NamingResolver#hqlFilterStemsNotWithPrivWhereClause(edu.internet2.middleware.subject.Subject, edu.internet2.middleware.grouper.hibernate.HqlQuery, java.lang.StringBuilder, java.lang.String, Privilege, boolean)
230    */
231   public boolean hqlFilterStemsNotWithPrivWhereClause(Subject subject, HqlQuery hqlQuery,
232       StringBuilder hql, String stemColumn, Privilege privilege, boolean considerAllSubject) {
233 
234     //Wheel can see all groups
235     if (this.isAndUseWheel(subject)) {
236       return false;
237     }
238     NamingResolver decoratedResolver = super.getDecoratedResolver();
239     //System.out.println(decoratedResolver.getClass().getName());
240     //CachingAccessResolver
241     return decoratedResolver.hqlFilterStemsNotWithPrivWhereClause(subject, hqlQuery, hql,
242         stemColumn, privilege, considerAllSubject);
243   }
