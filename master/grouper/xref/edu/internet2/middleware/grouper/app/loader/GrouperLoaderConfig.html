165   /**
166    * get a profile by name.  if "grouper" then get the hibernate db connection
167    * specify the db connection with user, pass, url, and driver class
168    * the string after "db." is the name of the connection, and it should not have
169    * spaces or other special chars in it
170    * db.warehouse.user = mylogin
171    * db.warehouse.pass = secret
172    * db.warehouse.url = jdbc:mysql://localhost:3306/grouper
173    * db.warehouse.driver = com.mysql.jdbc.Driver
174    * @param name
175    * @return the db
176    */
177   public static GrouperLoaderDb retrieveDbProfile(String name) {
178     
179     String pass = null;
180     String url = null;
181     String driver = null;
182     boolean isGrouper = StringUtils.equals(name, GROUPER_DB_NAME);
183     String user = null;
184 
185     if (isGrouper && StringUtils.isNotBlank(GrouperLoaderConfig.retrieveConfig().propertyValueString("db." + name + ".user"))) {
186       LOG.error("Cant have a database named 'grouper' in " +
187           "the grouper-loader.properties.  This is a special name for the " +
188           "grouper.hibernate.properties database");
189     }
190     
191     if (isGrouper) {
192       
193       //the name "hibernate" is a special term, which could be in the grouper-loader.properties, 
194       //but defaults to grouper.hibernate.properties
195       Properties properties = GrouperHibernateConfig.retrieveConfig().properties();
196       
197       user = properties.getProperty("hibernate.connection.username");
198       pass = properties.getProperty("hibernate.connection.password");
199       url = properties.getProperty("hibernate.connection.url");
200       driver = properties.getProperty("hibernate.connection.driver_class");
201     } else {      
202       
203       //first look in grouper-loader.properties:
204       user = GrouperLoaderConfig.retrieveConfig().propertyValueString("db." + name + ".user");
205       if (!StringUtils.isBlank(user)) {
206         pass = GrouperLoaderConfig.retrieveConfig().propertyValueString("db." + name + ".pass");
207         url = GrouperLoaderConfig.retrieveConfig().propertyValueString("db." + name + ".url");
208         driver = GrouperLoaderConfig.retrieveConfig().propertyValueString("db." + name + ".driver");
209       
210       } else {
211         throw new RuntimeException("Cant find the db connection named: '" + name + "' in " +
212         		"the grouper-loader.properties.  Should have entries: db." + name + ".user, db." + name 
213         		+ ".pass, db." + name + ".url, db." + name + ".driver");
214       }
215     }
216     //might be in external file
217     pass = Morph.decryptIfFile(pass);
218     GrouperLoaderDb grouperLoaderDb = new GrouperLoaderDb(user, pass, url, driver);
219     return grouperLoaderDb;
220   }

222   /**
223    * get a profile by name from grouper-loader.properties
224    * specify the ldap connection with user, pass, url, etc
225    * the string after "ldap." is the name of the connection, and it should not have
226    * spaces or other special chars in it
227    * ldap.personLdap.user
228    * ldap.personLdap.pass
229    * ldap.personLdap.url
230    * @param name
231    * @return the db
232    */
233   public static GrouperLoaderLdapServer retrieveLdapProfile(String name) {
234     
235     GrouperLoaderLdapServer grouperLoaderLdapServer = new GrouperLoaderLdapServer();
236 
237     grouperLoaderLdapServer.setConfigFileFromClasspath(getPropertyString("ldap." + name + ".configFileFromClasspath"));
238 
239     {
240       //#note the URL should start with ldap: or ldaps: if it is SSL.  
241       //#It should contain the server and port (optional if not default), and baseDn, 
242       //#e.g. ldaps://ldapserver.school.edu:636/dc=school,dc=edu
243       //#ldap.personLdap.url = ldaps://ldapserver.school.edu:636/dc=school,dc=edu
244       String url = GrouperLoaderConfig.retrieveConfig().propertyValueString("ldap." + name + ".url");
245       
246       if (StringUtils.isBlank(url) && StringUtils.isBlank(grouperLoaderLdapServer.getConfigFileFromClasspath())) {
247         throw new RuntimeException("Cant find the ldap connection named: '" + name + "' in " +
248         		"the grouper-loader.properties.  Should have entry: ldap." + name + ".url or ldap." + name + ".configFileFromClasspath");
249       }
250       
251       grouperLoaderLdapServer.setUrl(url);
252     }
253     
254     {
255       String user = GrouperLoaderConfig.retrieveConfig().propertyValueString("ldap." + name + ".user");
256       //#ldap.personLdap.user = uid=someapp,ou=people,dc=myschool,dc=edu
257       if (!StringUtils.isBlank(user)) {
258         grouperLoaderLdapServer.setUser(user);
259       }
260     }
261     
262     {
263       String pass = GrouperLoaderConfig.retrieveConfig().propertyValueString("ldap." + name + ".pass");
264       if (!StringUtils.isBlank(pass)) {
265         //might be in external file
266         pass = Morph.decryptIfFile(pass);
267         //#note the password can be stored encrypted in an external file
268         //#ldap.personLdap.pass = secret
269         grouperLoaderLdapServer.setPass(pass);
270       }
271     }
272 
273     
274     //#optional, if you are using tls, set this to TRUE.  Generally you will not be using an SSL URL to use TLS...
275     //#ldap.personLdap.tls = true
276     grouperLoaderLdapServer.setTls(getPropertyBoolean("ldap." + name + ".tls", false));
277     
278     //#optional, if using sasl
279     //#ldap.personLdap.saslAuthorizationId = 
280     //#ldap.personLdap.saslRealm = 
281     grouperLoaderLdapServer.setSaslAuthorizationId(GrouperLoaderConfig.retrieveConfig().propertyValueString("ldap." + name + ".saslAuthorizationId"));
282     grouperLoaderLdapServer.setSaslRealm(GrouperLoaderConfig.retrieveConfig().propertyValueString("ldap." + name + ".saslRealm"));
283     
284     //#ldap.personLdap.batchSize = 
285     grouperLoaderLdapServer.setBatchSize(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".batchSize", -1));
286         
287     //#ldap.personLdap.countLimit = 
288     grouperLoaderLdapServer.setCountLimit(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".countLimit", -1));
289     
290     grouperLoaderLdapServer.setTimeLimit(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".timeLimit", -1));
291 
292     grouperLoaderLdapServer.setTimeout(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".timeout", -1));
293 
294     grouperLoaderLdapServer.setMinPoolSize(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".minPoolSize", -1));
295 
296     grouperLoaderLdapServer.setMaxPoolSize(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".maxPoolSize", -1));
297 
298     grouperLoaderLdapServer.setValidateOnCheckIn(getPropertyBoolean("ldap." + name + ".validateOnCheckIn", false));
299     grouperLoaderLdapServer.setValidateOnCheckOut(getPropertyBoolean("ldap." + name + ".validateOnCheckOut", false));
300     grouperLoaderLdapServer.setValidatePeriodically(getPropertyBoolean("ldap." + name + ".validatePeriodically", false));
301 
302     grouperLoaderLdapServer.setPagedResultsSize(getPropertyInt("ldap." + name + ".pagedResultsSize", -1));
303 
304     grouperLoaderLdapServer.setReferral(getPropertyString("ldap." + name + ".referral"));
305 
306     //#validateOnCheckout defaults to true if all other validate methods are false
307     if (!grouperLoaderLdapServer.isValidateOnCheckIn() && !grouperLoaderLdapServer.isValidateOnCheckOut() && !grouperLoaderLdapServer.isValidatePeriodically()) {
308       grouperLoaderLdapServer.setValidateOnCheckOut(true);
309     }
310     
311     //if there is a validation requested, set up the validation class with the ldap factory
312     if (grouperLoaderLdapServer.isValidateOnCheckIn() || grouperLoaderLdapServer.isValidateOnCheckOut() || grouperLoaderLdapServer.isValidatePeriodically()) {
313       LdapValidator<Ldap> validator = null;
314 
315       String ldapValidator = getPropertyString("ldap." + name + ".validator");
316 
317       if (StringUtils.equalsIgnoreCase(ldapValidator, CompareLdapValidator.class.getSimpleName())) {
318         String validationDn = getPropertyString("ldap." + name + ".validatorCompareDn");
319         String validationSearchFilterString = getPropertyString("ldap." + name + ".validatorCompareSearchFilterString");
320         validator = new CompareLdapValidator(validationDn, new SearchFilter(validationSearchFilterString)); // perform a simple compare
321       } else if (StringUtils.equalsIgnoreCase(ldapValidator, ConnectLdapValidator.class.getSimpleName())) {
322         //this is the default, why not
323         validator = new ConnectLdapValidator(); // perform a simple connect
324       } else if (!StringUtils.isBlank(ldapValidator)) {
325         //get the class
326         Class<LdapValidator<Ldap>> validatorClass = GrouperUtil.forName(ldapValidator);
327         validator = GrouperUtil.newInstance(validatorClass);
328       }
329 
330       grouperLoaderLdapServer.setValidator(validator);
331     }
332 
333     //#ldap.personLdap.validateTimerPeriod = 
334     grouperLoaderLdapServer.setValidateTimerPeriod(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".validateTimerPeriod", -1));
335     
336     //#ldap.personLdap.pruneTimerPeriod = 
337     grouperLoaderLdapServer.setPruneTimerPeriod(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".pruneTimerPeriod", -1));
338 
339     //#if connections expire after a certain amount of time, this is it, in millis, defaults to 300000 (5 minutes)
340     //#ldap.personLdap.expirationTime = 
341     grouperLoaderLdapServer.setExpirationTime(GrouperLoaderConfig.retrieveConfig().propertyValueInt("ldap." + name + ".expirationTime", -1));
342     
343     if (LOG.isDebugEnabled()) {
344       LOG.debug("LDAP config for server id: " + name + ": " + grouperLoaderLdapServer);
345     }
346     
347     return grouperLoaderLdapServer;
348   }
