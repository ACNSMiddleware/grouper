85    protected void setUp(String jdbcConnectionProviderString) {
86      this.source = new JDBCSourceAdapter("jdbc", "JDBC Subject Source");
87      //note, this might be null
88      if (!StringUtils.isBlank(jdbcConnectionProviderString)) {
89        this.source.addInitParam("jdbcConnectionProvider", jdbcConnectionProviderString);
90      }
91      this.source.addInitParam("maxActive", "4");
92      this.source.addInitParam("maxIdle", "4");
93      this.source.addInitParam("maxWait", "30");
94      this.source.addInitParam("dbDriver", "org.hsqldb.jdbcDriver");
95      this.source.addInitParam("dbUrl", "jdbc:hsqldb:file:testDB/hsqldb/subject");
96      this.source.addInitParam("dbUser", "sa");
97      this.source.addInitParam("dbPwd", "");
98      this.source.addInitParam("SubjectID_AttributeType", "id");
99      this.source.addInitParam("Name_AttributeType", "name");
100     this.source.addInitParam("Description_AttributeType", "description");
101     SourceManager.getInstance().sourceMap.put("jdbc", this.source);
102 
103     Search search = null;
104 
105     search = new Search();
106     search.setSearchType("searchSubject");
107     search
108         .addParam(
109             "sql",
110             "select subject.subjectid as id, subject.name as name, lfnamet.lfname as "
111                 + "lfname, loginidt.loginid as loginid, desct.description as description from subject "
112                 + "left join (select subjectid, value as lfname from subjectattribute where name = 'name') "
113                 + "as lfnamet on subject.subjectid = lfnamet.subjectid left join (select subjectid, value "
114                 + "as loginid from subjectattribute where name = 'loginid') as loginidt on subject.subjectid "
115                 + "= loginidt.subjectid left join (select subjectid, value as description from subjectattribute "
116                 + "where name = 'description') as desct on subject.subjectid = desct.subjectid where subject.subjectid = ?");
117     this.source.loadSearch(search);
118 
119     search = new Search();
120     search.setSearchType("searchSubjectByIdentifier");
121     search.addParam("numParameters", "1");
122     search
123         .addParam(
124             "sql",
125             "select subject.subjectid as id, subject.name as name, lfnamet.lfname as "
126                 + "lfname, loginidt.loginid as loginid, desct.description as description from subject "
127                 + "left join (select subjectid, value as lfname from subjectattribute where name = 'name') "
128                 + "as lfnamet on subject.subjectid = lfnamet.subjectid left join (select subjectid, value "
129                 + "as loginid from subjectattribute where name = 'loginid') as loginidt on subject.subjectid "
130                 + "= loginidt.subjectid left join (select subjectid, value as description from "
131                 + "subjectattribute where name = 'description') as desct on subject.subjectid = "
132                 + "desct.subjectid where loginidt.loginid = ?");
133     this.source.loadSearch(search);
134 
135     search = new Search();
136     search.setSearchType("search");
137     search
138         .addParam(
139             "sql",
140             "select subject.subjectid as id, subject.name as name, lfnamet.lfname as "
141                 + "lfname, loginidt.loginid as loginid, desct.description as description from subject "
142                 + "left join (select subjectid, value as lfname from subjectattribute where name = 'name') "
143                 + "as lfnamet on subject.subjectid = lfnamet.subjectid left join (select subjectid, value "
144                 + "as loginid from subjectattribute where name = 'loginid') as loginidt on subject.subjectid "
145                 + "= loginidt.subjectid left join (select subjectid, value as description from "
146                 + "subjectattribute where name = 'description') as desct on subject.subjectid = "
147                 + "desct.subjectid where (lower(name) like '%'||?||'%') or (lower(lfnamet.lfname) "
148                 + "like '%'||?||'%') or (lower(loginidt.loginid) like '%'||?||'%') or "
149                 + "(lower(desct.description) like '%'||?||'%')");
150     this.source.loadSearch(search);
151     this.source.addSubjectType(SubjectTypeEnum.PERSON.getName());

187   public void testColumnAliases() {
188     
189     List<String> aliases = JdbcDatabaseType.columnAliases("select a, c as b, this as that");
190     
191     assertEquals(3, aliases.size());
192     assertEquals("a", aliases.get(0));
193     assertEquals("b", aliases.get(1));
194     assertEquals("that", aliases.get(2));
195     
196     assertEquals("select * ", JdbcDatabaseType.selectPart("select * from whatever"));
197     assertNull(JdbcDatabaseType.selectPart("select *, this, that"));
198     aliases = JdbcDatabaseType.columnAliases("select\n" +
199         "   s.subjectid as id, s.name as name,\n" +
200         "   (select sa2.value from subjectattribute sa2 where name='name' and sa2.SUBJECTID = s.subjectid) as lfname,\n" +
201         "   (select sa3.value from subjectattribute sa3 where name='loginid' and sa3.SUBJECTID = s.subjectid) as loginid,\n" +
202         "   (select sa4.value from subjectattribute sa4 where name='description' and sa4.SUBJECTID = s.subjectid) as description,\n" +
203         "   (select sa5.value from subjectattribute sa5 where name='email' and sa5.SUBJECTID = s.subjectid) as email\n");
204     
205     assertEquals(6, aliases.size());
206     assertEquals("id", aliases.get(0));
207     assertEquals("name", aliases.get(1));
208     assertEquals("lfname", aliases.get(2));
209     assertEquals("loginid", aliases.get(3));
210     assertEquals("description", aliases.get(4));
211     assertEquals("email", aliases.get(5));

332   /**
333    * test search on all pooling strategies
334    * @throws Throwable 
335    */
336   public void testGenericSearchThread() throws Throwable {
337     String[] jdbcConnectionProviders = new String[] {
338         // DbcpJdbcConnectionProvider.class.getName(),
339         C3p0JdbcConnectionProvider.class.getName() };
340     int numberOfThreads = 50;
341     Thread[] threads = new Thread[numberOfThreads];
342     final Throwable[] throwables = new Throwable[1];
343     for (final String jdbcConnectionProvider : jdbcConnectionProviders) {
344 
345       try {
346         this.setUp(jdbcConnectionProvider);
347         //test pooling
348         for (int i = 0; i < numberOfThreads; i++) {
349           threads[i] = new Thread(new Runnable() {
350 
351             public void run() {
352               try {
353                 JDBCSourceAdapterTest.this.genericSearchHelper(jdbcConnectionProvider);
354               } catch (Throwable t) {
355                 throwables[0] = t;
356               }
357             }
358 
359           });
360           threads[i].start();
361 
362         }
363         //wait for all the threads
364         for (int i = 0; i < numberOfThreads; i++) {
365           threads[i].join();
366         }
367         //see if any had exceptions
368         if (throwables[0] != null) {
369           throw throwables[0];
370         }
371       } catch (Throwable re) {
372         SubjectUtils.injectInException(re, "Problem with jdbcConnectionProvider: "
373             + jdbcConnectionProvider);
374         throw re;
375       }
376     }
377   }

434   /**
435    * A test of Subject search capability.
436    * @param jdbcConnectionProvider 
437    */
438   public void genericSearchHelper(String jdbcConnectionProvider) {
439     Set<Subject> set = null;
440     Subject subject = null;
441 
442     // In the test subject database, IDs are not included in generic search.
443     set = this.source.search("1012");
444     assertEquals("Searching value = 1012, result size: " + jdbcConnectionProvider, 0, set
445         .size());
446 
447     set = this.source.search("babl");
448     assertEquals("Searching value = babl, result size: " + jdbcConnectionProvider, 1, set
449         .size());
450     subject = set.toArray(new Subject[0])[0];
451     assertEquals("Searching value = babl: " + jdbcConnectionProvider, "1012", subject
452         .getId());
453 
454     set = this.source.search("barry");
455     assertEquals("Searching value = barry, result size: " + jdbcConnectionProvider, 12,
456         set.size());
457 
458     set = this.source.search("beth%porter");
459     assertEquals("Searching value = beth%porter, result size: " + jdbcConnectionProvider,
460         1, set.size());
461     subject = set.toArray(new Subject[0])[0];
462     assertEquals("Searching value = beth%porter, id: " + jdbcConnectionProvider, "1119",
463         subject.getId());
464   }
